17/**
18 * The enumeration type that provides additional information about the dxFeed Graal C++-API event type.
19 * Useful when used as a subscription parameter.
20 */
21class DXFCPP_EXPORT EventTypeEnum {
22 /// The dxFeed Graal Native C-API event class id corresponding to the current enum element.
23 const std::uint32_t id_;
24
25 const std::string name_;
26
27 const std::string className_;
28
29 // A flag that indicates that the current enum element is characterizing the Lasting (TICKER) event.
30 const bool isLasting_;
31
32 // A flag that indicates that the current enum element is characterizing the Indexed event.
33 const bool isIndexed_;
34
35 // A flag that indicates that the current enum element is characterizing the TimeSeries (HISTORY) event.
36 const bool isTimeSeries_;
37
38 // A flag that indicates that the current enum element is characterizing the OnlyIndexed (Indexed, but not
39 // TimeSeries) event.
40 const bool isOnlyIndexed_;
41
42 // A flag that indicates that the current enum element is characterizing the Market event.
43 const bool isMarket_;
44
45 EventTypeEnum(std::uint32_t id, std::string name, std::string className, bool isLasting, bool isIndexed = false,
46 bool isTimeSeries = false, bool isMarket = true) noexcept
47 : id_{id}, name_{std::move(name)}, className_{std::move(className)}, isLasting_{isLasting},
48 isIndexed_{isIndexed || isTimeSeries}, isTimeSeries_{isTimeSeries},
