19/**
20 * <h3><a name="eventFlagsSection">Event flags, transactions and snapshots</a></h3>
21 *
22 * Some indexed event sources provide a consistent view of a set of events for a given symbol. Their updates
23 * may incorporate multiple changes that have to be processed at the same time.
24 * The corresponding information is carried in @ref IndexedEvent::getEventFlags() "eventFlags" property.
25 *
26 * <p> The value of @ref IndexedEvent::getEventFlags() "eventFlags" property has several significant bits that are
27 * packed into an integer in the following way:
28 *
29 * <pre><tt>
30 * 31..7 6 5 4 3 2 1 0
31 * +---------+----+----+----+----+----+----+----+
32 * | | SM | | SS | SE | SB | RE | TX |
33 * +---------+----+----+----+----+----+----+----+
34 * </tt></pre>
35 *
36 * Each source updates its transactional state using these bits separately.
37 * The state of each source has to be tracked separately in a map for each source.
38 * However, event @ref IndexedEvent::getIndex() "index" is unique across the sources. This is achieved by allocating
39 * an event-specific number of most significant bits of @ref IndexedEvent::getIndex() "index" for use as
40 * a @ref IndexedEvent::getSource() "source" @ref IndexedEventSource::id() "id".
41 *
42 * <p> `TX` (bit 0) &mdash; ::TX_PENDING is an indicator of pending transactional update.
43 * It can be retrieved from `eventFlags` with the following piece of code:
44 *
45 * ```cpp
46 * bool txPending = (event-> getEventFlags() & IndexedEvent::TX_PENDING) != 0;
47 * ```
48 *
49 * <p>When `txPending` is `true` it means, that an ongoing transaction update that spans multiple events is
50 * in process. All events with `txPending` `true` shall be put into a separate <em>pending list</em>
51 * for each source id and should be processed later when an event for this source id with `txPending` `false` comes.
52 *
53 * <p> `RE` (bit 1) &mdash; ::REMOVE_EVENT is used to indicate that that the event with the
54 * corresponding index has to be removed.
55 *
56 * ```cpp
57 * bool removeEvent = (event-> getEventFlags() & IndexedEvent::REMOVE_EVENT) != 0;
58 * ```
59 *
60 * <p> `SB` (bit 2) &mdash; ::SNAPSHOT_BEGIN is used to indicate when the loading of a snapshot starts.
61 *
62 * ```cpp
63 * bool snapshotBegin = (event-> getEventFlags() & IndexedEvent::SNAPSHOT_BEGIN) != 0;
64 * ```
65 *
66 * <p> Snapshot load starts on new subscription and the first indexed event that arrives for each non-zero source id
67 * on new subscription may have `snapshotBegin` set to `true`. It means, that an ongoing snapshot
68 * consisting of multiple events is incoming. All events for this source id shall be put into a separate
69 * <em>pending list</em> for each source id.
70 *
71 * <p> `SE` (bit 3) &mdash; ::SNAPSHOT_END or `SS` (bit 4) &mdash; ::SNAPSHOT_SNIP are used to indicate the end of a
72 * snapshot.
73 *
74 * ```cpp
75 * bool snapshotEnd = (event-> getEventFlags() & IndexedEvent::SNAPSHOT_END) != 0;
76 * bool snapshotSnip = (event-> getEventFlags() & IndexedEvent::SNAPSHOT_SNIP) != 0;
77 * ```
78 *
79 * <p>The last event of a snapshot is marked with either `snapshotEnd` or `snapshotSnip`. At this time, all events
80 * from a <em>pending list</em> for the corresponding source can be processed, unless `txPending` is also
81 * set to `true`. In the later case, the processing shall be further delayed due to ongoing transaction.
82 *
83 * <p>The difference between `snapshotEnd` and `snapshotSnip` is the following.
84 * `snapshotEnd` indicates that the data source had sent all the data pertaining to the subscription
85 * for the corresponding indexed event, while `snapshotSnip` indicates that some limit on the amount
86 * of data was reached and while there still might be more data available, it will not be provided.
87 *
88 * <p> `SM` (bit 6) &mdash; ::SNAPSHOT_MODE is used to instruct dxFeed to use snapshot mode.
89 * It is intended to be used only for publishing to activate (if not yet activated) snapshot mode.
90 * The difference from ::SNAPSHOT_BEGIN flag is that ::SNAPSHOT_MODE only switches on snapshot mode
91 * without starting snapshot synchronization protocol.
92 *
93 * <p> When a snapshot is empty or consists of a single event, then the event can have both `snapshotBegin`
94 * and `snapshotEnd` or `snapshotSnip` flags. In case of an empty snapshot, `removeEvent` on this event is
95 * also set to `true`.
96 */
97class DXFCPP_EXPORT EventFlag final {
98 std::uint32_t flag_;
99 std::string name_;
100
101 EventFlag(std::uint32_t flag, std::string name) : flag_{flag}, name_{std::move(name)} {}
102
103 public:
104 /**
105 * `0x01` - A bitmask to get transaction pending indicator from the value of @ref ::getEventFlags() "eventFlags"
106 * property.
107 *
108 * ```cpp
109 * bool txPending = (event-> getEventFlags() & IndexedEvent::TX_PENDING) != 0;
110 * ```
111 *
112 * See <a href="./de/d03/classdxfcpp_1_1_event_flag.html#eventFlagsSection">"Event Flags" section</a>
113 */
114 static const EventFlag TX_PENDING;
115
116 /**
117 * `0x02` - A bitmask to get removal indicator from the value of @ref ::getEventFlags() "eventFlags" property.
118 *
119 * ```cpp
120 * bool removeEvent = (event-> getEventFlags() & IndexedEvent::REMOVE_EVENT) != 0;
121 * ```
122 *
123 * See <a href="#eventFlagsSection">"Event Flags" section</a>
124 */
125 static const EventFlag REMOVE_EVENT;
126
127 /**
128 * `0x04` - A bitmask to get snapshot begin indicator from the value of @ref ::getEventFlags() "eventFlags"
129 * property.
130 *
131 * ```cpp
132 * bool snapshotBegin = (event-> getEventFlags() & IndexedEvent::SNAPSHOT_BEGIN) != 0;
133 * ```
134 *
135 * See <a href="./de/d03/classdxfcpp_1_1_event_flag.html#eventFlagsSection">"Event Flags" section</a>
136 */
137 static const EventFlag SNAPSHOT_BEGIN;
138
139 /**
140 * `0x08` - A bitmask to get snapshot end indicator from the value of @ref ::getEventFlags() "eventFlags" property.
141 *
142 * ```cpp
143 * bool snapshotEnd = (event-> getEventFlags() & IndexedEvent::SNAPSHOT_END) != 0;
144 * ```
145 *
146 * See <a href="./de/d03/classdxfcpp_1_1_event_flag.html#eventFlagsSection">"Event Flags" section</a>
147 */
148 static const EventFlag SNAPSHOT_END;
149
150 /**
151 * `0x10` - A bitmask to get snapshot snip indicator from the value of @ref ::getEventFlags() "eventFlags" property.
152 *
153 * ```cpp
154 * bool snapshotSnip = (event-> getEventFlags() & IndexedEvent::SNAPSHOT_SNIP) != 0;
155 * ```
156 *
157 * See <a href="./de/d03/classdxfcpp_1_1_event_flag.html#eventFlagsSection">"Event Flags" section</a>
158 */
159 static const EventFlag SNAPSHOT_SNIP;
160
161 // 0x20 is reserved. This flag will fit into 1-byte on the wire in QTP protocol
162
163 /**
164 * `0x40` - A bitmask to set snapshot mode indicator into the value of @ref ::setEventFlags() "eventFlags" property.
165 * This flag is intended for publishing only.
166 *
167 * See <a href="./de/d03/classdxfcpp_1_1_event_flag.html#eventFlagsSection">"Event Flags" section</a>
168 */
169 static const EventFlag SNAPSHOT_MODE;
170
171 /**
172 * `0x80` - For internal use. Marks a subscription for deletion.
173 */
174 static const EventFlag REMOVE_SYMBOL;
175
176 /**
177 * Creates the invalid event flag
178 */
179 explicit EventFlag() : flag_{unsigned(-1)}, name_{"INVALID"} {}
180
181 /**
182 * @return The event flag's value
183 */
184 std::uint32_t getFlag() const { return flag_; }
185
186 /**
187 * Determines if the given flag is in the mask.
188 *
189 * @param eventFlagsMask The event flags mask
190 *
191 * @return `true` the given flag is in the mask.
192 */
193 bool in(std::uint32_t eventFlagsMask) const { return (eventFlagsMask & flag_) != 0; }
194
195 /**
196 * Determines if the given flag is in the mask.
197 *
198 * @tparam EventFlagsMask An event flags mask type that satisfies the condition: there is a `getMask` method that
199 * returns std::uint32_t
200 * @param eventFlagsMask The event flags mask.
201 * @return `true` the given flag is in the mask.
202 */
203 template <typename EventFlagsMask>
204 bool in(const EventFlagsMask &eventFlagsMask) const
205#if __cpp_concepts
206 requires requires {
207 { eventFlagsMask.getMask() } -> std::same_as<std::uint32_t>;
208 }
209#endif
210 {
211 return in(eventFlagsMask.getMask());
212 }
213
214 ///
215 [[nodiscard]] const std::string &getName() const { return name_; }
216
217 ///
218 std::string toString() const { return name_; }
219};

230///
231class EventFlagsMask final {
232 std::uint32_t mask_;
233
234 public:
235 /**
236 * Creates an empty event flags mask
237 */
238 explicit EventFlagsMask() : mask_{0u} {}
239
240 /**
241 * Create event flags mask by integer value
242 *
243 * @tparam MaskType The type of integer mask
244 * @param mask The integer mask value
245 */
246 template <Integral MaskType> explicit EventFlagsMask(MaskType mask) : mask_{static_cast<std::uint32_t>(mask)} {}
247
248 /**
249 * Creates event flags mask by iterators of container with flags
250 *
251 * @tparam EventFlagIt The iterator type
252 * @param begin The start position
253 * @param end The end position
254 */
255 template <typename EventFlagIt> EventFlagsMask(EventFlagIt begin, EventFlagIt end) {
256 mask_ =
257 std::accumulate(begin, end, 0u, [](unsigned mask, const EventFlag &flag) { return mask | flag.getFlag(); });
258 }
259
260 /**
261 * Creates event flags mask by initializer list with flags\
262 *
263 * @param eventTypes The list with flags
264 */
265 EventFlagsMask(std::initializer_list<EventFlag> eventTypes)
266 : EventFlagsMask(eventTypes.begin(), eventTypes.end()) {}
267
268 /**
269 * Returns an integer representation of event mask
270 *
271 * @return an integer representation
272 */
273 constexpr std::uint32_t getMask() const { return mask_; }
274
275 /**
276 *
277 * @param eventTypesMask
278 * @param eventType
279 * @return
280 */
281 friend EventFlagsMask operator|(const EventFlagsMask &eventTypesMask, const EventFlag &eventType) {
282 return EventFlagsMask{eventTypesMask.mask_ | eventType.getFlag()};
283 }
284
285 /**
286 *
287 * @param eventType1
288 * @param eventType2
289 * @return
290 */
291 friend EventFlagsMask operator|(const EventFlag &eventType1, const EventFlag &eventType2) {
292 return EventFlagsMask{eventType1.getFlag() | eventType2.getFlag()};
293 }
294
295 ///
296 std::string toString() const {
297 bool addOrSign = false;
298 std::ostringstream result{};
299
300 for (const auto &flag :
301 {EventFlag::TX_PENDING, EventFlag::REMOVE_EVENT, EventFlag::SNAPSHOT_BEGIN, EventFlag::SNAPSHOT_END,
302 EventFlag::SNAPSHOT_SNIP, EventFlag::SNAPSHOT_MODE, EventFlag::REMOVE_SYMBOL}) {
303 if (flag.in(mask_)) {
304 if (addOrSign) {
305 result << "|";
306 } else {
307 addOrSign = true;
308 }
309
310 result << flag.toString();
311 }
312 }
313
314 result << "(0x" << std::hex << mask_ << ")";
315
316 return result.str();
317 }
318};
