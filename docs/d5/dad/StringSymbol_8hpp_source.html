15 /**
16  * A helper wrapper class needed to pass heterogeneous string symbols using a container and convert them to internal
17  * Graal representation.
18  *
19  * The current implementation is suboptimal (by memory usage) and will be enhanced.
20  */
21 struct DXFCPP_EXPORT StringSymbol final {
22  private:
23  std::string data_;
24 
25  public:
26  StringSymbol(const StringSymbol &stringSymbol) noexcept;
27  StringSymbol(StringSymbol &&) noexcept;
28  StringSymbol &operator=(const StringSymbol &stringSymbol) noexcept;
29  StringSymbol &operator=(StringSymbol &&) noexcept;
30  StringSymbol() noexcept;
31  virtual ~StringSymbol() noexcept;
32 
33  /**
34  * Constructs StringSymbol from a char array
35  *
36  * @param chars The array of chars
37  */
38  StringSymbol(const char *chars) noexcept : StringSymbol() {
39  if constexpr (Debugger::isDebug) {
40  Debugger::debug("StringSymbol(chars = " + toStringAny(chars) + ")");
41  }
42 
43  data_ = std::string(chars);
44  }
45 
46  /**
47  * Constructs StringSymbol from a std::string_view
48  *
49  * @param stringView The string view
50  */
51  StringSymbol(std::string_view stringView) noexcept : StringSymbol() {
52  if constexpr (Debugger::isDebug) {
53  Debugger::debug("StringSymbol(stringView = " + toStringAny(stringView) + ")");
54  }
55 
56  data_ = std::string(stringView);
57  }
58 
59  StringSymbol(std::string string) noexcept : StringSymbol() {
60  if constexpr (Debugger::isDebug) {
61  Debugger::debug("StringSymbol(string = " + toStringAny(string) + ")");
62  }
63 
64  data_ = std::move(string);
65  }
66 
67  /**
68  * Allocates memory for the dxFeed Graal SDK structure (recursively if necessary).
69  * Fills the dxFeed Graal SDK structure's fields by the data of the current entity (recursively if necessary).
70  * Returns the pointer to the filled structure.
71  *
72  * @return The pointer to the filled dxFeed Graal SDK structure
73  */
74  void *toGraal() const noexcept;
75 
76  /**
77  * Releases the memory occupied by the dxFeed Graal SDK structure (recursively if necessary).
78  *
79  * @param graalNative The pointer to the dxFeed Graal SDK structure.
80  */
81  static void freeGraal(void *graalNative) noexcept;
82 
83  static StringSymbol fromGraal(void *graalNative) noexcept;
84 
85  /**
86  * Returns a string representation of the current object.
87  *
88  * @return a string representation
89  */
90  std::string toString() const noexcept {
91  if constexpr (Debugger::isDebug) {
92  return "StringSymbol{" + data_ + "}";
93  } else {
94  return data_;
95  }
96  }
97 
98  const std::string &getData() const;
99 
100  bool operator==(const StringSymbol &stringSymbol) const {
101  return getData() == stringSymbol.getData();
102  }
103 
104  bool operator<(const StringSymbol &stringSymbol) const {
105  return getData() < stringSymbol.getData();
106  }
107 };
