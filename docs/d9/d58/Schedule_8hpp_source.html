10 /**
11  * <b>Schedule</b> class provides API to retrieve and explore trading schedules of different exchanges
12  * and different classes of financial instruments. Each instance of schedule covers separate trading schedule
13  * of some class of instruments, i.e. NYSE stock trading schedule or CME corn futures trading schedule.
14  * Each schedule splits entire time scale into separate @ref Day "days" that are aligned to the specific
15  * trading hours of covered trading schedule.
16  */
17 struct DXFCPP_EXPORT Schedule {
18  friend struct Day;
19 
20  /// The alias to a type of shared pointer to the Schedule object
21  using Ptr = std::shared_ptr<Schedule>;
22 
23  /// The alias to a type of unique pointer to the Schedule object
24  using Unique = std::unique_ptr<Schedule>;
25 
26  private:
27  JavaObjectHandle<Schedule> handle_;
28 
29  explicit Schedule(void *handle = nullptr) noexcept;
30 
31  /**
32  * Checks the handle, attempts to allocate memory for the pointer and return Schedule::Ptr
33  *
34  * @param handle The graal Schedule's handle
35  * @return The smart pointer for the Schedule object.
36  */
37  static Schedule::Ptr create(void *handle) noexcept;
38 
39  public:
40  /**
41  * Returns default schedule instance for specified instrument profile.
42  *
43  * @param profile The instrument profile those schedule is requested
44  * @return The default schedule instance for specified instrument profile
45  */
46  static Schedule::Ptr getInstance(std::shared_ptr<InstrumentProfile> profile) noexcept;
47 
48  /**
49  * Returns default schedule instance for specified schedule definition.
50  *
51  * @param scheduleDefinition The schedule definition of requested schedule
52  * @return The default schedule instance for specified schedule definition
53  */
54  static Schedule::Ptr getInstance(const std::string &scheduleDefinition) noexcept;
55 
56  /**
57  * Returns schedule instance for specified instrument profile and trading venue.
58  *
59  * @param profile The instrument profile those schedule is requested
60  * @param venue The trading venue those schedule is requested
61  * @return The schedule instance for specified instrument profile and trading venue
62  */
63  static Schedule::Ptr getInstance(std::shared_ptr<InstrumentProfile> profile, const std::string &venue) noexcept;
64 
65  /**
66  * Returns trading venues for specified instrument profile.
67  *
68  * @param profile The instrument profile those trading venues are requested
69  * @return trading venues for specified instrument profile
70  */
71  static std::vector<std::string> getTradingVenues(std::shared_ptr<InstrumentProfile> profile) noexcept;
72 
73  /**
74  * Downloads defaults using specified download config and optionally start periodic download.
75  * The specified config can be one of the following:<ul>
76  * <li>"" - stop periodic download
77  * <li>URL - download once from specified URL and stop periodic download
78  * <li>URL,period - start periodic download from specified URL
79  * <li>"auto" - start periodic download from default location
80  * </ul>
81  *
82  * @param downloadConfig download config
83  */
84  static void downloadDefaults(const std::string &downloadConfig) noexcept;
85 
86  /**
87  * Sets shared defaults that are used by individual schedule instances.
88  *
89  * @param data The content of default data
90  * @return `true` if OK
91  */
92  static bool setDefaults(const std::vector<char> &data) noexcept;
93 
94  /**
95  * Returns session that contains specified time.
96  * This method will return `Session::Ptr{nullptr}` if specified time
97  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
98  *
99  * @param time the time to search for
100  * @return session that contains specified time or `Session::Ptr{nullptr}` if specified time
101  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
102  */
103  Session::Ptr getSessionByTime(std::int64_t time) const noexcept;
104 
105  /**
106  * Returns day that contains specified time.
107  * This method will return `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if specified time
108  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
109  *
110  * @param time The time to search for
111  * @return The day that contains specified time or `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if
112  * specified time falls outside of valid date range
113  */
114  Day::Ptr getDayByTime(std::int64_t time) const noexcept;
115 
116  /**
117  * Returns day for specified day identifier.
118  * This method will return `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if specified day identifier
119  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
120  *
121  * @param dayId The day identifier to search for
122  * @return The day for specified day identifier or `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if
123  * specified day identifier falls outside of valid date range
124  * @see Day::getDayId()
125  */
126  Day::Ptr getDayById(std::int32_t dayId) const noexcept;
127 
128  /**
129  * Returns day for specified year, month and day numbers.
130  * Year, month, and day numbers shall be packed in the following way:
131  * <pre>YearMonthDay = year * 10000 + month * 100 + day</pre>
132  * For example, September 28, 1977 has value 19770928.
133  * <p>
134  * If specified day does not exist then this method returns day with
135  * the lowest valid YearMonthDay that is greater than specified one.
136  * This method will return `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if specified year, month and day
137  * numbers fall outside of valid date range from 0001-01-02 to 9999-12-30.
138  *
139  * @param yearMonthDay The year, month and day numbers to search for
140  * @return The day for specified year, month and day numbers or `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr})
141  * if specified year, month and day numbers fall outside of valid date range
142  * @see Day#getYearMonthDay()
143  */
144  Day::Ptr getDayByYearMonthDay(std::int32_t yearMonthDay) const noexcept;
145 
146  /**
147  * Returns session that is nearest to the specified time and that is accepted by specified filter.
148  * This method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time falls
149  * outside of valid date range from 0001-01-02 to 9999-12-30 or if no sessions acceptable by specified filter are
150  * found within one year this method
151  * <p> To find nearest trading session of any type use this code:
152  * <pre>session = schedule->getNearestSessionByTime(time, SessionFilter::TRADING);</pre>
153  * To find nearest regular trading session use this code:
154  * <pre>session = schedule->getNearestSessionByTime(time, SessionFilter::REGULAR);</pre>
155  *
156  * @param time The time to search for
157  * @param filter the filter to test sessions
158  * @return session that is nearest to the specified time and that is accepted by specified filter or
159  * `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time falls outside of valid date range
160  * or if no such day was found within one year
161  */
162  Session::Ptr getNearestSessionByTime(std::int64_t time, const SessionFilter &filter) const noexcept;
163 
164  /**
165  * Returns session that is nearest to the specified time and that is accepted by specified filter.
166  * This method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time
167  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
168  * If no sessions acceptable by specified filter are found within one year this method will return
169  * `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}).
170  * <p>
171  * To find nearest trading session of any type use this code:
172  * <pre>session = schedule->findNearestSessionByTime(time, SessionFilter::TRADING);</pre>
173  * To find nearest regular trading session use this code:
174  * <pre>session = schedule->findNearestSessionByTime(time, SessionFilter::REGULAR);</pre>
175  *
176  * @param time The time to search for
177  * @param filter The filter to test sessions
178  * @return The session that is nearest to the specified time and that is accepted by specified filter or
179  * `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time falls outside of valid date range
180  */
181  Session::Ptr findNearestSessionByTime(std::int64_t time, const SessionFilter &filter) const noexcept;
182 
183  /**
184  * Returns name of this schedule.
185  *
186  * @return The name of this schedule
187  */
188  std::string getName() const noexcept;
189 
190  /**
191  * Returns time zone display name in which this schedule is defined.
192  *
193  * @return time zone display name in which this schedule is defined
194  */
195  std::string getTimeZoneDisplayName() const noexcept;
196 
197  /**
198  * Returns time zone id in which this schedule is defined.
199  *
200  * @return time zone id in which this schedule is defined
201  */
202  std::string getTimeZoneId() const noexcept;
203 };
