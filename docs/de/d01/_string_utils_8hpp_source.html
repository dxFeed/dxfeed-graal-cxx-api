241class IsIEqual {
242 std::locale locale_;
243
244 public:
245 explicit IsIEqual(const std::locale &locale = std::locale()) : locale_{locale} {
246 }
247
248 template <typename T, typename U> bool operator()(const T &t, const U &u) const {
249 return std::tolower<T>(t, locale_) == std::tolower<U>(u, locale_);
250 }
251};

255template <typename Range1, typename Range2, typename Predicate>
256inline bool equals(const Range1 &first, const Range2 &second, Predicate cmp) {
257 auto firstIt = std::begin(first);
258 auto secondIt = std::begin(second);
259
260 for (; firstIt != std::end(first) && secondIt != std::end(second); ++firstIt, ++secondIt) {
261 if (!cmp(*firstIt, *secondIt)) {
262 return false;
263 }
264 }
265
266 return (secondIt == std::end(second)) && (firstIt == std::end(first));
267}
