26template <typename T> std::string toStringAny(T &&t) {
27 if constexpr (requires { t.toString(); }) {
28 return t.toString();
29 } else if constexpr (requires { *t.toString(); }) {
30 return *t.toString();
31 } else if constexpr (requires { toString(t); }) {
32 return toString(t);
33 } else if constexpr (requires { std::to_string(t); }) {
34 return std::to_string(t);
35 } else if constexpr (requires { std::string(t); }) {
36 return std::string(t);
37 } else {
38 return "unknown";
39 }
40}

55 std::string result{"["};

57 for (auto it = begin; it != end; it++) {
58 result += String::EMPTY + "'" + it->getName() + "'" + (std::next(it) == end ? "" : ", ");
