10 enum class SessionTypeEnum : std::uint32_t {
11  /// Non-trading session type is used to mark periods of time during which trading is not allowed.
12  NO_TRADING = 0,
13  /// Pre-market session type marks extended trading session before regular trading hours.
14  PRE_MARKET,

16  REGULAR,
17  /// After-market session type marks extended trading session after regular trading hours.
18  AFTER_MARKET,
19 };

21 /**
22  * Defines type of session - what kind of trading activity is allowed (if any),
23  * what rules are used, what impact on daily trading statistics it has, etc.
24  * The SessionType::NO_TRADING session type is used for non-trading sessions.
25  * <p>
26  * Some exchanges support all session types defined here, others do not.
27  * <p>
28  * Some sessions may have zero duration - e.g. indices that post value once a day.
29  * Such sessions can be of any appropriate type, trading or non-trading.
30  */
31 struct DXFCPP_EXPORT SessionType {
32  /// Non-trading session type is used to mark periods of time during which trading is not allowed.
33  static const SessionType NO_TRADING;
34  /// Pre-market session type marks extended trading session before regular trading hours.
35  static const SessionType PRE_MARKET;
36  /// Regular session type marks regular trading hours session.
37  static const SessionType REGULAR;
38  /// After-market session type marks extended trading session after regular trading hours.
39  static const SessionType AFTER_MARKET;
40 
41  private:
42  SessionTypeEnum code_{};
43  std::string name_{};
44  bool trading_{};
45 
46  SessionType(SessionTypeEnum code, std::string name, bool trading) noexcept
47  : code_{code}, name_{std::move(name)}, trading_{trading} {

62  /**
63  * Returns `true` if trading activity is allowed for this type of session.
64  * <p>
65  * Some sessions may have zero duration - e.g. indices that post value once a day.
66  * Such sessions can be of any appropriate type, trading or non-trading.
67  */
