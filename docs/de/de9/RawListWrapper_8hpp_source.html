19 template <typename T>
20 concept RawGraalList = requires(T list) {
21  { list.size };
22  { list.elements };
23 };
24 
25 template <RawGraalList List> struct RawGraalListTraits {
26  using SizeType = typename std::decay_t<decltype(List::size)>;
27  using ElementType = typename std::decay_t<RemoveAllPointers<decltype(List::elements)>>;
28 };
29 
30 template <RawGraalList List, auto ElementSetter> struct RawListWrapper {
31 #if DXFCPP_DEBUG == 1
32  static auto getDebugName() {
33  return std::string("RawListWrapper<") + typeid(List).name() + ", " + typeid(ElementSetter).name() + ">";
34  }
35 #endif
36 
37  mutable List list_;
38 
39  RawListWrapper() noexcept : list_{0, nullptr} {
40  if constexpr (Debugger::traceLists) {
41  Debugger::trace(getDebugName() + "::()");
42  }
43  }
44 
45  void set(std::size_t index, const auto &value) const noexcept {
46  if constexpr (Debugger::traceLists) {
47  Debugger::trace(getDebugName() + "::set(" + std::to_string(index) + ", " + std::to_string(value) + ")");
48  }
49 
50  if (list_.size == 0) {
51  if constexpr (Debugger::traceLists) {
52  Debugger::trace(getDebugName() + "::set(" + std::to_string(index) + ", " + std::to_string(value) +
53  "): list_.size == 0");
54  }
55 
56  return;
57  }
58 
59  if (index < static_cast<std::size_t>(list_.size)) {
60  if constexpr (Debugger::traceLists) {
61  Debugger::trace(getDebugName() + "::set(" + std::to_string(index) + ", " + std::to_string(value) +
62  "): index < list_.size");
63  }
64 
65  ElementSetter(list_, index, value);
66  }
67  }
68 
69  [[nodiscard]] bool isEmpty() const noexcept {
70  if constexpr (Debugger::traceLists) {
71  Debugger::trace(getDebugName() + "::isEmpty() -> " + dxfcpp::toString(list_.size == 0));
72  }
73 
74  return list_.size == 0;
75  }
76 
77  [[nodiscard]] std::size_t size() const noexcept {
78  if constexpr (Debugger::traceLists) {
79  Debugger::trace(getDebugName() + "::size() -> " + std::to_string(static_cast<std::size_t>(list_.size)));
80  }
81 
82  return static_cast<std::size_t>(list_.size);
83  }
84 
85  void *getHandle() noexcept {
86  if constexpr (Debugger::traceLists) {
87  Debugger::trace(getDebugName() + "::getHandle() -> " + dxfcpp::toString(dxfcpp::bit_cast<void *>(&list_)));
88  }
89 
90  return dxfcpp::bit_cast<void *>(&list_);
91  }
92 
93  void init(std::size_t size) noexcept {
94  if constexpr (Debugger::traceLists) {
95  Debugger::trace(getDebugName() + "::init(" + std::to_string(size) + ")");
96  }
97 
98  if (size <= 0) {
99  if constexpr (Debugger::traceLists) {
100  Debugger::trace(getDebugName() + "::init(" + std::to_string(size) + "): size <= 0");
101  }
102 
103  return;
104  }
105 
106  list_.size = static_cast<typename RawGraalListTraits<List>::SizeType>(size);
107  list_.elements = new (std::nothrow) typename RawGraalListTraits<List>::ElementType *[list_.size];
108 
109  if (!list_.elements) {
110  if constexpr (Debugger::traceLists) {
111  Debugger::trace(getDebugName() + "::init(" + std::to_string(size) + "): !list_.elements");
112  }
113 
114  release();
115 
116  return;
117  }
118 
119  bool needToRelease = false;
120 
121  for (typename RawGraalListTraits<List>::SizeType i = 0; i < list_.size; i++) {
122  list_.elements[i] = new (std::nothrow) typename RawGraalListTraits<List>::ElementType{};
123 
124  if (!list_.elements[i]) {
125  needToRelease = true;
126  }
127  }
128 
129  if (needToRelease) {
130  if constexpr (Debugger::traceLists) {
131  Debugger::trace(getDebugName() + "::init({}): needToRelease");
132  }
133 
134  release();
135  }
136  }
137 
138  void release() {
139  if constexpr (Debugger::traceLists) {
140  Debugger::trace(getDebugName() + "::release()");
141  }
142 
143  if (list_.size == 0 || list_.elements == nullptr) {
144  if constexpr (Debugger::traceLists) {
145  Debugger::trace(getDebugName() + "::release(): list_.size == 0 || list_.elements == nullptr");
146  }
147 
148  return;
149  }
150 
151  for (auto i = list_.size - 1; i >= 0; i--) {
152  delete list_.elements[i];
153  }
154 
155  delete[] list_.elements;
156  list_.size = 0;
157  list_.elements = nullptr;
158  }
159 
160  ~RawListWrapper() noexcept {
161  if constexpr (Debugger::traceLists) {
162  Debugger::trace(getDebugName() + "::~()");
163  }
164 
165  release();
166  }
167 };
