24 /**
25  * Underlying event is a snapshot of computed values that are available for an option underlying
26  * symbol based on the option prices on the market.
27  * It represents the most recent information that is available about the corresponding values on
28  * the market at any given moment of time.
29  *
30  * <h3><a name="eventFlagsSection">Event flags, transactions and snapshots</a></h3>
31  *
32  * Some Underlying sources provide a consistent view of the set of known Underlying events.
33  * The corresponding information is carried in @ref ::getEventFlags() "eventFlags" property.
34  * The logic behind this property is detailed in IndexedEvent class documentation.
35  * Multiple event sources for the same symbol are not supported for Underlying, thus
36  * @ref ::getSource() "source" property is always @ref IndexedEventSource::DEFAULT "DEFAULT".
37  *
38  * <p>TimeSeriesEventModel class handles all the snapshot and transaction logic and conveniently represents
39  * a list current of time-series events order by their @ref ::getTime() "time".
40  *
41  * <h3>Publishing Underlying</h3>
42  *
43  * Publishing of Underlying events follows the general rules explained in TimeSeriesEvent class
44  * documentation.
45  *
46  * <h3>Implementation details</h3>
47  *
48  * This event is implemented on top of QDS record `Underlying`.
49  */
50 class DXFCPP_EXPORT Underlying final : public MarketEvent, public TimeSeriesEvent, public LastingEvent {

66  */

176  /**
177  * Returns timestamp of the event in milliseconds.
178  *
179  * @return timestamp of the event in milliseconds
180  */
181  std::int64_t getTime() const noexcept override {
182  return sar(data_.index, SECONDS_SHIFT) * 1000 + andOp(sar(data_.index, MILLISECONDS_SHIFT), MILLISECONDS_MASK);
183  }

185  /**
186  * Changes timestamp of the event in milliseconds.
187  *

189  * @see ::getTime()
190  */
191  void setTime(std::int64_t time) noexcept {
192  data_.index = orOp(orOp(sal(time_util::getSecondsFromTime(time), SECONDS_SHIFT),
193  sal(time_util::getMillisFromTime(time), MILLISECONDS_SHIFT)),
194  getSequence());
195  }

197  /**
198  * Returns the sequence number of this event to distinguish events that have the same @ref ::getTime() "time".
199  * This sequence number does not have to be unique and does not need to be sequential.
200  * Sequence can range from 0 to ::MAX_SEQUENCE.
201  *
202  * @return The sequence number of this event
203  */
204  std::int32_t getSequence() const noexcept {
205  return static_cast<std::int32_t>(andOp(data_.index, MAX_SEQUENCE));
206  }

213  void setSequence(int sequence) noexcept {
214  // TODO: Improve error handling
215  assert(sequence >= 0 && sequence <= MAX_SEQUENCE);

217  data_.index = orOp(andOp(data_.index, ~MAX_SEQUENCE), sequence);
