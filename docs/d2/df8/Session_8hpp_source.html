10 /**
11  * <b>Session</b> represents a continuous period of time during which apply same rules regarding trading activity.
12  * For example, <code>regular trading session</code> is a period of time consisting of one day of business activities
13  * in a financial market, from the opening bell to the closing bell, when regular trading occurs.
14  * <p>
15  * Sessions can be either <b>trading</b> or <b>non-trading</b>, with different sets of rules and reasons to exist.
16  * Sessions do not overlap with each other - rather they form consecutive chain of adjacent periods of time that
17  * cover entire time scale. The point on a border line is considered to belong to following session that starts there.
18  * Each session completely fits inside a certain day. Day may contain sessions with zero duration - e.g. indices
19  * that post value once a day. Such sessions can be of any appropriate type, trading or non-trading.
20  */
21 struct DXFCPP_EXPORT Session {
22  friend struct Schedule;
23  friend struct Day;
24 
25  /// The alias to a type of shared pointer to the Session object
26  using Ptr = std::shared_ptr<Session>;
27 
28  /// The alias to a type of unique pointer to the Session object
29  using Unique = std::unique_ptr<Session>;
30

32  JavaObjectHandle<Session> handle_;
33 
34  explicit Session(void *handle) noexcept;
35 
36  /**
37  * Checks the handle, attempts to allocate memory for the pointer and return Session::Ptr
38  *
39  * @param handle The graal Session's handle
40  * @return The smart pointer for the Session object.
41  */
42  static Session::Ptr create(void *handle) noexcept;
43 
44  public:
45  /**
46  * Returns day to which this session belongs.
47  *
48  * @return The day to which this session belongs.
49  */
50  Day::Ptr getDay() const noexcept;
51 
52  /**
53  * Returns type of this session.
54  *
55  * @return The type of this session.
56  */
57  const SessionType &getType() const & noexcept;
58 
59  /**
60  * @return <code>true</code> if trading activity is allowed within this session.
61  * This method is equivalent to expression @ref SessionType::isTrading() "getType().isTrading()".
62  * <p>
63  * Some sessions may have zero duration - e.g. indices that post value once a day.
64  * Such sessions can be of any appropriate type, trading or non-trading.
65  */
66  bool isTrading() const noexcept;
67 
68  /**
69  * @return <code>true</code> if this session has zero duration.
70  * Empty sessions can be used for indices that post value once a day or for convenience.
71  * Such sessions can be of any appropriate type, trading or non-trading.
72  */
73  bool isEmpty() const noexcept;
74 
75  /**
76  * @return start time of this session (inclusive).
77  * For normal sessions the start time is less than the end time, for empty sessions they are equal.
78  */
79  std::int64_t getStartTime() const noexcept;
80 
81  /**
82  * @return end time of this session (exclusive).
83  * For normal sessions the end time is greater than the start time, for empty sessions they are equal.
84  */
85  std::int64_t getEndTime() const noexcept;
86 
87  /**
88  * @return <code>true</code> if specified time belongs to this session.
89  */
90  bool containsTime(std::int64_t time) const noexcept;
91 
92  /**
93  * Returns previous session accepted by specified filter.
94  * This method may cross the day boundary and return appropriate session from
95  * previous days - up to a year back in time. If no such session was found
96  * within one year this method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}).
97  * <p>
98  * To find previous trading session of any type use this code:
99  * <pre>session = session->getPrevSession(SessionFilter::TRADING);</pre>
100  * To find previous regular trading session use this code:
101  * <pre>session = session->getPrevSession(SessionFilter::REGULAR);</pre>
102  *
103  * @param filter The filter to test sessions
104  * @return The nearest previous session that is accepted by the filter or `Session::Ptr{nullptr}`
105  * (std::shared_ptr<Session>{nullptr}) if no such session was found within one year
106  */
107  Session::Ptr getPrevSession(const SessionFilter &filter) const noexcept;
108 
109  /**
110  * Returns following session accepted by specified filter.
111  * This method may cross the day boundary and return appropriate session from
112  * following days - up to a year in the future. If no such session was found
113  * within one year this method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}).
114  * <p>
115  * To find following trading session of any type use this code:
116  * <pre>session = session->getNextSession(SessionFilter::TRADING);</pre>
117  * To find following regular trading session use this code:
118  * <pre>session = session->getNextSession(SessionFilter::REGULAR);</pre>
119  *
120  * @param filter The filter to test sessions
121  * @return The nearest following session that is accepted by the filter or `Session::Ptr{nullptr}`
122  * (std::shared_ptr<Session>{nullptr}) if no such
123  * session was found within one year
124  */
125  Session::Ptr getNextSession(const SessionFilter &filter) const noexcept;
126 
127  /**
128  * Returns previous session accepted by specified filter.
129  * This method may cross the day boundary and return appropriate session from
130  * previous days - up to a year back in time. If no such session was found
131  * within one year this method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}).
132  * <p>
133  * To find previous trading session of any type use this code:
134  * <pre>session = session->findPrevSession(SessionFilter::TRADING);</pre>
135  * To find previous regular trading session use this code:
136  * <pre>session = session->findPrevSession(SessionFilter::REGULAR);</pre>
137  *
138  * @param filter The filter to test sessions
139  * @return nearest previous session that is accepted by the filter
140  */
141  Session::Ptr findPrevSession(const SessionFilter &filter) const noexcept;

144  * Returns following session accepted by specified filter.
145  * This method may cross the day boundary and return appropriate session from
146  * following days - up to a year in the future. If no such session was found
147  * within one year this method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}).
148  * <p>
149  * To find following trading session of any type use this code:
150  * <pre>session = session->findNextSession(SessionFilter::TRADING);</pre>
151  * To find following regular trading session use this code:
152  * <pre>session = session->findNextSession(SessionFilter::REGULAR);</pre>
153  *
154  * @param filter The filter to test sessions
155  * @return nearest following session that is accepted by the filter
156  */
157  Session::Ptr findNextSession(const SessionFilter &filter) const noexcept;

187  std::string toString() const noexcept;
