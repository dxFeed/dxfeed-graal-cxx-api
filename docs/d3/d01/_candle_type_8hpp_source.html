20/**
21 * Type of the candle aggregation period constitutes CandlePeriod type together its actual
22 * @ref CandlePeriod::getValue() "value".
23 */
24struct DXFCPP_EXPORT CandleType {
25 /**
26 * Certain number of ticks.
27 */
28 static const CandleType TICK;
29
30 /**
31 * Certain number of seconds.
32 */
33 static const CandleType SECOND;
34
35 /**
36 * Certain number of minutes.
37 */
38 static const CandleType MINUTE;
39
40 /**
41 * Certain number of hours.
42 */
43 static const CandleType HOUR;
44
45 /**
46 * Certain number of days.
47 */
48 static const CandleType DAY;
49
50 /**
51 * Certain number of weeks.
52 */
53 static const CandleType WEEK;
54
55 /**
56 * Certain number of months.
57 */
58 static const CandleType MONTH;
59
60 /**
61 * Certain number of option expirations.
62 */
63 static const CandleType OPTEXP;
64
65 /**
66 * Certain number of years.
67 */
68 static const CandleType YEAR;
69
70 /**
71 * Certain volume of trades.
72 */
73 static const CandleType VOLUME;
74
75 /**
76 * Certain price change, calculated according to the following rules:
77 * <ol>
78 * <li>high(n) - low(n) = price range</li>
79 * <li>close(n) = high(n) or close(n) = low(n)</li>
80 * <li>open(n+1) = close(n)</li>
81 * </ol>
82 * where n is the number of the bar.
83 */
84 static const CandleType PRICE;
85
86 /**
87 * Certain price change, calculated according to the following rules:
88 * <ol>
89 * <li>high(n) - low(n) = price range</li>
90 * <li>close(n) = high(n) or close(n) = low(n)</li>
91 * <li>open(n+1) = close(n) + tick size, if close(n) = high(n)</li>
92 * <li>open(n+1) = close(n) - tick size, if close(n) = low(n)</li>
93 * </ol>
94 * where n is the number of the bar.
95 */
96 static const CandleType PRICE_MOMENTUM;
97
98 /**
99 * Certain price change, calculated according to the following rules:
100 * <ol>
101 * <li>high(n+1) - high(n) = price range or low(n) - low(n+1) = price range</li>
102 * <li>close(n) = high(n) or close(n) = low(n)</li>
103 * <li>open(n+1) = high(n), if high(n+1) - high(n) = price range</li>
104 * <li>open(n+1) = low(n), if low(n) - low(n+1) = price range</li>
105 * </ol>
106 * where n is the number of the bar.
107 */
108 static const CandleType PRICE_RENKO;
109
110 static const std::unordered_map<std::string, std::reference_wrapper<const CandleType>> BY_STRING;
111 static const std::vector<std::reference_wrapper<const CandleType>> VALUES;
112
113 private:
114 std::string name_{};
115 std::string string_{};
116 std::int64_t periodIntervalMillis_{};
117
118 CandleType(std::string name, std::string string, std::int64_t periodIntervalMillis) noexcept
119 : name_{std::move(name)}, string_{std::move(string)}, periodIntervalMillis_{periodIntervalMillis} {
120 }
121
122 public:
123 CandleType() noexcept = default;
124 virtual ~CandleType() noexcept = default;
125
126 /**
127 * Returns candle type period in milliseconds as closely as possible.
128 * Certain types like CandleType::SECOND and CandleType::DAY span a specific number of milliseconds.
129 * CandleType::MONTH, CandleType::OPTEXP and CandleType::YEAR are approximate. Candle type period of
130 * CandleType::TICK, CandleType::VOLUME, CandleType::PRICE, CandleType::PRICE_MOMENTUM and CandleType::PRICE_RENKO
131 * is not defined and this method returns `0`.
132 *
133 * @return aggregation period in milliseconds.
134 */
135 std::int64_t getPeriodIntervalMillis() const noexcept {
136 return periodIntervalMillis_;
137 }
138
139 /**
140 * Returns a name of this candle type
141 *
142 * @return A name of this candle type
143 */
144 const std::string &getName() const & noexcept {
145 return name_;
146 }
147
148 /**
149 * Returns string representation of this candle type.
150 * The string representation of candle type is the shortest unique prefix of the lower case string that corresponds
151 * to its @ref CandleType::getName() "name".
152 * For example, CandleType::TICK is represented as `"t"`, while CandleType::MONTH is represented as `"mo"` to
153 * distinguish it from CandleType::MINUTE that is represented as `"m"`.
154 *
155 * @return string representation of this candle price type.
156 */
157 const std::string &toString() const & noexcept {
158 return string_;
159 }
160
161 /**
162 * Parses string representation of candle type into object.
163 * Any string that that is a prefix of candle type CandleType::getName() can be parsed
164 * (including the one that was returned by CandleType::toString()) and case is ignored for parsing.
165 *
166 * @param s The string representation of candle type.
167 * @return A candle type or std::nullopt if the string representation is invalid.
168 */
169 static std::optional<std::reference_wrapper<const CandleType>> parse(const std::string &s) noexcept {
170 auto n = s.length();
171
172 if (n == 0) {
173 // TODO: error handling [EN-8232] throw IllegalArgumentException("Missing candle type");
174
175 return std::nullopt;
176 }
177
178 auto result = BY_STRING.find(s);
179
180 if (result != BY_STRING.end()) {
181 return result->second;
182 }
183
184 for (const auto &typeRef : VALUES) {
185 const auto &name = typeRef.get().getName();
186
187 // Tick|TICK|tick, Minute|MINUTE|minute, Second|SECOND|second, etc
188 if (name.length() >= n && iEquals(name.substr(0, n), s)) {
189 return typeRef;
190 }
191
192 // Ticks, Minutes, Seconds, etc
193 if (s.ends_with("s") && iEquals(name, s.substr(0, n - 1))) {
194 return typeRef;
195 }
196 }
197
198 // TODO: error handling [EN-8232] throw IllegalArgumentException("Unknown candle type: " + s);
199
200 return std::nullopt;
201 }
202
203 bool operator==(const CandleType &candleType) const noexcept {
204 return string_ == candleType.string_;
205 }
206};
207
208} // namespace dxfcpp

210template <> struct std::hash<dxfcpp::CandleType> {
211 std::size_t operator()(const dxfcpp::CandleType &candleType) const noexcept {
212 return std::hash<std::string>{}(candleType.toString());
213 }
214};
