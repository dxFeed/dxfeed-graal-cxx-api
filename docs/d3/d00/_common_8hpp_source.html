71template <typename To, typename From>
72constexpr To bit_cast(const From &from)
73#if __cpp_concepts
74 requires(sizeof(To) == sizeof(From))
75#endif
76{
77#ifdef __cpp_lib_bit_cast
78 if (is_constant_evaluated())
79 return std::bit_cast<To>(from);
80#endif
81 auto to = To();
82 // The cast suppresses a bogus -Wclass-memaccess on GCC.
83 std::memcpy(static_cast<void *>(&to), &from, sizeof(to));
84 return to;
85}

94template <typename GraalList, typename ElementWrapper> struct GraalListUtils {
95 static std::ptrdiff_t calculateSize(std::ptrdiff_t initSize) noexcept {
96 using ListType = GraalList;
97 using SizeType = decltype(ListType::size);
98
99 if (initSize < 0) {
100 return 0;
101 }
102
103 if (initSize > std::numeric_limits<SizeType>::max()) {
104 return std::numeric_limits<SizeType>::max();
105 }
106
107 return initSize;
108 }

110 static void *newList(std::ptrdiff_t size) {
111 using ListType = GraalList;
112 using SizeType = decltype(ListType::size);
113 using ElementType = std::remove_pointer_t<decltype(ListType::elements)>;
114
115 auto *list = new ListType{static_cast<SizeType>(size), nullptr};
116
117 if (size == 0) {
118 return static_cast<void *>(list);
119 }
120
121 list->elements = new ElementType[size]{nullptr};
122
123 return list;
124 }

126 static bool setElement(void *graalList, std::ptrdiff_t elementIdx, void *element) noexcept {
127 using ListType = GraalList;
128 using SizeType = decltype(ListType::size);
129 using ElementType = std::remove_pointer_t<decltype(ListType::elements)>;
130
131 if (graalList == nullptr || elementIdx < 0 || elementIdx >= std::numeric_limits<SizeType>::max() ||
132 element == nullptr) {
133 return false;
134 }
135
136 static_cast<ListType *>(graalList)->elements[elementIdx] = static_cast<ElementType>(element);
137
138 return true;
139 }
140
141 static bool freeElements(void *graalList, std::ptrdiff_t count) {
142 using ListType = GraalList;
143 using SizeType = decltype(ListType::size);
144
145 if (graalList == nullptr || count < 0 || count >= std::numeric_limits<SizeType>::max()) {
146 return false;
147 }
148
149 auto *list = static_cast<ListType *>(graalList);
150
151 for (SizeType i = 0; i < count; i++) {
152 if (list->elements[i]) {
153 ElementWrapper::freeGraal(static_cast<void *>(list->elements[i]));
154 }
155 }
156
157 delete[] list->elements;
158 delete list;
159
160 return true;
161 }
162
163 static void freeList(void *graalList) {
164 using ListType = GraalList;
165 using SizeType = decltype(ListType::size);
166
167 if (graalList == nullptr) {
168 return;
169 }
170
171 auto list = static_cast<ListType *>(graalList);
172
173 if (list->size > 0 && list->elements != nullptr) {
174 for (SizeType elementIndex = 0; elementIndex < list->size; elementIndex++) {
175 if (list->elements[elementIndex]) {
176 ElementWrapper::freeGraal(static_cast<void *>(list->elements[elementIndex]));
177 }
178 }
179
180 delete[] list->elements;
181 }
182
183 delete list;
184 }
185
186 static std::vector<ElementWrapper> fromList(void *graalList) {
187 using ListType = GraalList;
188 using SizeType = decltype(ListType::size);
189
190 if (!graalList) {
191 return {};
192 }
193
194 std::vector<ElementWrapper> result{};
195
196 auto list = static_cast<ListType *>(graalList);
197
198 if (list->size > 0 && list->elements != nullptr) {
199 for (SizeType elementIndex = 0; elementIndex < list->size; elementIndex++) {
200 if (list->elements[elementIndex]) {
201 result.emplace_back(ElementWrapper::fromGraal(static_cast<void *>(list->elements[elementIndex])));
202 }
203 }
204 }
205
206 return result;
207 }

219static constexpr std::int64_t floorDiv(std::int64_t x, std::int64_t y) {
220 std::int64_t r = x / y;
221
222 // if the signs are different and modulo not zero, round down
223 if ((x < 0) != (y < 0) && (r * y != x)) {
224 r--;
225 }
226
227 return r;
228}

231 return x - (floorDiv(x, y) * y);

256static constexpr std::int64_t NANOS_IN_MILLIS = 1'000'000LL;
257
258/**
259 * Returns time measured in nanoseconds since epoch from the time in milliseconds and its nano part.
260 * The result of this method is `timeMillis * 1'000'000 + timeNanoPart`.
261 *
262 * @param timeMillis time in milliseconds since epoch.
263 * @param timeNanoPart nanoseconds part that shall lie within [0..999999] interval.
264 * @return time measured in nanoseconds since epoch.
265 */
266static constexpr std::int64_t getNanosFromMillisAndNanoPart(std::int64_t timeMillis, std::int32_t timeNanoPart) {
267 return (timeMillis * NANOS_IN_MILLIS) + timeNanoPart;
268}
269
270/**
271 * Returns time measured in milliseconds since Java epoch from the time in nanoseconds.
272 * Idea is that nano part of time shall be within [0..999999] interval
273 * so that the following equation always holds
274 * `getMillisFromNanos(timeNanos) * 1'000'000 + getNanoPartFromNanos(timeNanos) == timeNanos`.
275 *
276 * @param timeNanos time measured in nanoseconds since epoch
277 * @return time measured in milliseconds since epoch.
278 * @see ::getNanoPartFromNanos()
279 */
280static constexpr std::int64_t getMillisFromNanos(std::int64_t timeNanos) {
281 return math::floorDiv(timeNanos, NANOS_IN_MILLIS);
282}
283
284/**
285 * Returns nano part of time.
286 * Idea is that nano part of time shall be within [0..999999] interval
287 * so that the following equation always holds
288 * `getMillisFromNanos(timeNanos) * 1'000'000 + getNanoPartFromNanos(timeNanos) == timeNanos`.
289 *
290 * @param timeNanos time measured in nanoseconds since epoch
291 * @return time measured in milliseconds since epoch.
292 * @see ::getMillisFromNanos()
293 */
294static constexpr std::int32_t getNanoPartFromNanos(std::int64_t timeNanos) {
295 return static_cast<std::int32_t>(math::floorMod(timeNanos, NANOS_IN_MILLIS));
296}

303
304/// Number of milliseconds in a minute.
305static constexpr std::int64_t MINUTE = 60LL * SECOND;

310/// Number of milliseconds in an day.
311static constexpr std::int64_t DAY = 24LL * HOUR;

313/**
314 * Returns correct number of milliseconds with proper handling negative values.
315 * Idea is that number of milliseconds shall be within [0..999] interval
316 *
317 * @param timeMillis The timestamp in milliseconds
318 * @return a correct number of milliseconds
319 */
320static constexpr std::int32_t getMillisFromTime(std::int64_t timeMillis) {
321 return static_cast<std::int32_t>(math::floorMod(timeMillis, SECOND));
322}
323
324/**
325 * Returns correct number of seconds with proper handling negative values and overflows.
326 * Idea is that number of milliseconds shall be within [0..999] interval
327 *
328 * @param timeMillis The timestamp in milliseconds
329 * @return a correct number of second
330 */
331static constexpr std::int32_t getSecondsFromTime(std::int64_t timeMillis) {
332 if (timeMillis >= 0) {
333 return static_cast<std::int32_t>(
334 std::min(timeMillis / SECOND, static_cast<std::int64_t>(std::numeric_limits<std::int32_t>::max())));
335 }
336
337 return static_cast<std::int32_t>(
338 std::max((timeMillis + 1) / SECOND - 1, static_cast<std::int64_t>(std::numeric_limits<std::int32_t>::min())));
339}

344/**
345 * Returns quotient according to number theory - i.e. when remainder is zero or positive.
346 *
347 * @tparam T The dividend's and divisor's type
348 * @param a dividend
349 * @param b divisor
350 * @return quotient according to number theory
351 */
352template <Integral T> constexpr static T div(T a, T b) {
353 if (a >= 0) {
354 return a / b;
355 }
356
357 if (b >= 0) {
358 return ((a + 1) / b) - 1;
359 }
360
361 return ((a + 1) / b) + 1;
362}

376/**
377 * Returns yyyymmdd integer in Gregorian calendar for a specified day identifier.
378 * The day identifier is defined as the number of days since Unix epoch of January 1, 1970.
379 * The result is equal to `yearSign * (abs(year) * 10000 + month * 100 + day)`, where year,
380 * month, and day are in Gregorian calendar, month is between 1 and 12 inclusive, and day is counted from 1.
381 *
382 * @param dayId The day id
383 * @return integer date
384 */
385constexpr static std::int32_t getYearMonthDayByDayId(std::int32_t dayId) {
386 std::int32_t j = dayId + 2472632; // this shifts the epoch back to astronomical year -4800
387 std::int32_t g = math_util::div(j, 146097);
388 std::int32_t dg = j - g * 146097;
389 std::int32_t c = (dg / 36524 + 1) * 3 / 4;
390 std::int32_t dc = dg - c * 36524;
391 std::int32_t b = dc / 1461;
392 std::int32_t db = dc - b * 1461;
393 std::int32_t a = (db / 365 + 1) * 3 / 4;
394 std::int32_t da = db - a * 365;
395 std::int32_t y = g * 400 + c * 100 + b * 4 +
396 a; // this is the integer number of full years elapsed since March 1, 4801 BC at 00:00 UTC
397 std::int32_t m = (da * 5 + 308) / 153 -
398 2; // this is the integer number of full months elapsed since the last March 1 at 00:00 UTC
399 std::int32_t d =
400 da - (m + 4) * 153 / 5 + 122; // this is the number of days elapsed since day 1 of the month at 00:00 UTC
401 std::int32_t yyyy = y - 4800 + (m + 2) / 12;
402 std::int32_t mm = (m + 2) % 12 + 1;
403 std::int32_t dd = d + 1;
404 std::int32_t yyyymmdd = math_util::abs(yyyy) * 10000 + mm * 100 + dd;
405
406 return yyyy >= 0 ? yyyymmdd : -yyyymmdd;
407}

434/**
435 * Performs a right arithmetic bit shift operation (>> in Java, C, etc). The sign bit is extended to preserve the
436 * signedness of the number.
437 *
438 * The result of the shift will be of the same type as the `value` being shifted.
439 * If the shift is a negative number of bits, then a @ref ::sal() "left arithmetic shift" will be performed.
440 * If the shift size is greater than or equal to the number of bits in the shifted `value`, then if the `value` is
441 * negative (signed integer type), `-1` will be returned, and if positive, then `0` will be returned.
442 *
443 * @tparam V The type of `value`
444 * @tparam S The type of `shift`
445 * @param value The value to be shifted.
446 * @param shift The shift in bits
447 * @return The shifted `value`
448 */
449template <Integral V, Integral S> static constexpr V sar(V value, S shift) noexcept;

451/**
452 * Performs a left arithmetic bit shift operation (<< in Java, C, etc).
453 *
454 * The result of the shift will be of the same type as the `value` being shifted.
455 * If the shift is a negative number of bits, then a @ref ::sar() "right arithmetic shift" will be performed.
456 * If the shift size is greater than or equal to the number of bits in the shifted `value`, then `0` will be
457 * returned.
458 *
459 * @tparam V The type of `value`
460 * @tparam S The type of `shift`
461 * @param value The value to be shifted
462 * @param shift The shift in bits
463 * @return The shifted `value`
464 */
465template <Integral V, Integral S> static constexpr V leftArithmeticShift(V value, S shift) noexcept {
466 if constexpr (std::is_signed_v<S>) {
467 if (shift < 0) {
468 return sar(value, -shift);
469 }
470 }
471
472 if (shift == 0 || value == 0) {
473 return value;
474 }
475
476 auto unsignedShift = static_cast<std::make_unsigned_t<S>>(shift);
477
478 if (unsignedShift >= sizeof(V) * CHAR_BIT) {
479 return 0;
480 }
481
482 return value << unsignedShift;
483}

699template <Integral T> static constexpr T setBits(T flags, T mask, T shift, T bits) noexcept {
700 if constexpr (std::is_signed_v<T>) {
701 using U = std::make_unsigned_t<T>;
702
703 return static_cast<T>((static_cast<U>(flags) & ~(static_cast<U>(mask) << static_cast<U>(shift))) |
704 ((static_cast<U>(bits) & static_cast<U>(mask)) << static_cast<U>(shift)));
705 } else {
706 return (flags & ~(mask << shift)) | ((bits & mask) << shift);
707 }

713 using U = std::make_unsigned_t<Max<F, M, S, B>>;
714
715 return static_cast<F>((static_cast<U>(flags) & ~(static_cast<U>(mask) << static_cast<U>(shift))) |
716 ((static_cast<U>(bits) & static_cast<U>(mask)) << static_cast<U>(shift)));
717 } else {
