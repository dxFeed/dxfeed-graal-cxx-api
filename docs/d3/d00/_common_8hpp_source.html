50template <typename To, typename From>
51constexpr To bit_cast(const From &from)
52#if __cpp_concepts
53 requires(sizeof(To) == sizeof(From))
54#endif
55{
56#ifdef __cpp_lib_bit_cast
57 if (is_constant_evaluated())
58 return std::bit_cast<To>(from);
59#endif
60 auto to = To();
61 // The cast suppresses a bogus -Wclass-memaccess on GCC.
62 std::memcpy(static_cast<void *>(&to), &from, sizeof(to));
63 return to;

66// final_action allows you to ensure something gets run at the end of a scope
67template <class F> class final_action {
68 public:
69 explicit final_action(const F &ff) noexcept : f{ff} {}
70 explicit final_action(F &&ff) noexcept : f{std::move(ff)} {}
71
72 ~final_action() noexcept {
73 if (invoke)
74 f();
75 }
76
77 final_action(final_action &&other) noexcept : f(std::move(other.f)), invoke(std::exchange(other.invoke, false)) {}
78
79 final_action(const final_action &) = delete;
80 void operator=(const final_action &) = delete;
81 void operator=(final_action &&) = delete;
82
83 private:
84 F f;
85 bool invoke = true;
86};

127static constexpr std::int64_t floorDiv(std::int64_t x, std::int64_t y) {
128 std::int64_t r = x / y;
129
130 // if the signs are different and modulo not zero, round down
131 if ((x < 0) != (y < 0) && (r * y != x)) {
132 r--;
133 }
134
135 return r;
136}

145static constexpr std::int64_t NANOS_IN_MILLIS = 1'000'000LL;
146
147/**
148 * Returns time measured in nanoseconds since epoch from the time in milliseconds and its nano part.
149 * The result of this method is `timeMillis * 1'000'000 + timeNanoPart`.
150 *
151 * @param timeMillis time in milliseconds since epoch.
152 * @param timeNanoPart nanoseconds part that shall lie within [0..999999] interval.
153 * @return time measured in nanoseconds since epoch.
154 */
155static constexpr std::int64_t getNanosFromMillisAndNanoPart(std::int64_t timeMillis, std::int32_t timeNanoPart) {
156 return (timeMillis * NANOS_IN_MILLIS) + timeNanoPart;
157}
158
159/**
160 * Returns time measured in milliseconds since Java epoch from the time in nanoseconds.
161 * Idea is that nano part of time shall be within [0..999999] interval
162 * so that the following equation always holds
163 * `getMillisFromNanos(timeNanos) * 1'000'000 + getNanoPartFromNanos(timeNanos) == timeNanos`.
164 *
165 * @param timeNanos time measured in nanoseconds since epoch
166 * @return time measured in milliseconds since epoch.
167 * @see ::getNanoPartFromNanos()
168 */
169static constexpr std::int64_t getMillisFromNanos(std::int64_t timeNanos) {
170 return math::floorDiv(timeNanos, NANOS_IN_MILLIS);
171}
172
173/**
174 * Returns nano part of time.
175 * Idea is that nano part of time shall be within [0..999999] interval
176 * so that the following equation always holds
177 * `getMillisFromNanos(timeNanos) * 1'000'000 + getNanoPartFromNanos(timeNanos) == timeNanos`.
178 *
179 * @param timeNanos time measured in nanoseconds since epoch
180 * @return time measured in milliseconds since epoch.
181 * @see ::getMillisFromNanos()
182 */
183static constexpr std::int32_t getNanoPartFromNanos(std::int64_t timeNanos) {
184 return static_cast<std::int32_t>(math::floorMod(timeNanos, NANOS_IN_MILLIS));
185}

190/// Number of milliseconds in a second.
191static const std::int64_t SECOND = 1000LL;
192
193/**
194 * Returns correct number of milliseconds with proper handling negative values.
195 * Idea is that number of milliseconds shall be within [0..999] interval
196 *
197 * @param timeMillis The timestamp in milliseconds
198 * @return a correct number of milliseconds
199 */
200static std::int32_t getMillisFromTime(std::int64_t timeMillis) {
201 return static_cast<std::int32_t>(math::floorMod(timeMillis, SECOND));
202}
203
204/**
205 * Returns correct number of seconds with proper handling negative values and overflows.
206 * Idea is that number of milliseconds shall be within [0..999] interval
207 *
208 * @param timeMillis The timestamp in milliseconds
209 * @return a correct number of second
210 */
211static constexpr std::int32_t getSecondsFromTime(std::int64_t timeMillis) {
212 if (timeMillis >= 0) {
213 return static_cast<std::int32_t>(
214 std::min(timeMillis / SECOND, static_cast<std::int64_t>(std::numeric_limits<std::int32_t>::max())));
215 }
216
217 return static_cast<std::int32_t>(
218 std::max((timeMillis + 1) / SECOND - 1, static_cast<std::int64_t>(std::numeric_limits<std::int32_t>::min())));
219}

224/**
225 * Returns quotient according to number theory - i.e. when remainder is zero or positive.
226 *
227 * @tparam T The dividend's and divisor's type
228 * @param a dividend
229 * @param b divisor
230 * @return quotient according to number theory
231 */
232template <Integral T> constexpr static T div(T a, T b) {
233 if (a >= 0) {
234 return a / b;
235 }
236
237 if (b >= 0) {
238 return ((a + 1) / b) - 1;
239 }
240
241 return ((a + 1) / b) + 1;
242}

250static std::int32_t DAY_OF_YEAR[] = {0, 0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334, 365};
251
252/**
253 * Returns yyyymmdd integer in Gregorian calendar for a specified day identifier.
254 * The day identifier is defined as the number of days since Unix epoch of January 1, 1970.
255 * The result is equal to `yearSign * (abs(year) * 10000 + month * 100 + day)`, where year,
256 * month, and day are in Gregorian calendar, month is between 1 and 12 inclusive, and day is counted from 1.
257 *
258 * @param dayId The day id
259 * @return integer date
260 */
261constexpr static std::int32_t getYearMonthDayByDayId(std::int32_t dayId) {
262 std::int32_t j = dayId + 2472632; // this shifts the epoch back to astronomical year -4800
263 std::int32_t g = math_util::div(j, 146097);
264 std::int32_t dg = j - g * 146097;
265 std::int32_t c = (dg / 36524 + 1) * 3 / 4;
266 std::int32_t dc = dg - c * 36524;
267 std::int32_t b = dc / 1461;
268 std::int32_t db = dc - b * 1461;
269 std::int32_t a = (db / 365 + 1) * 3 / 4;
270 std::int32_t da = db - a * 365;
271 std::int32_t y = g * 400 + c * 100 + b * 4 +
272 a; // this is the integer number of full years elapsed since March 1, 4801 BC at 00:00 UTC
273 std::int32_t m = (da * 5 + 308) / 153 -
274 2; // this is the integer number of full months elapsed since the last March 1 at 00:00 UTC
275 std::int32_t d =
276 da - (m + 4) * 153 / 5 + 122; // this is the number of days elapsed since day 1 of the month at 00:00 UTC
277 std::int32_t yyyy = y - 4800 + (m + 2) / 12;
278 std::int32_t mm = (m + 2) % 12 + 1;
279 std::int32_t dd = d + 1;
280 std::int32_t yyyymmdd = math_util::abs(yyyy) * 10000 + mm * 100 + dd;
281
282 return yyyy >= 0 ? yyyymmdd : -yyyymmdd;
283}

285constexpr static std::int32_t getDayIdByYearMonthDay(std::int32_t year, std::int32_t month, std::int32_t day) {
286 // TODO: error handling
287 if (month < 1 || month > 12) {
288 return -1;
289 }
290
291 std::int32_t dayOfYear = DAY_OF_YEAR[month] + day - 1;
292
293 if (month > 2 && year % 4 == 0 && (year % 100 != 0 || year % 400 == 0)) {
294 dayOfYear++;
295 }
296
297 return year * 365 + math_util::div(year - 1, 4) - math_util::div(year - 1, 100) + math_util::div(year - 1, 400) +
298 dayOfYear - 719527;
299}

324/**
325 * Performs a right arithmetic bit shift operation.
326 *
327 * The result of the shift will be of the same type as the `value` being shifted.
328 * If the shift is a negative number of bits, then a @ref ::sal() "left arithmetic shift" will be performed.
329 * If the shift size is greater than or equal to the number of bits in the shifted `value`, then if the `value` is
330 * negative (signed integer type), `-1` will be returned, and if positive, then `0` will be returned.
331 *
332 * @tparam V The type of `value`
333 * @tparam S The type of `shift`
334 * @param value The value to be shifted.
335 * @param shift The shift in bits
336 * @return The shifted `value`
337 */
338template <Integral V, Integral S> static constexpr V sar(V value, S shift);

340/**
341 * Performs a left arithmetic bit shift operation.
342 *
343 * The result of the shift will be of the same type as the `value` being shifted.
344 * If the shift is a negative number of bits, then a @ref ::sar() "right arithmetic shift" will be performed.
345 * If the shift size is greater than or equal to the number of bits in the shifted `value`, then `0` will be
346 * returned.
347 *
348 * @tparam V The type of `value`
349 * @tparam S The type of `shift`
350 * @param value The value to be shifted
351 * @param shift The shift in bits
352 * @return The shifted `value`
353 */
354template <Integral V, Integral S> static constexpr V leftArithmeticShift(V value, S shift) {
355 if constexpr (std::is_signed_v<S>) {
356 if (shift < 0) {
357 return sar(value, -shift);
358 }
359 }
360
361 if (shift == 0 || value == 0) {
362 return value;
363 }
364
365 if (shift >= sizeof(V) * 8) {
366 return 0;
367 }
368
369 return value << static_cast<std::make_unsigned_t<S>>(shift);
370}
