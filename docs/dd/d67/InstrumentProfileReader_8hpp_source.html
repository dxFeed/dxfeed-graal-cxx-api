26 /**
27  * Reads instrument profiles from the stream using Instrument Profile Format (IPF).
28  * Please see <b>Instrument Profile Format</b> documentation for complete description.
29  * This reader automatically uses data formats as specified in the stream.
30  *
31  * <p>This reader is intended for "one time only" usage: create new instances for new IPF reads.
32  * <p>Use {@link InstrumentProfileConnection} if support for streaming updates of instrument profiles is needed.
33  *
34  * <p>For backward compatibility reader can be configured with system property "-Dcom.dxfeed.ipf.complete" to control
35  * the strategy for missing "##COMPLETE" tag when reading IPF, possible values are:
36  * <ul>
37  * <li>{@code warn} - show warning in the log (default)</li>
38  * <li>{@code error} - throw exception (future default)</li>
39  * <li>{@code ignore} - do nothing (for backward compatibility)</li>
40  * </ul>
41  */
42 class DXFCPP_EXPORT InstrumentProfileReader final : public SharedEntity {
43  Id<InstrumentProfileReader> id_;
44  JavaObjectHandle<InstrumentProfileReader> handle_;
45 
46  InstrumentProfileReader() noexcept;
47 
48  public:
49  /// The alias to a type of shared pointer to the InstrumentProfileReader object
50  using Ptr = std::shared_ptr<InstrumentProfileReader>;
51 
52  /// The alias to a type of unique pointer to the InstrumentProfileReader object
53  using Unique = std::unique_ptr<InstrumentProfileReader>;
54 
55  /**
56  * Creates the new InstrumentProfileReader
57  *
58  * @return The new InstrumentProfileReader
59  */
60  static Ptr create() noexcept;
61 
62  /**
63  * Returns last modification time (in milliseconds) from last InstrumentProfileReader::readFromFile() operation
64  * or zero if it is unknown.
65  */
66  std::int64_t getLastModified() const noexcept;
67 
68  /**
69  * Returns `true` if IPF was fully read on last InstrumentProfileReader::readFromFile() operation.
70  */
71  bool wasComplete() const noexcept;
72 
73  /**
74  * Converts a specified string address specification into an URL that will be read by
75  * InstrumentProfileReader::readFromFile() method.
76  */
77  static std::string resolveSourceURL(const std::string &address) noexcept;
78 
79  /**
80  * Reads and returns instrument profiles from specified file.
81  * This method recognizes data compression formats "zip" and "gzip" automatically.
82  * In case of <em>zip</em> the first file entry will be read and parsed as a plain data stream.
83  * In case of <em>gzip</em> compressed content will be read and processed.
84  * In other cases data considered uncompressed and will be parsed as is.
85  *
86  * <p>Authentication information can be supplied to this method as part of URL user info
87  * like {@code "http://user:password@host:port/path/file.ipf"}.
88  *
89  * <p>This is a shortcut for
90  * <code>{@link #readFromFile(String, String, String) readFromFile}(address, <b>null</b>, <b>null</b>)</code>.
91  *
92  * <p>This operation updates {@link #getLastModified() lastModified} and {@link #wasComplete() wasComplete}.
93  *
94  * @param address URL of file to read from
95  * @return list of instrument profiles
96  */
97  std::vector<std::shared_ptr<InstrumentProfile>> readFromFile(const std::string &address) const noexcept;
98 
99  /**
100  * Reads and returns instrument profiles from specified address with a specified basic user and password
101  * credentials.
102  * This method recognizes data compression formats "zip" and "gzip" automatically.
103  * In case of <em>zip</em> the first file entry will be read and parsed as a plain data stream.
104  * In case of <em>gzip</em> compressed content will be read and processed.
105  * In other cases data considered uncompressed and will be parsed as is.
106  *
107  * <p>Specified user and password take precedence over authentication information that is supplied to this method
108  * as part of URL user info like {@code "http://user:password@host:port/path/file.ipf"}.
109  *
110  * <p>This operation updates {@link #getLastModified() lastModified} and {@link #wasComplete() wasComplete}.
111  *
112  * @param address URL of file to read from
113  * @param user the user name (may be null)
114  * @param password the password (may be null)
115  * @return list of instrument profiles
116  */
117  std::vector<std::shared_ptr<InstrumentProfile>> readFromFile(const std::string &address, const std::string &user,
118  const std::string &password) const noexcept;
119 };
