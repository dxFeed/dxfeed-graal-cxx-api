20 /**
21  * Identifies source of Order, AnalyticOrder, OtcMarketsOrder and SpreadOrder events.
22  * There are the following kinds of order sources:
23  * <ul>
24  * <li><em>Synthetic</em> sources OrderSource::COMPOSITE_BID, OrderSource::COMPOSITE_ASK,
25  * OrderSource::REGIONAL_BID, and OrderSource::REGIONAL_ASK are provided for convenience of a consolidated
26  * order book and are automatically generated based on the corresponding Quote events.
27  * <li><em>Aggregate</em> sources OrderSource::AGGREGATE_BID and OrderSource::AGGREGATE_ASK provide
28  * futures depth (aggregated by price level) and NASDAQ Level II (top of book for each market maker).
29  * These source cannot be directly published to via dxFeed API.
30  * <li><em>Publishable</em> sources OrderSource::DEFAULT, OrderSource::NTV, OrderSource::ISE etc
31  * support full range of dxFeed API features.
32  * </ul>
33  */

65  static std::int32_t composeId(const std::string &name) noexcept {
66  std::int32_t sourceId = 0;
67 
68  auto n = name.length();
69 
70  if (n == 0 || n > 4) {
71  // TODO: error handling: [EN-8232] IllegalArgumentException("Source name must contain from 1 to 4 characters");
72 
73  return -1;
74  }
75 
76  for (auto c : name) {
77  if (!checkChar(c)) {
78  // TODO: error handling [EN-8232]
79 
80  return -1;
81  }
82 
83  sourceId = orOp(sal(sourceId, 8), c);
84  }
85 
86  return sourceId;
87  }

89  static bool checkChar(char c) noexcept {
90  if (c >= 'A' && c <= 'Z' || c >= 'a' && c <= 'z' || c >= '0' && c <= '9') {
91  return true;
92  }
93 
94  // TODO: error handling: [EN-8232] IllegalArgumentException("Source name must contain only alphanumeric characters");
95 
96  return false;
97  }
98 
99  static std::string decodeName(int id) noexcept {
100  if (id == 0) {
101  // TODO: error handling: [EN-8232] throw IllegalArgumentException("Source name must contain from 1 to 4 characters");
102 
103  return "";
104  }
105 
106  std::string name(4, '\0');
107 
108  std::size_t n = 0;
109 
110  for (int i = 24; i >= 0; i -= 8) {
111  if (sar(id, i) == 0) { // skip highest contiguous zeros
112  continue;
113  }
114 
115  char c = static_cast<char>(andOp(sar(id, i), 0xff));
116 
117  if (!checkChar(c)) {
118  return "";
119  }
120 
121  name[n++] = c;
122  }
123 
124  return name.substr(0, n);
125  }

141 
142  return 0;
143  }

478  /**
479  * Returns order source for the specified source identifier.
480  *
481  * @param sourceId the source identifier.
482  * @return order source.
483  */
484  static const OrderSource &valueOf(std::int32_t sourceId) noexcept {
485  if (auto found = PREDEFINED_SOURCES.find(sourceId); found != PREDEFINED_SOURCES.end()) {
486  return found->second;
487  }
488 
489  std::lock_guard lock(MTX_);
490 
491  if (auto found = USER_SOURCES_.find(sourceId); found != USER_SOURCES_.end()) {
492  return found->second;
493  }
494 
495  return USER_SOURCES_.try_emplace(sourceId, OrderSource(sourceId, decodeName(sourceId))).first->second;
496  }
497 
498  /**
499  * Returns order source for the specified source name.
500  * The name must be either predefined, or contain at most 4 alphanumeric characters.
501  *
502  * @param name the name of the source.
503  * @return order source.
504  */
505  static const OrderSource &valueOf(const std::string &name) noexcept {
506  if (auto found = PREDEFINED_SOURCES.find(name); found != PREDEFINED_SOURCES.end()) {
507  return found->second;
508  }
509 
510  std::lock_guard lock(MTX_);
511 
512  auto sourceId = composeId(name);
513 
514  if (auto found = USER_SOURCES_.find(sourceId); found != USER_SOURCES_.end()) {
515  return found->second;
516  }
517 
518  return USER_SOURCES_.try_emplace(sourceId, OrderSource(sourceId, name)).first->second;
519  }
