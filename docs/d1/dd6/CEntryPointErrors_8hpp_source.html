115  static const std::unordered_map<CodeType, std::reference_wrapper<const CEntryPointErrors>> ALL;

117  template <Integral Code> static const CEntryPointErrors &valueOf(Code code) {
118  if (auto found = ALL.find(static_cast<CodeType>(code)); found != ALL.end()) {
