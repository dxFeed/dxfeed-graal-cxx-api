29 template <typename T> std::string toStringAny(T &&t) {
30  if constexpr (requires { t.toString(); }) {
31  return t.toString();
32  } else if constexpr (requires { *t.toString(); }) {
33  return *t.toString();
34  } else if constexpr (requires { toString(t); }) {
35  return toString(t);
36  } else if constexpr (requires { std::to_string(t); }) {
37  return std::to_string(t);
38  } else if constexpr (requires { std::string(t); }) {
39  return std::string(t);
40  } else {
41  return "unknown";
42  }
43 }

105 template <typename Range1, typename Range2, typename Predicate>
106 inline bool equals(const Range1 &first, const Range2 &second, Predicate cmp) {
107  auto firstIt = std::begin(first);
108  auto secondIt = std::begin(second);
109 
110  for (; firstIt != std::end(first) && secondIt != std::end(second); ++firstIt, secondIt++) {
111  if (!cmp(*firstIt, *secondIt)) {
112  return false;
113  }
114  }
115 
116  return (secondIt == std::end(second)) && (firstIt == std::end(first));
117 }
