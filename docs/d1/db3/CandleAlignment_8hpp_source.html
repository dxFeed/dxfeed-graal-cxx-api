19 /**
20  * Candle alignment attribute of CandleSymbol defines how candle are aligned with respect to time.
21  *
22  * <h3>Implementation details</h3>
23  *
24  * This attribute is encoded in a symbol string with
25  * @ref MarketEventSymbols::getAttributeStringByKey() "MarketEventSymbols.getAttributeStringByKey",
26  * @ref MarketEventSymbols::changeAttributeStringByKey() "changeAttributeStringByKey", and
27  * @ref MarketEventSymbols::removeAttributeStringByKey() "removeAttributeStringByKey" methods.
28  * The key to use with these methods is available via CandleAlignment::ATTRIBUTE_KEY constant.
29  * The value that this key shall be set to is equal to the corresponding CandleAlignment::toString()
30  */
31 struct DXFCPP_EXPORT CandleAlignment : public CandleSymbolAttribute {
32  /**
33  * Align candles on midnight.
34  */
35  static const CandleAlignment MIDNIGHT;
36 
37  /**
38  * Align candles on trading sessions.
39  */
40  static const CandleAlignment SESSION;
41 
42  /**
43  * Default alignment is CandleAlignment::MIDNIGHT.
44  */
45  static const CandleAlignment DEFAULT;
46 
47  /**
48  * The attribute key that is used to store the value of CandleAlignment in
49  * a symbol string using methods of MarketEventSymbols class.
50  * The value of this constant is "a".
51  * The value that this key shall be set to is equal to the corresponding CandleAlignment::toString()
52  */
53  static const std::string ATTRIBUTE_KEY;
54 
55  static const std::unordered_map<std::string, std::reference_wrapper<const CandleAlignment>> BY_STRING;
56  static const std::vector<std::reference_wrapper<const CandleAlignment>> VALUES;
57 
58  private:
59  std::string string_{};
60 
61  explicit CandleAlignment(const std::string &string) noexcept : string_{string} {
62  }
63 
64  public:
65  CandleAlignment() noexcept = default;
66  virtual ~CandleAlignment() noexcept = default;
67 
68  /**
69  * Returns candle event symbol string with this candle alignment set.
70  *
71  * @param symbol The original candle event symbol.
72  * @return candle event symbol string with this candle alignment set.
73  */
74  std::string changeAttributeForSymbol(const std::string &symbol) const noexcept override {
75  return *this == DEFAULT ? MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY)
76  : MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY, toString());
77  }
78 
79  /**
80  * Returns string representation of this candle alignment.
81  * The string representation of candle alignment "m" for CandleAlignment::MIDNIGHT and "s" for
82  * CandleAlignment::SESSION.
83  *
84  * @return string representation of this candle alignment.
85  */
86  std::string toString() const noexcept {
87  return string_;
88  }
89 
90  bool operator==(const CandleAlignment &candleAlignment) const noexcept {
91  return string_ == candleAlignment.string_;
92  }
93 
94  /**
95  * Parses string representation of candle alignment into object.
96  * Any string that was returned by CandleAlignment::toString() can be parsed and case is ignored for parsing.
97  *
98  * @param s The string representation of candle alignment.
99  * @return The candle alignment (reference) or std::nullopt if there is no supported attribute's value.
100  */
101  static std::optional<std::reference_wrapper<const CandleAlignment>> parse(const std::string &s) noexcept {
102  auto found = BY_STRING.find(s);
103 
104  if (found != BY_STRING.end()) {
105  return found->second;
106  }
107 
108  for (const auto &alignmentRef : VALUES) {
109  const auto &alignmentStr = alignmentRef.get().toString();
110 
111  if (iEquals(alignmentStr, s)) {
112  return alignmentRef;
113  }
114  }
115 
116  // TODO: error handling [EN-8232] throw IllegalArgumentException("Unknown candle alignment: " + s);
117 
118  return std::nullopt;
119  }
120 
121  /**
122  * Returns candle alignment of the given candle symbol string.
123  * The result is CandleAlignment::DEFAULT if the symbol does not have candle alignment attribute or std::nullopt if
124  * there is no supported attribute's value.
125  *
126  * @param symbol The candle symbol string.
127  * @return candle alignment of the given candle symbol string or std::nullopt if there is no supported attribute's
128  * value.
129  */
130  static std::optional<std::reference_wrapper<const CandleAlignment>>
131  getAttributeForSymbol(const std::string &symbol) noexcept {
132  auto stringOpt = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
133 
134  return !stringOpt ? DEFAULT : parse(stringOpt.value());
135  }
136 
137  /**
138  * Returns candle symbol string with the normalized representation of the candle alignment attribute.
139  *
140  * @param symbol The candle symbol string.
141  * @return candle symbol string with the normalized representation of the the candle alignment attribute.
142  */
143  static DXFCPP_CXX20_CONSTEXPR_STRING std::string normalizeAttributeForSymbol(const std::string &symbol) noexcept {
144  auto a = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
145 
146  if (!a) {
147  return symbol;
148  }
149 
150  auto other = parse(a.value());
151 
152  if (other.has_value()) {
153  if (other.value() == DEFAULT) {
154  return MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY);
155  }
156 
157  if (a.value() != other.value().get().toString()) {
158  return MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY,
159  other.value().get().toString());
160  }
161  }
162 
163  return symbol;
164  }
165 };
166

171  return std::hash<std::string>{}(candleAlignment.toString());
172  }
173 };
