22/**
23 * Base class for common fields of Trade} and TradeETH events.
24 * Trade events represent the most recent information that is available about the last trade on the market at any
25 * given moment of time.
26 *
27 * Trade event represents last trade information for <b>regular trading hours</b> (RTH) with an official volume and
28 * turnover <b>for the whole trading day</b> identified by @ref TradeBase::getDayId() "dayId" field.
29 *
30 * TradeETH event is defined only for symbols (typically stocks and ETFs) with a designated
31 * <b>extended trading hours</b> (ETH, pre market and post market trading sessions). It represents last trade price
32 * during ETH and also accumulated volume and turnover during ETH for current trading day.
33 */
34class DXFCPP_EXPORT TradeBase : public MarketEvent, public LastingEvent {
35 friend struct EventMapper;
36
37 protected:
38 static constexpr std::uint64_t SECONDS_SHIFT = 32ULL;
39 static constexpr std::uint64_t MILLISECONDS_SHIFT = 22ULL;
40 static constexpr std::uint64_t MILLISECONDS_MASK = 0x3ffULL;
41
42 /*
43 * Flags property has several significant bits that are packed into an integer in the following way:
44 * 31..4 3 2 1 0
45 * +--------+----+----+----+----+
46 * | | Direction | ETH|
47 * +--------+----+----+----+----+
48 */
49
50 // DIRECTION values are taken from Direction enum.
51 static constexpr std::int32_t DIRECTION_MASK = 7;
52 static constexpr std::int32_t DIRECTION_SHIFT = 1;
53
54 static constexpr std::int32_t ETH = 1;
55
56 struct TradeBaseData {
57 std::int64_t timeSequence{};
58 std::int32_t timeNanoPart{};
59 std::int16_t exchangeCode{};
60 double price = math::NaN;
61 double change = math::NaN;
62 double size = math::NaN;
63 std::int32_t dayId{};
64 double dayVolume{};
65 double dayTurnover{};
66 std::int32_t flags{};
67 };

103 * Changes time and sequence of last trade.
104 * <b>Do not use this method directly.</b>
105 * Change @ref TradeBase::setTime() "time" and/or @ref TradeBase::setSequence() "sequence".

114 /**
115 * Returns time of the last trade.
116 * Time is measured in milliseconds between the current time and midnight, January 1, 1970 UTC.
117 *
118 * @return time of the last trade.
119 */
120 std::int64_t getTime() const noexcept {
121 return sar(tradeBaseData_.timeSequence, SECONDS_SHIFT) * 1000 +
122 andOp(sar(tradeBaseData_.timeSequence, MILLISECONDS_SHIFT), MILLISECONDS_MASK);
123 }
124
125 /**
126 * Changes time of the last trade.
127 * Time is measured in milliseconds between the current time and midnight, January 1, 1970 UTC.
128 *
129 * @param time time of the last trade.
130 */
131 void setTime(std::int64_t time) noexcept {
132 tradeBaseData_.timeSequence = orOp(
133 sal(static_cast<std::int64_t>(time_util::getSecondsFromTime(time)), SECONDS_SHIFT),
134 orOp(sal(static_cast<std::int64_t>(time_util::getMillisFromTime(time)), MILLISECONDS_MASK), getSequence()));
135 }
136
137 /**
138 * Returns time of the last trade in nanoseconds.
139 * Time is measured in nanoseconds between the current time and midnight, January 1, 1970 UTC.
140 *
141 * @return time of the last trade in nanoseconds.
142 */
143 std::int64_t getTimeNanos() const noexcept {
144 return time_nanos_util::getNanosFromMillisAndNanoPart(getTime(), tradeBaseData_.timeNanoPart);
145 }
146
147 /**
148 * Changes time of the last trade.
149 * Time is measured in nanoseconds between the current time and midnight, January 1, 1970 UTC.
150 *
151 * @param timeNanos time of the last trade in nanoseconds.
152 */
153 void setTimeNanos(std::int64_t timeNanos) noexcept {
154 setTime(time_nanos_util::getMillisFromNanos(timeNanos));
155 tradeBaseData_.timeNanoPart = time_nanos_util::getNanoPartFromNanos(timeNanos);
156 }

172 std::int32_t getTimeNanoPart() const noexcept {
173 return tradeBaseData_.timeNanoPart;
174 }

176 /**
177 * Returns sequence number of the last trade to distinguish trades that have the same
178 * @ref TradeBase::getTime() "time". This sequence number does not have to be unique and
179 * does not need to be sequential. Sequence can range from 0 to TradeBase::MAX_SEQUENCE.
180 *
181 * @return sequence of the last trade.
182 */
183 std::int32_t getSequence() const noexcept {
184 return static_cast<std::int32_t>(andOp(tradeBaseData_.timeSequence, MAX_SEQUENCE));
185 }
186
187 /**
188 * Changes @ref TradeBase::getSequence() "sequence number" of the last trade.
189 *
190 * @param sequence the sequence.
191 * @see TradeBase::getSequence()
192 */
193 void setSequence(std::int32_t sequence) noexcept {
194 // TODO: Improve error handling [EN-8232]
195 assert(sequence >= 0 && sequence <= MAX_SEQUENCE);
196
197 tradeBaseData_.timeSequence = orOp(andOp(tradeBaseData_.timeSequence, ~MAX_SEQUENCE), sequence);
198 }
