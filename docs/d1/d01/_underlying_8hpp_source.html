24/**
25 * Underlying event is a snapshot of computed values that are available for an option underlying
26 * symbol based on the option prices on the market.
27 * It represents the most recent information that is available about the corresponding values on
28 * the market at any given moment of time.
29 *
30 * <h3><a name="eventFlagsSection">Event flags, transactions and snapshots</a></h3>
31 *
32 * Some Underlying sources provide a consistent view of the set of known Underlying events.
33 * The corresponding information is carried in @ref ::getEventFlags() "eventFlags" property.
34 * The logic behind this property is detailed in IndexedEvent class documentation.
35 * Multiple event sources for the same symbol are not supported for Underlying, thus
36 * @ref ::getSource() "source" property is always @ref IndexedEventSource::DEFAULT "DEFAULT".
37 *
38 * <p>TimeSeriesEventModel class handles all the snapshot and transaction logic and conveniently represents
39 * a list current of time-series events order by their @ref ::getTime() "time".
40 *
41 * <h3>Publishing Underlying</h3>
42 *
43 * Publishing of Underlying events follows the general rules explained in TimeSeriesEvent class
44 * documentation.
45 *
46 * <h3>Implementation details</h3>
47 *
48 * This event is implemented on top of QDS record `Underlying`.
49 */
50class DXFCPP_EXPORT Underlying final: public MarketEvent, public TimeSeriesEvent, public LastingEvent {
51 friend struct EventMapper;

53 /**
54 * Maximum allowed sequence value.
55 *
56 * @see ::setSequence()
57 */
58 static constexpr std::uint32_t MAX_SEQUENCE = (1U << 22U) - 1U;
59 static constexpr std::uint64_t SECONDS_SHIFT = 32ULL;
60 static constexpr std::uint64_t MILLISECONDS_SHIFT = 22ULL;
61 static constexpr std::uint64_t MILLISECONDS_MASK = 0x3ffULL;
62
63 /**
64 * EventFlags property has several significant bits that are packed into an integer in the following way:
65 * <br>
66 * <pre><tt>
67 * 31..7 6 5 4 3 2 1 0
68 * +--------+----+----+----+----+----+----+----+
69 * | | SM | | SS | SE | SB | RE | TX |
70 * +--------+----+----+----+----+----+----+----+
71 * </tt></pre>
72 */

114 * Returns unique per-symbol index of this event.

122 /**
123 * Changes unique per-symbol index of this event.
124 * The index is composed of @ref ::getTime() "time" and @ref ::getSequence() "sequence" and
125 * invocation of this method changes time and sequence.
126 * <b>Do not use this method directly.</b>
127 * Change @ref ::setTime() "time" and/or @ref ::setSequence() "sequence".
128 *
129 * @param index the event index.
130 * @see ::getIndex()

132 void setIndex(std::int64_t index) override { data_.index = index; }
133
134 /**
135 * Returns timestamp of the event in milliseconds.
136 *
137 * @return timestamp of the event in milliseconds
138 */
139 std::int64_t getTime() const override {
140 return sar(data_.index, SECONDS_SHIFT) * 1000 + andOp(sar(data_.index, MILLISECONDS_SHIFT), MILLISECONDS_MASK);
141 }
142
143 /**
144 * Changes timestamp of the event in milliseconds.
145 *
146 * @param time timestamp of the event in milliseconds.
147 * @see ::getTime()
148 */
149 void setTime(std::int64_t time) {
150 data_.index = orOp(orOp(sal(time_util::getSecondsFromTime(time), SECONDS_SHIFT),
151 sal(time_util::getMillisFromTime(time), MILLISECONDS_SHIFT)),
152 getSequence());
153 }
154
155 /**
156 * Returns the sequence number of this event to distinguish events that have the same @ref ::getTime() "time".
157 * This sequence number does not have to be unique and does not need to be sequential.
158 * Sequence can range from 0 to ::MAX_SEQUENCE.
159 *
160 * @return The sequence number of this event
161 */
162 std::int32_t getSequence() const { return static_cast<std::int32_t>(andOp(data_.index, MAX_SEQUENCE)); }
163
164 /**
165 * Changes @ref ::getSequence() "sequence number" of this event.
166 * @param sequence the sequence.
167 * @see ::getSequence()
168 */
169 void setSequence(int sequence) {
170 // TODO: Improve error handling
171 assert(sequence >= 0 && sequence <= MAX_SEQUENCE);
172
173 data_.index = orOp(andOp(data_.index, ~MAX_SEQUENCE), sequence);
174 }
