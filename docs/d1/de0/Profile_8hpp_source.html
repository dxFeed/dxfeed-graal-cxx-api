23 /**
24  * Profile information snapshot that contains security instrument description.
25  * It represents the most recent information that is available about the traded security

31  /*
32  * Flags property has several significant bits that are packed into an integer in the following way:
33  * 31..4 3 2 1 0
34  * +--------+----+----+----+----+
35  * | | SSR | Status |
36  * +--------+----+----+----+----+
37  */
38 
39  // SSR values are taken from ShortSaleRestriction enum.
40  static const std::uint32_t SSR_MASK = 3;
41  static const std::uint32_t SSR_SHIFT = 2;
42 
43  // STATUS values are taken from TradingStatus enum.
44  static const std::uint32_t STATUS_MASK = 3;
45  static const std::uint32_t STATUS_SHIFT = 0;
