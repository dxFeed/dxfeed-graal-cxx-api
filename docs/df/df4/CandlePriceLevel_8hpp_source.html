18 /**
19  * Candle price level attribute of CandleSymbol defines how candles shall be aggregated in respect to
20  * price interval. The negative or infinite values of price interval are treated as exceptional.
21  * <ul>
22  * <li>Price interval may be equal to zero. It means every unique price creates a particular candle
23  * to aggregate all events with this price for the chosen CandlePeriod.
24  * <li>Non-zero price level creates sequence of intervals starting from 0:
25  * ...,[-pl;0),[0;pl),[pl;2*pl),...,[n*pl,n*pl+pl). Events aggregated by chosen CandlePeriod and price
26  * intervals.
27  * </ul>
28  *
29  * <h3>Implementation details</h3>
30  * <p>
31  * This attribute is encoded in a symbol string with
32  * @ref MarketEventSymbols::getAttributeStringByKey() "MarketEventSymbols.getAttributeStringByKey",
33  * @ref MarketEventSymbols::changeAttributeStringByKey() "changeAttributeStringByKey", and
34  * @ref MarketEventSymbols::removeAttributeStringByKey() "removeAttributeStringByKey" methods.
35  * The key to use with these methods is available via CandlePriceLevel::ATTRIBUTE_KEY constant.
36  * The value that this key shall be set to is equal to the corresponding CandlePriceLevel::toString()
37  */
38 struct DXFCPP_EXPORT CandlePriceLevel : public CandleSymbolAttribute {
39  /**
40  * Default price level corresponds to `NaN (std::numeric_limits<double>::quiet_NaN())`
41  */
42  static const CandlePriceLevel DEFAULT;
43 
44  /**
45  * The attribute key that is used to store the value of `CandlePriceLevel` in a symbol string using methods
46  * of MarketEventSymbols class.
47  * The value of this constant is "pl".
48  * The value that this key shall be set to is equal to the corresponding CandlePriceLevel::toString()
49  */
50  static const std::string ATTRIBUTE_KEY;
51 
52  double value_{};
53 
54  explicit CandlePriceLevel(double value) noexcept : value_{value} {
55  }
56 
57  public:
58  CandlePriceLevel() noexcept = default;
59  virtual ~CandlePriceLevel() noexcept = default;
60 
61  /**
62  * Returns price level value. For example, the value of `1` represents `[0;1), [1;2)` and so on intervals
63  * to build candles.
64  *
65  * @return price level value.
66  */
67  double getValue() const noexcept {
68  return value_;
69  }
70 
71  /**
72  * Returns string representation of this price level.
73  * The string representation is composed of value.
74  * This string representation can be converted back into object with CandlePriceLevel::parse() method.
75  *
76  * @return string representation of this price level.
77  */
78  std::string toString() const noexcept {
79  if (math::equals(value_, static_cast<std::int64_t>(value_))) {
80  return std::to_string(static_cast<std::int64_t>(value_));
81  }
82 
83  return std::to_string(value_);
84  }
85 
86  bool operator==(const CandlePriceLevel &candlePriceLevel) const noexcept {
87  return math::equals(value_, candlePriceLevel.getValue());
88  }
89 
90  /**
91  * Returns candle event symbol string with this candle price level set.
92  *
93  * @param symbol original candle event symbol.
94  * @return candle event symbol string with this candle price level set.
95  */
96  std::string changeAttributeForSymbol(const std::string &symbol) const noexcept override {
97  return *this == DEFAULT ? MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY)
98  : MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY, toString());
99  }
100 
101  /**
102  * Parses string representation of candle price level into object.
103  * Any string that was returned by CandlePriceLevel::toString() can be parsed
104  * and case is ignored for parsing.
105  *
106  * @param s string representation of candle price level.
107  * @return candle price level or std::nullopt if the string representation is invalid or value is invalid
108  */
109  static std::optional<CandlePriceLevel> parse(const std::string &s) noexcept {
110  try {
111  return valueOf(std::stod(s));
112  } catch (...) {
113  return std::nullopt;
114  }
115  }
116 
117  /**
118  * Returns candle price level with the given value.
119  *
120  * @param value candle price level value.
121  * @return candle price level with the given value and type or std::nullopt if value is invalid
122  */
123  static std::optional<CandlePriceLevel> valueOf(double value) noexcept {
124  if (std::isinf(value) || (value == 0.0 && std::signbit(value))) {
125  // TODO: error handling [EN-8232] throw IllegalArgumentException("Incorrect candle price level: " + value);
126 
127  return std::nullopt;
128  }
129 
130  return std::isnan(value) ? DEFAULT : CandlePriceLevel(value);
131  }
132 
133  /**
134  * Returns candle price level of the given candle symbol string.
135  * The result is CandlePriceLevel::DEFAULT if the symbol does not have candle price level attribute.
136  *
137  * @param symbol candle symbol string.
138  * @return candle price level of the given candle symbol string or std::nullopt if there is no supported attribute's
139  */
140  static std::optional<CandlePriceLevel> getAttributeForSymbol(const std::string &symbol) noexcept {
141  auto stringOpt = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
142 
143  return !stringOpt ? DEFAULT : parse(stringOpt.value());
144  }
145 
146  /**
147  * Returns candle symbol string with the normalized representation of the candle price level attribute.
148  *
149  * @param symbol candle symbol string.
150  * @return candle symbol string with the normalized representation of the the candle price level attribute.
151  */
152  static DXFCPP_CXX20_CONSTEXPR_STRING std::string normalizeAttributeForSymbol(const std::string &symbol) noexcept {
153  auto a = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
154 
155  if (!a) {
156  return symbol;
157  }
158 
159  auto other = parse(a.value());
160 
161  if (other.has_value()) {
162  if (other.value() == DEFAULT) {
163  return MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY);
164  }
165 
166  if (a.value() != other.value().toString()) {
167  return MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY, other.value().toString());
168  }
169  }
170 
171  return symbol;
172  }
173 };

179  return std::hash<double>{}(candlePriceLevel.getValue());
180  }
181 };
