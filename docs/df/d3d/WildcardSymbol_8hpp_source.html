15 /**
16  * Represents [wildcard] subscription to all events of the specific event type.
17  * The @ref WildcardSymbol::ALL constant can be added to any DXFeedSubscription instance with
18  * @ref DXFeedSubscription::addSymbols() "addSymbols" method to the effect of subscribing to all possible event symbols. The
19  * corresponding subscription will start receiving all published events of the corresponding types.
20  */
21 struct DXFCPP_EXPORT WildcardSymbol final {
22  /**
23  * Symbol prefix that is reserved for wildcard subscriptions.
24  * Any subscription starting with "*" is ignored with the exception of WildcardSymbol subscription.
25  */
26  static const std::string RESERVED_PREFIX;
27 
28  /**
29  * Represents [wildcard] subscription to all events of the specific event type.
30  *
31  * <p><b>NOTE:</b> Wildcard subscription can create extremely high network and CPU load for certain kinds of
32  * high-frequency events like quotes. It requires a special arrangement on the side of upstream data provider and
33  * is disabled by default in upstream feed configuration. Make that sure you have adequate resources and understand
34  * the impact before using it. It can be used for low-frequency events only (like Forex quotes), because each
35  * instance of DXFeedSubscription processes events in a single thread and there is no provision to load-balance
36  * wildcard subscription amongst multiple threads. Contact your data provider for the corresponding configuration
37  * arrangement if needed.
38  *
39  * @see WildcardSymbol
40  */
41  static const WildcardSymbol ALL;
42 
43  private:
44  std::string symbol_;
45 
46  WildcardSymbol(const std::string &symbol) noexcept : symbol_{symbol} {
47  }
48 
49  public:
50  WildcardSymbol(const WildcardSymbol &) noexcept = default;
51  WildcardSymbol(WildcardSymbol &&) noexcept = default;
52  WildcardSymbol &operator=(const WildcardSymbol &) noexcept = default;
53  WildcardSymbol &operator=(WildcardSymbol &&) noexcept = default;
54  WildcardSymbol() noexcept = default;
55  virtual ~WildcardSymbol() noexcept = default;
56 
57  const std::string &getSymbol() const noexcept {
58  return symbol_;
59  }
60 
61  /**
62  * Allocates memory for the dxFeed Graal SDK structure (recursively if necessary).
63  * Fills the dxFeed Graal SDK structure's fields by the data of the current entity (recursively if necessary).
64  * Returns the pointer to the filled structure.
65  *
66  * @return The pointer to the filled dxFeed Graal SDK structure
67  */
68  void *toGraal() const noexcept;
69 
70  /**
71  * Releases the memory occupied by the dxFeed Graal SDK structure (recursively if necessary).
72  *
73  * @param graalNative The pointer to the dxFeed Graal SDK structure.
74  */
75  static void freeGraal(void *graalNative) noexcept;
76 
77  static const WildcardSymbol &fromGraal(void *graalNative) noexcept;
78 
79  /**
80  * Returns string representation of this wildcard subscription symbol.
81  *
82  * @return string representation of this wildcard subscription symbol.
83  */
84  std::string toString() const noexcept {
85  if constexpr (Debugger::isDebug) {
86  return "WildcardSymbol{" + symbol_ + "}";
87  } else {
88  return symbol_;
89  }
90  }
91 
92  bool operator==(const WildcardSymbol &wildcardSymbol) const {
93  return symbol_ == wildcardSymbol.symbol_;
94  }
95 
96  bool operator<(const WildcardSymbol &wildcardSymbol) const {
97  return symbol_ < wildcardSymbol.symbol_;
98  }
99 };
