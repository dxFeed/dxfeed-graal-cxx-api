18 /**
19  * Exchange attribute of CandleSymbol defines exchange identifier where data is
20  * taken from to build the candles.
21  *
22  * <h3>Implementation details</h3>
23  *
24  * This attribute is encoded in a symbol string with
25  * @ref MarketEventSymbols::getExchangeCode() "MarketEventSymbols.getExchangeCode" and
26  * @ref MarketEventSymbols::changeExchangeCode() "changeExchangeCode" methods.
27  */
28 struct DXFCPP_EXPORT CandleExchange : public CandleSymbolAttribute {
29  /**
30  * Composite exchange where data is taken from all exchanges.
31  */
32  static const CandleExchange COMPOSITE;
33 
34  /**
35  * Default exchange is CandleExchange::COMPOSITE.
36  */
37  static const CandleExchange DEFAULT;
38 
39  private:
40  const char exchangeCode_{};
41 
42  explicit CandleExchange(char exchangeCode) noexcept : exchangeCode_{exchangeCode} {
43  }
44 
45  public:
46  CandleExchange() noexcept = default;
47 
48  /**
49  * Returns exchange code. It is `'\0'` for CandleExchange::COMPOSITE exchange.
50  * @return exchange code.
51  */
52  char getExchangeCode() const noexcept {
53  return exchangeCode_;
54  }
55 
56  /**
57  * Returns candle event symbol string with this exchange set.
58  *
59  * @param symbol The original candle event symbol.
60  * @return candle event symbol string with this exchange set.
61  */
62  std::string changeAttributeForSymbol(const std::string &symbol) const noexcept override {
63  return MarketEventSymbols::changeExchangeCode(symbol, exchangeCode_);
64  }
65 
66  /**
67  * Returns string representation of this exchange.
68  * It is the string `"COMPOSITE"` for CandleExchange::COMPOSITE exchange or
69  * exchange character otherwise.
70  *
71  * @return string representation of this exchange.
72  */
73  std::string toString() const noexcept {
74  return exchangeCode_ == '\0' ? "COMPOSITE" : std::string(1, exchangeCode_);
75  }
76 
77  bool operator==(const CandleExchange &candleExchange) const noexcept {
78  return exchangeCode_ == candleExchange.exchangeCode_;
79  }
80 
81  /**
82  * Returns exchange attribute object that corresponds to the specified exchange code character.
83  *
84  * @param exchangeCode The exchange code character.
85  * @return exchange attribute object.
86  */
87  static CandleExchange valueOf(char exchangeCode) noexcept {
88  return exchangeCode == '\0' ? COMPOSITE : CandleExchange(exchangeCode);
89  }
90 
91  /**
92  * Returns exchange attribute object of the given candle symbol string.
93  * The result is CandleExchange::DEFAULT if the symbol does not have exchange attribute.
94  *
95  * @param symbol candle symbol string.
96  * @return exchange attribute object of the given candle symbol string.
97  */
98  static CandleExchange getAttributeForSymbol(const std::string &symbol) noexcept {
99  return valueOf(MarketEventSymbols::getExchangeCode(symbol));
100  }
101 };
102 
103 } // namespace dxfcpp

105 template <> struct std::hash<dxfcpp::CandleExchange> {
106  std::size_t operator()(const dxfcpp::CandleExchange &candleExchange) const noexcept {
107  return static_cast<std::size_t>(candleExchange.getExchangeCode());
108  }
109 };
