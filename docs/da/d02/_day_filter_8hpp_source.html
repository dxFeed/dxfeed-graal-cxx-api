57/**
58 * A filter for days used by various search methods.
59 * This class provides predefined filters for certain Day attributes,
60 * although users can create their own filters to suit their needs.
61 * <p>
62 * Please note that days can be either trading or non-trading, and this distinction can be
63 * either based on rules (e.g. weekends) or dictated by special occasions (e.g. holidays).
64 * Different filters treat this distinction differently - some accept only trading days,
65 * some only non-trading, and some ignore type of day altogether.
66 */
67struct DXFCPP_EXPORT DayFilter {
68 friend struct Day;

77 static const DayFilter NON_TRADING;
78
79 /// Accepts holidays only - those with <code>(Day::isHoliday() == true)</code>.
80 static const DayFilter HOLIDAY;
81
82 /// Accepts short days only - those with <code>(Day::isShortDay() == true)</code>.
83 static const DayFilter SHORT_DAY;
84
85 /// Accepts Mondays only - those with <code>(Day::getDayOfWeek() == 1)</code>.
86 static const DayFilter MONDAY;
87
88 /// Accepts Tuesdays only - those with <code>(Day::getDayOfWeek() == 2)</code>.
89 static const DayFilter TUESDAY;
90
91 /// Accepts Wednesdays only - those with <code>(Day::getDayOfWeek() == 3)</code>.
92 static const DayFilter WEDNESDAY;

94 /// Accepts Thursdays only - those with <code>(Day::getDayOfWeek() == 4)</code>.
95 static const DayFilter THURSDAY;

113 DayFilter(DayFilter &&) noexcept = delete;
114 DayFilter &operator=(const DayFilter &) = delete;
115 DayFilter &operator=(const DayFilter &&) noexcept = delete;

117 private:
118 DayFilterEnum code_;
119
120 std::string name_;
121
122 /// Bitwise mask of required days (2 for Monday, 4 for Tuesday, 128 for Sunday), <code>0</code> if not relevant.
123 std::uint32_t dayOfWeekMask_;
124 /// Required holiday flag, Tristate::NONE if not relevant.
125 Tristate holiday_;
126 /// Required short day flag, Tristate::NONE if not relevant.
127 Tristate shortDay_;
128 /// Required trading flag, Tristate::NONE if not relevant.
129 Tristate trading_;
130
131 JavaObjectHandle<DayFilter> handle_;
132
133 /**
134 * Creates filter with specified conditions.
135 * <p>
136 * The <code>dayOfWeekMask</code> is a bitwise mask with individual bits for each day of week.
137 * For the day of week number N the N'th bit is used - the Day will be accepted if corresponding bit is set.
138 * If no bits is set (if mask is zero) then day of week attribute is ignored (any value is accepted).
139 * <p>
140 * The boolean parameters specify what value corresponding attributes should have.
141 * If some parameter is Tristate::NONE then corresponding attribute is ignored (any value is accepted).
142 *
143 * @param code The enum code
144 * @param name The enum name
145 * @param dayOfWeekMask The bitwise mask of required days (2 for Monday, 4 for Tuesday, 128 for Sunday),
146 * <code>0</code> if not relevant
147 * @param holiday The required holiday flag, Tristate::NONE if not relevant
148 * @param shortDay The required short day flag, Tristate::NONE if not relevant
149 * @param trading The required trading flag, Tristate::NONE if not relevant
150 */
151 DayFilter(DayFilterEnum code, std::string name, std::uint32_t dayOfWeekMask, Tristate holiday, Tristate shortDay,
152 Tristate trading) noexcept;
153};
