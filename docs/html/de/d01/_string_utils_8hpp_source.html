145class IsIEqual {
146 std::locale locale_;
147
148 public:
149 explicit IsIEqual(const std::locale &locale = std::locale()) : locale_{locale} {
150 }
151
152 template <typename T, typename U> bool operator()(const T &t, const U &u) const {
153 return std::tolower<T>(t, locale_) == std::tolower<U>(u, locale_);
154 }
155};
156
157} // namespace detail

159template <typename Range1, typename Range2, typename Predicate>
160inline bool equals(const Range1 &first, const Range2 &second, Predicate cmp) {
161 auto firstIt = std::begin(first);
162 auto secondIt = std::begin(second);
163
164 for (; firstIt != std::end(first) && secondIt != std::end(second); ++firstIt, ++secondIt) {
165 if (!cmp(*firstIt, *secondIt)) {
166 return false;
167 }
168 }
169
170 return (secondIt == std::end(second)) && (firstIt == std::end(first));
171}

173DXFCPP_EXPORT inline bool iEquals(const std::string &first, const std::string &second) noexcept {
174 const std::locale &locale = std::locale();
175
176 return equals(first, second, detail::IsIEqual(locale));
177}

179DXFCPP_EXPORT inline std::size_t icHash(const std::string &s) noexcept {
180 const std::locale &locale = std::locale();
181 std::string result{};
182
183 for (auto c : s) {
184 result += std::tolower(c, locale);
185 }
186
187 return std::hash<std::string>()(result);
188}
