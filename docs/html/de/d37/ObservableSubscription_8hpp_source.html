21 /**
22  * Observable set of subscription symbols.
23  */
24 struct DXFCPP_EXPORT ObservableSubscription {
25  virtual ~ObservableSubscription() = default;
26 
27  /**
28  * @return `true` if this subscription is closed.
29  * The observable subscription that is returned as a result of DXPublisher::getSubscription() is closed when the
30  * corresponding DXEndpoint is closed.
31  */
32  virtual bool isClosed() = 0;
33 
34  /**
35  * @return a set of event types for this subscription. The resulting set cannot be modified.
36  * The observable subscription that is returned as a result of DXPublisher::getSubscription() has a singleton set
37  * of event types.
38  */
39  virtual std::unordered_set<EventTypeEnum> getEventTypes() = 0;
40 
41  /**
42  * Returns `true` if this subscription contains the corresponding event type.
43  *
44  * @param eventType The type of event that is checked.
45  * @return `true` if this subscription contains the corresponding event type.
46  * @see #getEventTypes()
47  */
48  virtual bool containsEventType(const EventTypeEnum &eventType) = 0;
49 
50  /**
51  * Adds subscription change listener. This method does nothing if subscription is closed.
52  * Otherwise, it installs the corresponding listener and immediately
53  * invokes ObservableSubscriptionChangeListener::symbolsAdded() on the given listener while holding the lock for
54  * this subscription. This way the given listener synchronously receives existing subscription state and is
55  * synchronously notified on all changes in subscription afterwards.
56  *
57  * @param listener The subscription change listener.
58  * @return The listener id
59  */
60  virtual std::size_t addChangeListener(std::shared_ptr<ObservableSubscriptionChangeListener> listener) = 0;
61 
62  /**
63  * Removes subscription change listener by id. This method does nothing if the listener with the given id was not
64  * installed or was already removed as subscription change listener for this subscription. Otherwise it removes the
65  * corresponding listener and immediately invokes ObservableSubscriptionChangeListener::subscriptionClosed() on the
66  * given listener while holding the lock for this subscription.
67  *
68  * @param id The subscription change listener id.
69  */
70  virtual void removeChangeListener(std::size_t id) = 0;
71 };
