15/**
16 * Helper class to compose and parse symbols for market events.
17 *
18 * <h3>Regional symbols</h3>
19 *
20 * Regional symbol subscription receives events only from a designated exchange, marketplace, or venue
21 * instead of receiving composite events from all venues (by default). Regional symbol is composed from a
22 * <i>base symbol</i>, ampersand character ('&amp;'), and an exchange code character. For example,
23 * <ul>
24 * <li>"SPY" is the symbol for composite events for SPDR S&amp;P 500 ETF from all exchanges,
25 * <li>"SPY&amp;N" is the symbol for event for SPDR S&amp;P 500 ETF that originate only from NYSE marketplace.
26 * </ul>
27 *
28 * <h3>Symbol attributes</h3>
29 *
30 * Market event symbols can have a number of attributes attached to then in curly braces
31 * with `<key>=<value>` paris separated by commas. For example,
32 * <ul>
33 * <li>"SPY{price=bid}" is the market symbol "SPY" with an attribute key "price" set to value "bid".
34 * <li>"SPY(=5m,tho=true}" is the market symbol "SPY" with two attributes. One has an empty key and
35 * value "5m", while the other has key "tho" and value "true".
36 * </ul>
37 * The methods in this class always maintain attribute keys in alphabetic order.
38 */
39struct DXFCPP_EXPORT MarketEventSymbols {
40 /**
41 * Returns `true` is the specified symbol has the exchange code specification.
42 *
43 * @param symbol The symbol.
44 * @return `true` is the specified symbol has the exchange code specification.
45 */
46 static bool hasExchangeCode(const std::string &symbol) noexcept {
47 return hasExchangeCodeInternal(symbol, getLengthWithoutAttributesInternal(symbol));
48 }
49
50 /**
51 * Returns exchange code of the specified symbol or `'\0'` if none is defined.
52 *
53 * @param symbol The symbol.
54 * @return exchange code of the specified symbol or `'\0'` if none is defined.
55 */
56 static char getExchangeCode(const std::string &symbol) noexcept {
57 return hasExchangeCode(symbol) ? symbol[getLengthWithoutAttributesInternal(symbol) - 1] : '\0';
58 }
59
60 /**
61 * Changes exchange code of the specified symbol or removes it if new exchange code is `'\0'`.
62 *
63 * @param symbol The old symbol.
64 * @param exchangeCode The new exchange code.
65 * @return new symbol with the changed exchange code.
66 */
67 static DXFCPP_CXX20_CONSTEXPR_STRING std::string changeExchangeCode(const std::string &symbol,
68 char exchangeCode) noexcept {
69 auto i = getLengthWithoutAttributesInternal(symbol);
70 auto result = exchangeCode == '\0' ? getBaseSymbolInternal(symbol, i)
71 : getBaseSymbolInternal(symbol, i) + EXCHANGE_SEPARATOR + exchangeCode;
72 return i == symbol.length() ? result : result + symbol.substr(i);
73 }
74
75 /**
76 * Returns base symbol without exchange code and attributes.
77 *
78 * @param symbol symbol.
79 * @return base symbol without exchange code and attributes.
80 */
81 static DXFCPP_CXX20_CONSTEXPR_STRING std::string getBaseSymbol(const std::string &symbol) noexcept {
82 return getBaseSymbolInternal(symbol, getLengthWithoutAttributesInternal(symbol));
83 }
84
85 /**
86 * Changes base symbol while leaving exchange code and attributes intact.
87 * @param symbol old symbol.
88 * @param baseSymbol new base symbol.
89 * @return new symbol with new base symbol and old symbol's exchange code and attributes.
90 */
91 static DXFCPP_CXX20_CONSTEXPR_STRING std::string changeBaseSymbol(const std::string &symbol,
92 const std::string &baseSymbol) noexcept {
93 auto i = getLengthWithoutAttributesInternal(symbol);
94
95 return hasExchangeCodeInternal(symbol, i) ? baseSymbol + EXCHANGE_SEPARATOR + symbol[i - 1] + symbol.substr(i)
96 : i == symbol.length() ? baseSymbol
97 : baseSymbol + symbol.substr(i);
98 }
99
100 /**
101 * Returns value of the attribute with the specified key.
102 * The result is std::nullopt if attribute with the specified key is not found.
103 *
104 * @param symbol symbol.
105 * @param key attribute key.
106 * @return value of the attribute with the specified key | std::nullopt if attribute with the specified key is not
107 * found.
108 */
109 static DXFCPP_CXX20_CONSTEXPR_STRING std::optional<std::string>
110 getAttributeStringByKey(const std::string &symbol, const std::string &key) noexcept {
111 return getAttributeInternal(symbol, getLengthWithoutAttributesInternal(symbol), key);
112 }
113
114 /**
115 * Changes value of one attribute value while leaving exchange code and other attributes intact.
116 *
117 * @param symbol old symbol.
118 * @param key attribute key.
119 * @param value attribute value.
120 *
121 * @return new symbol with key attribute with the specified value and everything else from the old symbol.
122 */
123 static DXFCPP_CXX20_CONSTEXPR_STRING std::string
124 changeAttributeStringByKey(const std::string &symbol, const std::string &key, const std::string &value) noexcept {
125 auto i = getLengthWithoutAttributesInternal(symbol);
126
127 if (i == symbol.length())
128 return symbol + ATTRIBUTES_OPEN + key + ATTRIBUTE_VALUE + value + ATTRIBUTES_CLOSE;
129 return addAttributeInternal(symbol, i, key, value);
130 }
131
132 /**
133 * Removes one attribute with the specified key while leaving exchange code and other attributes intact.
134 *
135 * @param symbol old symbol.
136 * @param key attribute key.
137 * @return new symbol without the specified key and everything else from the old symbol.
138 */
139 static DXFCPP_CXX20_CONSTEXPR_STRING std::string removeAttributeStringByKey(const std::string &symbol,
140 const std::string &key) noexcept {
141 return removeAttributeInternal(symbol, getLengthWithoutAttributesInternal(symbol), key);
142 }
143
144 private:
145 static constexpr char EXCHANGE_SEPARATOR = '&';
146 static constexpr char ATTRIBUTES_OPEN = '{';
147 static constexpr char ATTRIBUTES_CLOSE = '}';
148 static constexpr char ATTRIBUTES_SEPARATOR = ',';
149 static constexpr char ATTRIBUTE_VALUE = '=';
150
151 static constexpr bool hasExchangeCodeInternal(const std::string &symbol, std::size_t length) noexcept {
152 return length >= 2 && symbol[length - 2] == EXCHANGE_SEPARATOR;
153 }
154
155 static DXFCPP_CXX20_CONSTEXPR_STRING std::string getBaseSymbolInternal(const std::string &symbol,
156 std::size_t length) noexcept {
157 return hasExchangeCodeInternal(symbol, length) ? symbol.substr(0, length - 2) : symbol.substr(0, length);
158 }
159
160 static DXFCPP_CXX20_CONSTEXPR_STRING bool hasAttributesInternal(const std::string &symbol) noexcept {
161 if (symbol.length() >= 3 /* ATTRIBUTES_OPEN + ATTRIBUTES_CLOSE + ATTRIBUTE */ &&
162 symbol[symbol.length() - 1] == ATTRIBUTES_CLOSE) {
163 auto attributesOpenPos = symbol.find_last_of(ATTRIBUTES_OPEN, symbol.length() - 2);
164
165 return attributesOpenPos != std::string::npos && attributesOpenPos < symbol.length() - 2;
166 }
167
168 return false;
169 }
170
171 static DXFCPP_CXX20_CONSTEXPR_STRING std::size_t
172 getLengthWithoutAttributesInternal(const std::string &symbol) noexcept {
173 return hasAttributesInternal(symbol) ? symbol.find_last_of(ATTRIBUTES_OPEN) : symbol.length();
174 }
175
176 static DXFCPP_CXX20_CONSTEXPR_STRING std::optional<std::string> getKeyInternal(const std::string &symbol,
177 std::size_t i) noexcept {
178 try {
179 if (auto found = symbol.find_first_of(ATTRIBUTE_VALUE, i); found != std::string::npos) {
180 return symbol.substr(i, found - i);
181 }
182
183 return std::nullopt;
184 } catch (...) {
185 return std::nullopt;
186 }
187 }
188
189 static DXFCPP_CXX20_CONSTEXPR_STRING std::size_t getNextKeyInternal(const std::string &symbol,
190 std::size_t i) noexcept {
191 auto valuePos = symbol.find_first_of(ATTRIBUTE_VALUE, i) + 1;
192 auto separatorPos = symbol.find_first_of(ATTRIBUTES_SEPARATOR, valuePos);
193
194 return separatorPos == std::string::npos ? symbol.length() : separatorPos + 1;
195 }
196
197 static DXFCPP_CXX20_CONSTEXPR_STRING std::string getValueInternal(const std::string &symbol, std::size_t i,
198 std::size_t j) {
199 auto valueOffset = symbol.find_first_of(ATTRIBUTE_VALUE, i) + 1;
200
201 return symbol.substr(valueOffset, j - 1 - valueOffset);
202 }
203
204 static DXFCPP_CXX20_CONSTEXPR_STRING std::string
205 dropKeyAndValueInternal(const std::string &symbol, std::size_t length, std::size_t i, std::size_t j) noexcept {
206 try {
207 if (j == symbol.length()) {
208 if (i == length + 1) {
209 return symbol.substr(0, length);
210 } else {
211 return symbol.substr(0, i - 1) + symbol.substr(j - 1);
212 }
213 }
214
215 return symbol.substr(0, i) + symbol.substr(j);
216 } catch (...) {
217 return symbol;
218 }
219 }
220
221 static DXFCPP_CXX20_CONSTEXPR_STRING std::optional<std::string>
222 getAttributeInternal(const std::string &symbol, std::size_t lengthWithoutAttributes, const std::string &key) {
223 if (lengthWithoutAttributes == symbol.length()) {
224 return std::nullopt;
225 }
226
227 auto i = lengthWithoutAttributes + 1;
228
229 while (i < symbol.length()) {
230 auto currentKey = getKeyInternal(symbol, i);
231
232 if (!currentKey) {
233 break;
234 }
235
236 auto j = getNextKeyInternal(symbol, i);
237
238 if (key == currentKey.value()) {
239 return getValueInternal(symbol, i, j);
240 }
241
242 i = j;
243 }
244
245 return std::nullopt;
246 }
247
248 static DXFCPP_CXX20_CONSTEXPR_STRING std::string
249 removeAttributeInternal(std::string symbol, std::size_t lengthWithoutAttributes, const std::string &key) noexcept {
250 if (lengthWithoutAttributes == symbol.length()) {
251 return symbol;
252 }
253
254 auto i = lengthWithoutAttributes + 1;
255
256 while (i < symbol.length()) {
257 auto currentKey = getKeyInternal(symbol, i);
258
259 if (!currentKey) {
260 break;
261 }
262
263 auto j = getNextKeyInternal(symbol, i);
264
265 if (key == currentKey.value()) {
266 symbol = dropKeyAndValueInternal(symbol, lengthWithoutAttributes, i, j);
267 } else {
268 i = j;
269 }
270 }
271
272 return symbol;
273 }
274
275 static DXFCPP_CXX20_CONSTEXPR_STRING std::string addAttributeInternal(std::string symbol,
276 std::size_t lengthWithoutAttributes,
277 const std::string &key,
278 const std::string &value) {
279 if (lengthWithoutAttributes == symbol.length()) {
280 return symbol + ATTRIBUTES_OPEN + key + ATTRIBUTE_VALUE + value + ATTRIBUTES_CLOSE;
281 }
282
283 auto i = lengthWithoutAttributes + 1;
284 bool added = false;
285
286 while (i < symbol.length()) {
287 auto currentKey = getKeyInternal(symbol, i);
288
289 if (!currentKey) {
290 break;
291 }
292
293 auto j = getNextKeyInternal(symbol, i);
294 auto cmp = currentKey.value().compare(key);
295
296 if (cmp == 0) {
297 if (added) {
298 // drop, since we've already added this key
299 symbol = dropKeyAndValueInternal(symbol, lengthWithoutAttributes, i, j);
300 } else {
301 // replace value
302 symbol = symbol.substr(0, i) + key + ATTRIBUTE_VALUE + value + symbol.substr(j - 1);
303 added = true;
304 i += key.length() + value.length() + 2;
305 }
306 } else if (cmp > 0 && !added) {
307 // insert value here
308 symbol = symbol.substr(0, i) + key + ATTRIBUTE_VALUE + value + ATTRIBUTES_SEPARATOR + symbol.substr(i);
309 added = true;
310 i += key.length() + value.length() + 2;
311 } else {
312 i = j;
313 }
314 }
315
316 return added ? symbol
317 : symbol.substr(0, i - 1) + ATTRIBUTES_SEPARATOR + key + ATTRIBUTE_VALUE + value +
318 symbol.substr(i - 1);
319 }
320};
