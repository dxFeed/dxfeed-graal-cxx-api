22/**
23 * Quote event is a snapshot of the best bid and ask prices, and other fields that change with each quote.
24 * It represents the most recent information that is available about the best quote on the market
25 * at any given moment of time.
26 *
27 * Bid corresponds to the best (maximal price) order to buy,
28 * ask corresponds to the best (minimal price) order to sell.
29 */
30class DXFCPP_EXPORT Quote final : public MarketEvent, public LastingEvent {
31 friend struct EventMapper;
32
33 static constexpr std::uint64_t MILLISECONDS_SHIFT = 22ULL;
34
35 struct Data {
36 std::int32_t timeMillisSequence{};
37 std::int32_t timeNanoPart{};
38 std::int64_t bidTime{};
39 std::int16_t bidExchangeCode{};
40 double bidPrice = math::NaN;
41 double bidSize = math::NaN;
42 std::int64_t askTime{};
43 std::int16_t askExchangeCode{};
44 double askPrice = math::NaN;
45 double askSize = math::NaN;
46 };

50 void recomputeTimeMillisPart() noexcept {
51 data_.timeMillisSequence =
52 orOp(sal(time_util::getMillisFromTime(std::max(data_.askTime, data_.bidTime)), MILLISECONDS_SHIFT),
53 getSequence());
54 }

162 /**
163 * Returns sequence number of this quote to distinguish quotes that have the same @ref Quote::getTime() "time". This
