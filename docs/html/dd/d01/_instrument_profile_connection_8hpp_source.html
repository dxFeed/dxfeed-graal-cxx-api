17/**
18 * Connects to an instrument profile URL and reads instrument profiles with support of
19 * streaming live updates.
20 * Please see <b>Instrument Profile Format</b> documentation for complete description.
21 */
22class DXFCPP_EXPORT InstrumentProfileConnection final : public SharedEntity {
23 public:
24 /// The alias to a type of shared pointer to the InstrumentProfileConnection object
25 using Ptr = std::shared_ptr<InstrumentProfileConnection>;
26
27 /// The alias to a type of unique pointer to the InstrumentProfileConnection object
28 using Unique = std::unique_ptr<InstrumentProfileConnection>;

30 /**
31 * Instrument profile connection state.
32 */
33 enum class State : std::int32_t {
34 /**
35 * Instrument profile connection is not started yet.
36 * @ref InstrumentProfileConnection::start() "start" was not invoked yet.
37 */
38 NOT_CONNECTED,
39
40 /**
41 * Connection is being established.
42 */
43 CONNECTING,
44
45 /**
46 * Connection was established.
47 */
48 CONNECTED,
49
50 /**
51 * Initial instrument profiles snapshot was fully read (this state is set only once).
52 */
53 COMPLETED,
54
55 /**
56 * Instrument profile connection was @ref InstrumentProfileConnection::close() "closed".
57 */
58 CLOSED
59 };
60
61 private:
62 Id<InstrumentProfileConnection> id_;
63 JavaObjectHandle<InstrumentProfileConnection> handle_;
64 JavaObjectHandle<IpfPropertyChangeListener> stateChangeListenerHandle_;
65 SimpleHandler<void(State, State)> onStateChange_{};
66
67 InstrumentProfileConnection() noexcept;
68
69 struct Impl;
70
71 public:
72 static std::string stateToString(State state) noexcept {
73 switch (state) {
74 case State::NOT_CONNECTED:
75 return "NOT_CONNECTED";
76 case State::CONNECTING:
77 return "CONNECTING";
78 case State::CONNECTED:
79 return "CONNECTED";
80 case State::COMPLETED:
81 return "COMPLETED";
82 case State::CLOSED:
83 return "CLOSED";
84 }
85
86 return "";
87 }
88
89 /**
90 * Creates instrument profile connection with a specified address and collector.
91 * Address may be just "<host>:<port>" of server, URL, or a file path.
92 * The "[update=<period>]" clause can be optionally added at the end of the address to
93 * specify an @ref InstrumentProfileConnection::getUpdatePeriod() "update period" via an address string.
94 * Default update period is 1 minute.
95 *
96 * <p>Connection needs to be @ref InstrumentProfileConnection::start() "started" to begin an actual operation.
97 *
98 * @param address The address.
99 * @param collector The instrument profile collector to push updates into.
100 *
101 * @return new instrument profile connection.
102 */
103 static Ptr createConnection(const std::string &address, InstrumentProfileCollector::Ptr collector) noexcept;
104
105 /**
106 * Returns the address of this instrument profile connection.
107 * It does not include additional options specified as part of the address.
108 *
109 * @return The address of this instrument profile connection.
110 */
111 std::string getAddress() const noexcept;
112
113 /**
114 * Returns update period in milliseconds.
115 * It is period of an update check when the instrument profiles source does not support live updates
116 * and/or when connection is dropped.
117 * Default update period is 1 minute, unless overridden in an
118 * @ref InstrumentProfileConnection::createConnection() "address string".
119 *
120 * @return The update period in milliseconds.
121 */
122 std::int64_t getUpdatePeriod() const noexcept;
123
124 /**
125 * Returns update period in milliseconds as chrono::duration
126 * It is period of an update check when the instrument profiles source does not support live updates
127 * and/or when connection is dropped.
128 * Default update period is 1 minute, unless overridden in an
129 * @ref InstrumentProfileConnection::createConnection() "address string".
130 *
131 * @return The update period in milliseconds as chrono::duration.
132 */
133 std::chrono::milliseconds getUpdatePeriodAsDuration() const noexcept {
134 return std::chrono::milliseconds(getUpdatePeriod());
135 }
136
137 /**
138 * Changes the update period in milliseconds.
139 *
140 * @param updatePeriod The update period in milliseconds.
141 * @see InstrumentProfileConnection::getUpdatePeriod()
142 */
143 void setUpdatePeriod(std::int64_t updatePeriod) const noexcept;
144
145 /**
146 * Changes the update period in milliseconds as chrono::duration.
147 *
148 * @param updatePeriod The update period in milliseconds as chrono::duration.
149 * @see InstrumentProfileConnection::getUpdatePeriod()
150 */
151 void setUpdatePeriod(std::chrono::milliseconds updatePeriod) const noexcept {
152 setUpdatePeriod(updatePeriod.count());
153 }
154
155 /**
156 * Returns state of this instrument profile connections.
157 *
158 * @return The state of this instrument profile connections.
159 */
160 State getState() const noexcept;
161
162 /**
163 * Returns last modification time (in milliseconds) of instrument profiles or zero if it is unknown.
164 * Note, that while the time is represented in milliseconds, the actual granularity of time here is a second.
165 *
166 * @return The last modification time (in milliseconds) of instrument profiles or zero if it is unknown.
167 */
168 std::int64_t getLastModified() const noexcept;
169
170 /**
171 * Starts this instrument profile connection. This connection's state immediately changes to
172 * @ref InstrumentProfileConnection::State::CONNECTING "CONNECTING" and the actual connection establishment proceeds
173 * in the background.
174 */
175 void start() const noexcept;
176
177 /**
178 * Closes this instrument profile connection. This connection's state immediately changes to
179 * @ref InstrumentProfileConnection::State::CLOSED "CLOSED" and the background update procedures are terminated.
180 */
181 void close() const noexcept;
182
183 /**
184 * Adds listener that is notified about changes in @ref InstrumentProfileConnection::getState() "state" property.
185 *
186 * <p>Installed listener can be removed by `id` with InstrumentProfileConnection::removeStateChangeListener method
187 * or by call `InstrumentProfileConnection::onStateChange() -= id`;
188 *
189 * @tparam StateChangeListener The listener type. It can be any callable with signature: `void(State, State)`
190 * @param listener The listener to add
191 * @return the listener id
192 */
193 template <typename StateChangeListener>
194 std::size_t addStateChangeListener(StateChangeListener &&listener) noexcept
195#if __cpp_concepts
196 requires requires {
197 { listener(State{}, State{}) } -> std::same_as<void>;
198 }
199#endif
200 {
201 return onStateChange_ += listener;
202 }
203
204 /**
205 * Removes listener that is notified about changes in @ref InstrumentProfileConnection::getState() "state" property.
206 * It removes the listener that was previously installed with InstrumentProfileConnection::addStateChangeListener
207 * method.
208 *
209 * @param listenerId The listener id to remove
210 */
211 void removeStateChangeListener(std::size_t listenerId) noexcept {
212 onStateChange_ -= listenerId;
213 }
214
215 /**
216 * Returns the onStateChange @ref SimpleHandler<void(ArgTypes...)> "handler" that can be used to add or remove
217 * listeners.
218 *
219 * @return onStateChange handler with `void(State, State)` signature
220 */
221 auto &onStateChange() noexcept {
222 return onStateChange_;
223 }
224
225 /**
226 * Synchronously waits for full first snapshot read with the specified timeout.
227 *
228 * @param timeout The maximum time (in millis) to wait
229 * @return `true` if @ref InstrumentProfileConnection::State::COMPLETED "COMPLETED" state was reached and `false`
230 * if the waiting time elapsed before snapshot was fully read.
231 */
232 bool waitUntilCompleted(std::int64_t timeout) const noexcept;
233
234 /**
235 * Synchronously waits for full first snapshot read with the specified timeout.
236 *
237 * @param timeout The maximum time to wait
238 * @return `true` if @ref InstrumentProfileConnection::State::COMPLETED "COMPLETED" state was reached and `false`
239 * if the waiting time elapsed before snapshot was fully read.
240 */
241 bool waitUntilCompleted(std::chrono::milliseconds timeout) const noexcept {
242 return waitUntilCompleted(timeout.count());
243 }
244};
