17 /**
18  * Action enum for the Full Order Book (FOB) Orders. Action describes business meaning of the Order event:
19  * whether order was added or replaced, partially or fully executed, etc.
20  */
21 struct OrderAction : Enum<OrderAction, std::uint32_t> {
22  using Enum::Enum;
23 
24  /**
25  * Default enum value for orders that do not support "Full Order Book" and for backward compatibility -
26  * action must be derived from other Order fields.
27  *
28  * <p>All Full Order Book related fields for this action will be empty.
29  */
30  static const DXFCPP_EXPORT OrderAction UNDEFINED;
31 
32  /**
33  * New Order is added to Order Book.
34  *
35  * <p>Full Order Book fields:
36  * <ul>
37  * <li>@ref Order::getOrderId() "orderId" - always present</li>
38  * <li>@ref Order::getAuxOrderId() "auxOrderId" - ID of the order replaced by this new order - if available.</li>
39  * <li>Trade fields will be empty</li>
40  * </ul>
41  */
42  static const DXFCPP_EXPORT OrderAction NEW;
43 
44  /**
45  * Order is modified and price-time-priority is not maintained (i.e. order has re-entered Order Book).
46  * Order @ref Order::getEventSymbol() "symbol" and @ref Order::getOrderSide() "side" will remain the same.
47  *
48  * <p>Full Order Book fields:
49  * <ul>
50  * <li>@ref Order::getOrderId() "orderId" - always present</li>
51  * <li>Trade fields will be empty</li>
52  * </ul>
53  */
54  static const DXFCPP_EXPORT OrderAction REPLACE;
55 
56  /**
57  * Order is modified without changing its price-time-priority (usually due to partial cancel by user).
58  * Order's @ref Order::getSize() "size" will contain new updated size.
59  *
60  * <p>Full Order Book fields:
61  * <ul>
62  * <li>@ref Order::getOrderId() "orderId" - always present</li>
63  * <li>Trade fields will be empty</li>
64  * </ul>
65  */
66  static const DXFCPP_EXPORT OrderAction MODIFY;
67 
68  /**
69  * Order is fully canceled and removed from Order Book.
70  * Order's @ref Order::getSize() "size" will be equal to 0.
71  *
72  * <p>Full Order Book fields:
73  * <ul>
74  * <li>@ref Order::getOrderId() "orderId" - always present</li>
75  * <li>@ref Order::getAuxOrderId() "auxOrderId" - ID of the new order replacing this order - if available.</li>
76  * <li>Trade fields will be empty</li>
77  * </ul>
78  */
79  static const DXFCPP_EXPORT OrderAction DELETE;
80 
81  /**
82  * Size is changed (usually reduced) due to partial order execution.
83  * Order's @ref Order::getSize() "size" will be updated to show current outstanding size.
84  *
85  * <p>Full Order Book fields:
86  * <ul>
87  * <li>@ref Order::getOrderId() "orderId" - always present</li>
88  * <li>@ref Order::getAuxOrderId() "auxOrderId" - aggressor order ID, if available</li>
89  * <li>@ref Order::getTradeId() "tradeId" - if available</li>
90  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" -
91  * contain size and price of this execution</li>
92  * </ul>
93  */
94  static const DXFCPP_EXPORT OrderAction PARTIAL;
95 
96  /**
97  * Order is fully executed and removed from Order Book.
98  * Order's {@ref Order#getSize() size} will be equals to 0.
99  *
100  * <p>Full Order Book fields:
101  * <ul>
102  * <li>@ref Order::getOrderId() "orderId" - always present</li>
103  * <li>@ref Order::getAuxOrderId() "auxOrderId" - aggressor order ID, if available</li>
104  * <li>@ref Order::getTradeId() "tradeId" - if available</li>
105  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" -
106  * contain size and price of this execution - always present</li>
107  * </ul>
108  */
109  static const DXFCPP_EXPORT OrderAction EXECUTE;
110 
111  /**
112  * Non-Book Trade - this Trade not refers to any entry in Order Book.
113  * Order's @ref Order::getSize() "size" and @ref Order::getPrice() "price" will be equals to 0.
114  *
115  * <p>Full Order Book fields:
116  * <ul>
117  * <li>@ref Order::getOrderId() "orderId" - always empty</li>
118  * <li>@ref Order::getTradeId() "tradeId" - if available</li>
119  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" -
120  * contain size and price of this trade - always present</li>
121  * </ul>
122  */
123  static const DXFCPP_EXPORT OrderAction TRADE;
124 
125  /**
126  * Prior Trade/Order Execution bust.
127  * Order's @ref Order::getSize() "size" and @ref Order::getPrice() "price" will be equals to 0.
128  *
129  * <p>Full Order Book fields:
130  * <ul>
131  * <li>@ref Order::getOrderId() "orderId" - always empty</li>
132  * <li>@ref Order::getTradeId() "tradeId" - always present</li>
133  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" - always empty</li>
134  * </ul>
135  */
136  static const DXFCPP_EXPORT OrderAction BUST;
137 };
