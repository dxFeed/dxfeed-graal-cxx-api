22/**
23 * Summary information snapshot about the trading session including session highs, lows, etc.
24 * It represents the most recent information that is available about the trading session in
25 * the market at any given moment of time.
26 */
27class DXFCPP_EXPORT Summary final : public MarketEvent, public LastingEvent {
28 friend struct EventMapper;

30 /*
31 * Flags property has several significant bits that are packed into an integer in the following way:
32 * 31..4 3 2 1 0
33 * +--------+----+----+----+----+
34 * | | Close |PrevClose|
35 * +--------+----+----+----+----+
36 */

38 // PRICE_TYPE values are taken from PriceType enum.
39 static const std::uint32_t DAY_CLOSE_PRICE_TYPE_MASK = 3;
40 static const std::uint32_t DAY_CLOSE_PRICE_TYPE_SHIFT = 2;
41
42 // PRICE_TYPE values are taken from PriceType enum.
43 static const std::uint32_t PREV_DAY_CLOSE_PRICE_TYPE_MASK = 3;
44 static const std::uint32_t PREV_DAY_CLOSE_PRICE_TYPE_SHIFT = 0;

46 struct Data {

57 };
