18 /**
19  * Marks all event types that can be received via dxFeed API.
20  * Events are considered instantaneous, non-persistent, and unconflateable
21  * (each event is individually delivered) unless they implement one of interfaces
22  * defined in this package to further refine their meaning.
23  *
24  * <p>Event types are POCO (plain old cpp objects, not POD) that follow bean naming convention with
25  * getters and setters for their properties.
26  * All event types are serializable, because they are transferred over network from publishers to
27  * data feed consumers. However, they are using custom serialization format for this purpose.
28  *
29  * @see DXFeed
30  */
31 struct DXFCPP_EXPORT EventType : public SharedEntity {
32  /// The alias to a type of shared pointer to the EventType object.
33  using Ptr = std::shared_ptr<EventType>;
34 
35  /**
36  * Returns time when event was created or zero when time is not available.
37  *
38  * <p>This event time is available only when the corresponding DXEndpoint is created
39  * with @ref DXEndpoint::DXENDPOINT_EVENT_TIME_PROPERTY "DXENDPOINT_EVENT_TIME_PROPERTY" and
40  * the data source has embedded event times. This is typically true only for data events
41  * that are read from historical tape files and from OnDemandService.
42  * Events that are coming from a network connections do not have an embedded event time information and
43  * this method will return zero for them, meaning that event was received just now.
44  *
45  * Default implementation returns 0.
46  *
47  * @return The difference, measured in milliseconds, between the event creation time and
48  * midnight, January 1, 1970 UTC or zero when time is not available.
49  */
50  virtual std::int64_t getEventTime() const noexcept {
51  return 0;
52  }
53 
54  /**
55  * Changes event creation time.
56  *
57  * Default implementation does nothing.
58  *
59  * @param eventTime the difference, measured in milliseconds, between the event creation time and
60  * midnight, January 1, 1970 UTC.
61  */
62  virtual void setEventTime(std::int64_t /*eventTime*/) noexcept {
63  // The default implementation is empty
64  };
65 
66  /**
67  * Allocates memory for the dxFeed Graal SDK structure (recursively if necessary).
68  * Fills the dxFeed Graal SDK structure's fields by the data of the current entity (recursively if necessary).
69  * Returns the pointer to the filled structure.
70  *
71  * @return The pointer to the filled dxFeed Graal SDK structure
72  */
73  virtual void *toGraal() const = 0;
74 
75  ///
76  std::string toString() const noexcept override {
77  return "EventType{}";
78  }
79 
80  friend std::ostream &operator<<(std::ostream &os, const EventType &e) {
81  return os << e.toString();
82  }
83 
84  friend std::ostream &operator<<(std::ostream &os, const std::shared_ptr<EventType> &e) {
85  return os << e->toString();
86  }
87 
88  template <typename EntityType>
89  friend std::ostream &operator<<(std::ostream &os, const std::shared_ptr<EntityType> &e)
90 #if __cpp_concepts
91  requires(std::is_base_of_v<EventType, EntityType>)
92 #endif
93  {
94  return os << e->toString();
95  }
96 };
