12 /**
13  * <b>Schedule</b> class provides API to retrieve and explore trading schedules of different exchanges
14  * and different classes of financial instruments. Each instance of schedule covers separate trading schedule
15  * of some class of instruments, i.e. NYSE stock trading schedule or CME corn futures trading schedule.
16  * Each schedule splits entire time scale into separate @ref Day "days" that are aligned to the specific
17  * trading hours of covered trading schedule.
18  */
19 struct DXFCPP_EXPORT Schedule {
20  friend struct Day;
21 
22  /// The alias to a type of shared pointer to the Schedule object
23  using Ptr = std::shared_ptr<Schedule>;
24 
25  /// The alias to a type of unique pointer to the Schedule object
26  using Unique = std::unique_ptr<Schedule>;
27 
28  private:
29  JavaObjectHandle<Schedule> handle_;
30 
31  explicit Schedule(void *handle = nullptr) noexcept;
32 
33  /**
34  * Checks the handle, attempts to allocate memory for the pointer and return Schedule::Ptr
35  *
36  * @param handle The graal Schedule's handle
37  * @return The smart pointer for the Schedule object.
38  */
39  static Schedule::Ptr create(void *handle);
40 
41  public:
42  /**
43  * Returns default schedule instance for specified instrument profile.
44  *
45  * @param profile The instrument profile those schedule is requested
46  * @return The default schedule instance for specified instrument profile
47  */
48  static Schedule::Ptr getInstance(std::shared_ptr<InstrumentProfile> profile);
49 
50  /**
51  * Returns default schedule instance for specified schedule definition.
52  *
53  * @param scheduleDefinition The schedule definition of requested schedule
54  * @return The default schedule instance for specified schedule definition
55  */
56  static Schedule::Ptr getInstance(const std::string &scheduleDefinition);
57 
58  /**
59  * Returns schedule instance for specified instrument profile and trading venue.
60  *
61  * @param profile The instrument profile those schedule is requested
62  * @param venue The trading venue those schedule is requested
63  * @return The schedule instance for specified instrument profile and trading venue
64  */
65  static Schedule::Ptr getInstance(std::shared_ptr<InstrumentProfile> profile, const std::string &venue);
66 
67  /**
68  * Returns trading venues for specified instrument profile.
69  *
70  * @param profile The instrument profile those trading venues are requested
71  * @return trading venues for specified instrument profile
72  */
73  static std::vector<std::string> getTradingVenues(std::shared_ptr<InstrumentProfile> profile);
74 
75  /**
76  * Downloads defaults using specified download config and optionally start periodic download.
77  * The specified config can be one of the following:<ul>
78  * <li>"" - stop periodic download
79  * <li>URL - download once from specified URL and stop periodic download
80  * <li>URL,period - start periodic download from specified URL
81  * <li>"auto" - start periodic download from default location
82  * </ul>
83  *
84  * @param downloadConfig download config
85  */
86  static void downloadDefaults(const std::string &downloadConfig) noexcept;
87 
88  /**
89  * Sets shared defaults that are used by individual schedule instances.
90  *
91  * @param data The content of default data
92  * @return `true` if OK
93  */
94  static bool setDefaults(const std::vector<char> &data) noexcept;
95 
96  /**
97  * Returns session that contains specified time.
98  * This method will return `Session::Ptr{nullptr}` if specified time
99  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
100  *
101  * @param time the time to search for
102  * @return session that contains specified time or `Session::Ptr{nullptr}` if specified time
103  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
104  */
105  Session::Ptr getSessionByTime(std::int64_t time) const noexcept;
106 
107  /**
108  * Returns day that contains specified time.
109  * This method will return `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if specified time
110  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
111  *
112  * @param time The time to search for
113  * @return The day that contains specified time or `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if
114  * specified time falls outside of valid date range
115  */
116  Day::Ptr getDayByTime(std::int64_t time) const noexcept;
117 
118  /**
119  * Returns day for specified day identifier.
120  * This method will return `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if specified day identifier
121  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
122  *
123  * @param dayId The day identifier to search for
124  * @return The day for specified day identifier or `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if
125  * specified day identifier falls outside of valid date range
126  * @see Day::getDayId()
127  */
128  Day::Ptr getDayById(std::int32_t dayId) const noexcept;
129 
130  /**
131  * Returns day for specified year, month and day numbers.
132  * Year, month, and day numbers shall be packed in the following way:
133  * <pre>YearMonthDay = year * 10000 + month * 100 + day</pre>
134  * For example, September 28, 1977 has value 19770928.
135  * <p>
136  * If specified day does not exist then this method returns day with
137  * the lowest valid YearMonthDay that is greater than specified one.
138  * This method will return `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr}) if specified year, month and day
139  * numbers fall outside of valid date range from 0001-01-02 to 9999-12-30.
140  *
141  * @param yearMonthDay The year, month and day numbers to search for
142  * @return The day for specified year, month and day numbers or `Day::Ptr{nullptr}` (std::shared_ptr<Day>{nullptr})
143  * if specified year, month and day numbers fall outside of valid date range
144  * @see Day#getYearMonthDay()
145  */
146  Day::Ptr getDayByYearMonthDay(std::int32_t yearMonthDay) const noexcept;
147 
148  /**
149  * Returns session that is nearest to the specified time and that is accepted by specified filter.
150  * This method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time falls
151  * outside of valid date range from 0001-01-02 to 9999-12-30 or if no sessions acceptable by specified filter are
152  * found within one year this method
153  * <p> To find nearest trading session of any type use this code:
154  * <pre>session = schedule->getNearestSessionByTime(time, SessionFilter::TRADING);</pre>
155  * To find nearest regular trading session use this code:
156  * <pre>session = schedule->getNearestSessionByTime(time, SessionFilter::REGULAR);</pre>
157  *
158  * @param time The time to search for
159  * @param filter the filter to test sessions
160  * @return session that is nearest to the specified time and that is accepted by specified filter or
161  * `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time falls outside of valid date range
162  * or if no such day was found within one year
163  */
164  Session::Ptr getNearestSessionByTime(std::int64_t time, const SessionFilter &filter) const noexcept;
165 
166  /**
167  * Returns session that is nearest to the specified time and that is accepted by specified filter.
168  * This method will return `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time
169  * falls outside of valid date range from 0001-01-02 to 9999-12-30.
170  * If no sessions acceptable by specified filter are found within one year this method will return
171  * `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}).
172  * <p>
173  * To find nearest trading session of any type use this code:
174  * <pre>session = schedule->findNearestSessionByTime(time, SessionFilter::TRADING);</pre>
175  * To find nearest regular trading session use this code:
176  * <pre>session = schedule->findNearestSessionByTime(time, SessionFilter::REGULAR);</pre>
177  *
178  * @param time The time to search for
179  * @param filter The filter to test sessions
180  * @return The session that is nearest to the specified time and that is accepted by specified filter or
181  * `Session::Ptr{nullptr}` (std::shared_ptr<Session>{nullptr}) if specified time falls outside of valid date range
182  */
183  Session::Ptr findNearestSessionByTime(std::int64_t time, const SessionFilter &filter) const noexcept;
184 
185  /**
186  * Returns name of this schedule.
187  *
188  * @return The name of this schedule
189  */
190  std::string getName() const noexcept;
191 
192  /**
193  * Returns time zone display name in which this schedule is defined.
194  *
195  * @return time zone display name in which this schedule is defined
196  */
197  std::string getTimeZoneDisplayName() const noexcept;
198 
199  /**
200  * Returns time zone id in which this schedule is defined.
201  *
202  * @return time zone id in which this schedule is defined
203  */
204  std::string getTimeZoneId() const noexcept;
205 };
