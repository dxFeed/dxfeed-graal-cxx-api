24 /**
25  * Base class for common fields of Trade} and TradeETH events.
26  * Trade events represent the most recent information that is available about the last trade on the market at any
27  * given moment of time.
28  *
29  * Trade event represents last trade information for <b>regular trading hours</b> (RTH) with an official volume and
30  * turnover <b>for the whole trading day</b> identified by @ref TradeBase::getDayId() "dayId" field.
31  *
32  * TradeETH event is defined only for symbols (typically stocks and ETFs) with a designated
33  * <b>extended trading hours</b> (ETH, pre market and post market trading sessions). It represents last trade price
34  * during ETH and also accumulated volume and turnover during ETH for current trading day.
35  */
36 class DXFCPP_EXPORT TradeBase : public MarketEvent, public LastingEvent {
37  friend struct EventMapper;
38 
39  protected:
40  static constexpr std::uint64_t SECONDS_SHIFT = 32ULL;
41  static constexpr std::uint64_t MILLISECONDS_SHIFT = 22ULL;
42  static constexpr std::uint64_t MILLISECONDS_MASK = 0x3ffULL;
43 
44  /*
45  * Flags property has several significant bits that are packed into an integer in the following way:
46  * 31..4 3 2 1 0
47  * +--------+----+----+----+----+
48  * | | Direction | ETH|
49  * +--------+----+----+----+----+
50  */
51 
52  // DIRECTION values are taken from Direction enum.
53  static constexpr std::int32_t DIRECTION_MASK = 7;
54  static constexpr std::int32_t DIRECTION_SHIFT = 1;
55 
56  static constexpr std::int32_t ETH = 1;

116  /**
117  * Returns time of the last trade.
118  * Time is measured in milliseconds between the current time and midnight, January 1, 1970 UTC.
119  *
120  * @return time of the last trade.
121  */
122  std::int64_t getTime() const noexcept {
123  return sar(tradeBaseData_.timeSequence, SECONDS_SHIFT) * 1000 +
124  andOp(sar(tradeBaseData_.timeSequence, MILLISECONDS_SHIFT), MILLISECONDS_MASK);
125  }
126 
127  /**
128  * Changes time of the last trade.
129  * Time is measured in milliseconds between the current time and midnight, January 1, 1970 UTC.
130  *
131  * @param time time of the last trade.
132  */
133  void setTime(std::int64_t time) noexcept {
134  tradeBaseData_.timeSequence =
135  orOp(orOp(sal(static_cast<std::int64_t>(time_util::getSecondsFromTime(time)), SECONDS_SHIFT),
136  sal(static_cast<std::int64_t>(time_util::getMillisFromTime(time)), MILLISECONDS_SHIFT)),
137  getSequence());
138  }

179  /**
180  * Returns sequence number of the last trade to distinguish trades that have the same
181  * @ref TradeBase::getTime() "time". This sequence number does not have to be unique and
182  * does not need to be sequential. Sequence can range from 0 to TradeBase::MAX_SEQUENCE.
183  *
184  * @return sequence of the last trade.
185  */
186  std::int32_t getSequence() const noexcept {
187  return static_cast<std::int32_t>(andOp(tradeBaseData_.timeSequence, MAX_SEQUENCE));
188  }
189 
190  /**
191  * Changes @ref TradeBase::getSequence() "sequence number" of the last trade.
192  *
193  * @param sequence the sequence.
194  * @see TradeBase::getSequence()
195  */
196  void setSequence(std::int32_t sequence) {
197  assert(sequence >= 0 && static_cast<std::uint32_t>(sequence) <= MAX_SEQUENCE);
198 
199  if (sequence < 0 || static_cast<std::uint32_t>(sequence) > MAX_SEQUENCE) {
200  throw std::invalid_argument("Invalid sequence value = " + std::to_string(sequence));
201  }
202 
203  tradeBaseData_.timeSequence = orOp(andOp(tradeBaseData_.timeSequence, ~MAX_SEQUENCE), sequence);
204  }
