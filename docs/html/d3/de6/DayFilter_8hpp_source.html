59 /**
60  * A filter for days used by various search methods.
61  * This class provides predefined filters for certain Day attributes,
62  * although users can create their own filters to suit their needs.
63  * <p>
64  * Please note that days can be either trading or non-trading, and this distinction can be
65  * either based on rules (e.g. weekends) or dictated by special occasions (e.g. holidays).
66  * Different filters treat this distinction differently - some accept only trading days,
67  * some only non-trading, and some ignore type of day altogether.
68  */
69 struct DXFCPP_EXPORT DayFilter {
70  friend struct Day;
71 
72  /// Accepts any day - useful for pure calendar navigation.
73  static const DayFilter ANY;
74 
75  /// Accepts trading days only - those with <code>(Day::isTrading() == true)</code>.
76  static const DayFilter TRADING;
77 
78  /// Accepts non-trading days only - those with <code>(Day::isTrading() == false)</code>.
79  static const DayFilter NON_TRADING;
80 
81  /// Accepts holidays only - those with <code>(Day::isHoliday() == true)</code>.
82  static const DayFilter HOLIDAY;
83 
84  /// Accepts short days only - those with <code>(Day::isShortDay() == true)</code>.
85  static const DayFilter SHORT_DAY;
86 
87  /// Accepts Mondays only - those with <code>(Day::getDayOfWeek() == 1)</code>.
88  static const DayFilter MONDAY;
89 
90  /// Accepts Tuesdays only - those with <code>(Day::getDayOfWeek() == 2)</code>.
91  static const DayFilter TUESDAY;
92 
93  /// Accepts Wednesdays only - those with <code>(Day::getDayOfWeek() == 3)</code>.
94  static const DayFilter WEDNESDAY;
95 
96  /// Accepts Thursdays only - those with <code>(Day::getDayOfWeek() == 4)</code>.
97  static const DayFilter THURSDAY;
98 
99  /// Accepts Fridays only - those with <code>(Day::getDayOfWeek() == 5)</code>.
100  static const DayFilter FRIDAY;
101 
102  /// Accepts Saturdays only - those with <code>(Day::getDayOfWeek() == 6)</code>.
103  static const DayFilter SATURDAY;
104 
105  /// Accepts Sundays only - those with <code>(Day::getDayOfWeek() == 7)</code>.
106  static const DayFilter SUNDAY;
107 
108  /// Accepts week-days only - those with <code>(Day::getDayOfWeek() <= 5)</code>.
109  static const DayFilter WEEK_DAY;
110 
111  /// Accepts weekends only - those with <code>(Day::getDayOfWeek() >= 6)</code>.
112  static const DayFilter WEEK_END;
113 
114  DayFilter(const DayFilter &) = delete;
115  DayFilter(DayFilter &&) noexcept = delete;
116  DayFilter &operator=(const DayFilter &) = delete;
117  DayFilter &operator=(const DayFilter &&) noexcept = delete;
118 
119  private:
120  DayFilterEnum code_;
121 
122  std::string name_;
123 
124  /// Bitwise mask of required days (2 for Monday, 4 for Tuesday, 128 for Sunday), <code>0</code> if not relevant.
125  std::uint32_t dayOfWeekMask_;
126  /// Required holiday flag, Tristate::NONE if not relevant.
127  Tristate holiday_;
128  /// Required short day flag, Tristate::NONE if not relevant.
129  Tristate shortDay_;
130  /// Required trading flag, Tristate::NONE if not relevant.
131  Tristate trading_;
132 
133  JavaObjectHandle<DayFilter> handle_;
134 
135  /**
136  * Creates filter with specified conditions.
137  * <p>
138  * The <code>dayOfWeekMask</code> is a bitwise mask with individual bits for each day of week.
139  * For the day of week number N the N'th bit is used - the Day will be accepted if corresponding bit is set.
140  * If no bits is set (if mask is zero) then day of week attribute is ignored (any value is accepted).
141  * <p>
142  * The boolean parameters specify what value corresponding attributes should have.
143  * If some parameter is Tristate::NONE then corresponding attribute is ignored (any value is accepted).
144  *
145  * @param code The enum code
146  * @param name The enum name
147  * @param dayOfWeekMask The bitwise mask of required days (2 for Monday, 4 for Tuesday, 128 for Sunday),
148  * <code>0</code> if not relevant
149  * @param holiday The required holiday flag, Tristate::NONE if not relevant
150  * @param shortDay The required short day flag, Tristate::NONE if not relevant
151  * @param trading The required trading flag, Tristate::NONE if not relevant
152  */
153  DayFilter(DayFilterEnum code, std::string name, std::uint32_t dayOfWeekMask, Tristate holiday, Tristate shortDay,
154  Tristate trading) noexcept;
155 };
