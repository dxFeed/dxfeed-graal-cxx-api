30 /**
31  * Provides API for publishing of events to local or remote DXFeed.
32  *
33  * <h3>Sample usage</h3>
34  *
35  * This section gives sample usage scenarios.
36  *
37  * <h4>Default singleton instance</h4>
38  *
39  * There is a singleton instance of the publisher that is returned by DXPublisher::getInstance() method.
40  * It is created on the first use with default configuration properties that are explained in detail in
41  * documentation for DXEndpoint class in the "Default properties" section. In particular,
42  * you can provide a default address to connect using
43  * '@ref DXEndpoint::DXPUBLISHER_ADDRESS_PROPERTY "dxpublisher.address"'
44  * system property or by putting it into
45  * '@ref DXEndpoint::DXPUBLISHER_PROPERTIES_PROPERTY "dxpublisher.properties"'
46  * file in the current directory.
47  *
48  * <h4>Publish a single event</h4>
49  *
50  * The following code publishes a single quote for a "A:TEST" symbol:
51  * <pre><tt>
52  * auto @ref Quote "quote" = std::make_shared<@ref Quote "Quote">("A:TEST");
53  * quote->@ref Quote::setBidPrice "setBidPrice"(100);
54  * quote->@ref Quote::setAskPrice "setAskPrice"(101);
55  * @ref DXPublisher "DXPublisher"::@ref DXPublisher::getInstance() "getInstance"()->@ref DXPublisher::publishEvents()
56  * "publishEvents"(quote);</tt></pre>
57  *
58  * <h3>Threads and locks</h3>
59  *
60  * This class is thread-safe and can be used concurrently from multiple threads without external synchronization.
61  */

101  /**
102  * Publishes events to the corresponding feed. If the @ref DXEndpoint "endpoint" of this publisher has
103  * @ref DXEndpoint::getRole() "role" of DXEndpoint::Role::PUBLISHER and it is connected, the
104  * published events will be delivered to the remote endpoints. Local @ref DXEndpoint::getFeed() "feed" will
105  * always receive published events.
106  *
107  * <p> This method serializes all events into internal representation, so that the instance of the collection as
108  * well as the instances of events can be reused after invocation of this method returns.
109  *
110  * <p> DXFeed instances that are connected to this publisher either locally or via network receive published
111  * events if and only if they are subscribed to the corresponding symbols, or they are subscribed via
112  * WildcardSymbol::ALL, or, in case of TimeSeriesEvent type, they are subscribed via DXFeedTimeSeriesSubscription
113  * for the corresponding symbol and time frame.
114  *
115  * <p> Published events are not stored and get immediately lost if there is no subscription.
116  * Last published events of LastingEvent types are cached as long as subscription to them is maintained via a
117  * specific event symbol (WildcardSymbol::ALL does not count) and the cache is discarded as soon as subscription
118  * disappears.
119  *
120  * Example:
121  *
122  * ```cpp
123  * using namespace dxfcpp;
124  * using namespace dxfcpp::literals;
125  * using namespace std::literals;
126  *
127  * System::setProperty("dxendpoint.eventTime", "true");
128  * System::setProperty("dxscheme.nanoTime", "true");
129  *
130  * auto endpoint = DXEndpoint::create(DXEndpoint::Role::FEED);
131  * auto feed = endpoint->getFeed();
132  * auto publisher = endpoint->getPublisher();
133  * auto sub = feed->createSubscription({Quote::TYPE});
134  * auto quote = std::make_shared<Quote>("AAPL");
135  *
136  * quote->setEventTime(1692975409000);
137  * quote->setTimeNanoPart(123);
138  * quote->setSequence(123);
139  * quote->setBidTime(1692975409000);
140  * quote->setBidExchangeCode('B');
141  * quote->setBidPrice(176.08);
142  * quote->setBidSize(1.0);
143  * quote->setAskTime(1692975409000);
144  * quote->setAskExchangeCode('A');
145  * quote->setAskPrice(176.16);
146  * quote->setAskSize(2.0);
147  *
148  * sub->addEventListener([](const auto &events) {
149  * for (const auto &e : events) {
150  * if (const auto &q = e->template sharedAs<Quote>(); q) {
151  * std::cout << q << std::endl;
152  * }
153  * }
154  * });
155  *
156  * sub->addSymbols("AAPL");
157  * publisher->publishEvents(quote);
158  * publisher->publishEvents({quote, quote});
159  *
160  * std::vector<EventType::Ptr> events{};
161  *
162  * events.push_back(quote);
163  * events.push_back(quote);
164  * publisher->publishEvents(events);
165  * std::this_thread::sleep_for(1000ms);
166  * endpoint->close();
167  * ```
168  *
169  * @tparam EventsCollection The type of events collection (for example: std::vector<::EventType::Ptr>)
170  * @param events The collection of events to publish.
171  */
172  template <typename EventsCollection>
173  void publishEvents(EventsCollection &&events) noexcept
174 #if __cpp_concepts
175  requires requires {
176  { std::begin(std::forward<EventsCollection>(events)) };
177 
178  { std::end(std::forward<EventsCollection>(events)) };
179  }
180 #endif
181  {
182  this->template publishEvents<decltype(std::begin(std::forward<EventsCollection>(events)))>(
183  std::begin(std::forward<EventsCollection>(events)), std::end(std::forward<EventsCollection>(events)));
184  }

204  /**
205  * Publishes events to the corresponding feed.
206  *
207  * @tparam EventIt The collection's iterator type
208  * @param begin The beginning of the collection of events.
209  * @param end The end of events collection.
210  */
211  template <typename EventIt> void publishEvents(EventIt begin, EventIt end) {
212  if constexpr (Debugger::isDebug) {
213  Debugger::debug(toString() + "::publishEvents(events = " + elementsToString(begin, end) + ")");
214  }
215 
216  auto *list = EventMapper::toGraalList(begin, end);
217 
218  publishEventsImpl(list);
219  EventMapper::freeGraalList(list);
220  }

222  /**
223  * Returns observable set of subscribed symbols for the specified event type.
224  * Note, that subscription is represented by SymbolWrapper symbols. Check the type of each symbol
225  * in ObservableSubscription using SymbolWrapper::isStringSymbol(), SymbolWrapper::isWildcardSymbol(),
226  * SymbolWrapper::isIndexedEventSubscriptionSymbol(), SymbolWrapper::isTimeSeriesSubscriptionSymbol(),
227  * SymbolWrapper::isCandleSymbol().
228  *
229  * <p> The set of subscribed symbols contains WildcardSymbol::ALL if and
230  * only if there is a subscription to this wildcard symbol.
231  *
232  * <p> If DXFeedTimeSeriesSubscription is used to subscribe to time-service of the events of this type, then
233  * instances of TimeSeriesSubscriptionSymbol class represent the corresponding subscription item.
234  *
235  * <p> The resulting observable subscription can generate repeated ObservableSubscriptionChangeListener::onSymbolsAdded_ notifications to
236  * its listeners for the same symbols without the corresponding ObservableSubscriptionChangeListener::onSymbolsRemoved_
237  * notifications in between them. It happens when subscription disappears, cached data is lost, and subscription
238  * reappears again. On each ObservableSubscriptionChangeListener::onSymbolsAdded_
239  * notification data provider shall @ref DXPublisher::publishEvents() "publish" the most recent events for
240  * the corresponding symbols.
241  *
242  * @param eventType The event type.
243  * @return Observable subscription for the specified event type.
244  */
245  std::shared_ptr<ObservableSubscription> getSubscription(const EventTypeEnum &eventType);
246 
247  std::string toString() const noexcept override;
