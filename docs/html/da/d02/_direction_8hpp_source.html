40 /**

44 * It is also reported for cases when price sequence was broken and direction computation was restarted anew.
45 */
46 static const DXFCPP_EXPORT Direction ZERO;
47
48 /**
49 * Current price is the same as previous price and is higher than the last known price of different value.
50 */

53 /**
54 * Current price is higher than previous price.
55 */
56 static const DXFCPP_EXPORT Direction UP;
