15/**
16 * Source identifier for IndexedEvent.
17 *
18 * See IndexedEvent::getSource().
19 */
20class DXFCPP_EXPORT IndexedEventSource {
21 std::int32_t id_{};
22 std::string name_{};
23
24 public:
25 /**
26 * The default source with zero @ref ::id() "identifier" for all events that do not support multiple sources.
27 */
28 static const IndexedEventSource DEFAULT;
29
30
31 /**
32 * Allocates memory for the dxFeed Graal SDK structure (recursively if necessary).
33 * Fills the dxFeed Graal SDK structure's fields by the data of the current entity (recursively if necessary).
34 * Returns the pointer to the filled structure.
35 *
36 * @return The pointer to the filled dxFeed Graal SDK structure
37 */
38 virtual void *toGraal() const;
39
40 /**
41 * Releases the memory occupied by the dxFeed Graal SDK structure (recursively if necessary).
42 *
43 * @param graalNative The pointer to the dxFeed Graal SDK structure.
44 */
45 static void freeGraal(void *graalNative);
46
47 /**
48 * Creates an object of the current type and fills it with data from the the dxFeed Graal SDK structure.
49 *
50 * @param graalNative The pointer to the dxFeed Graal SDK structure.
51 * @return The object of current type.
52 * @throws std::invalid_argument
53 */
54 static IndexedEventSource fromGraal(void *graalNative);
55
56 IndexedEventSource() noexcept = default;
57 virtual ~IndexedEventSource() noexcept = default;
58
59 /**
60 * Creates the new IndexedEvent's source by id and name.
61 *
62 * @param id The source id
63 * @param name The source name
64 */
65 IndexedEventSource(std::int32_t id, std::string name) noexcept : id_{id}, name_{std::move(name)} {
66 }
67
68 /**
69 * Returns the source identifier. Source identifier is non-negative.
70 *
71 * @return The source identifier.
72 */
73 std::int32_t id() const noexcept {
74 return id_;
75 }
76
77 /**
78 * Returns the string representation of the object.
79 *
80 * @return The string representation of the object.
81 */
82 const std::string &name() const noexcept {
83 return name_;
84 }
85
86 /**
87 * Returns the string representation of the object.
88 *
89 * @return The string representation of the object.
90 */
91 std::string toString() const noexcept {
92 return name_;
93 }
94
95 bool operator==(const IndexedEventSource &indexedEventSource) const {
96 return id_ == indexedEventSource.id_;
97 }
98
99 auto operator<(const IndexedEventSource &indexedEventSource) const {
100 return id_ < indexedEventSource.id_;
101 }
102};
