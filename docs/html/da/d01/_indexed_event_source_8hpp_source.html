13/**
14 * Source identifier for IndexedEvent.
15 *
16 * See IndexedEvent::getSource().
17 */
18class DXFCPP_EXPORT IndexedEventSource {
19 std::int32_t id_{};
20 std::string name_{};
21
22 public:
23 /**
24 * The default source with zero @ref ::id() "identifier" for all events that do not support multiple sources.
25 */
26 static const IndexedEventSource DEFAULT;
27
28 IndexedEventSource() noexcept = default;
29 virtual ~IndexedEventSource() noexcept = default;
30
31 /**
32 * Creates the new IndexedEvent's source by id and name.
33 *
34 * @param id The source id
35 * @param name The source name
36 */
37 IndexedEventSource(std::int32_t id, std::string name) noexcept : id_{id}, name_{std::move(name)} {
38 }
39
40 /**
41 * Returns the source identifier. Source identifier is non-negative.
42 *
43 * @return The source identifier.
44 */
45 std::int32_t id() const noexcept {
46 return id_;
47 }
48
49 /**
50 * Returns the string representation of the object.
51 *
52 * @return The string representation of the object.
53 */
54 const std::string &name() const noexcept {
55 return name_;
56 }
57
58 /**
59 * Returns the string representation of the object.
60 *
61 * @return The string representation of the object.
62 */
63 std::string toString() const noexcept {
64 return name_;
65 }
66
67 bool operator==(const IndexedEventSource &indexedEventSource) const {
68 return id_ == indexedEventSource.id_;
69 }
70
71 auto operator<(const IndexedEventSource &indexedEventSource) const {
72 return id_ < indexedEventSource.id_;
73 }
74
75 /**
76 * Allocates memory for the dxFeed Graal SDK structure (recursively if necessary).
77 * Fills the dxFeed Graal SDK structure's fields by the data of the current entity (recursively if necessary).
78 * Returns the pointer to the filled structure.
79 *
80 * @return The pointer to the filled dxFeed Graal SDK structure
81 */
82 virtual void *toGraal() const noexcept;
83
84 /**
85 * Releases the memory occupied by the dxFeed Graal SDK structure (recursively if necessary).
86 *
87 * @param graalNative The pointer to the dxFeed Graal SDK structure.
88 */
89 static void freeGraal(void *graalNative) noexcept;
90
91 static IndexedEventSource fromGraal(void *graalNative) noexcept;
92};
93
94} // namespace dxfcpp
