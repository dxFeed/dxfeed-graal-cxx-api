52 static std::int32_t composeId(const std::string &name) noexcept {
53 std::int32_t sourceId = 0;
54
55 auto n = name.length();
56
57 if (n == 0 || n > 4) {
58 // TODO: error handling: IllegalArgumentException("Source name must contain from 1 to 4 characters");
59
60 return -1;
61 }
62
63 for (auto c : name) {
64 if (!checkChar(c)) {
65 // TODO: error handling
66
67 return -1;
68 }
69
70 sourceId = orOp(sal(sourceId, 8), c);
71 }
72
73 return sourceId;
74 }

76 static bool checkChar(char c) noexcept {
77 if (c >= 'A' && c <= 'Z' || c >= 'a' && c <= 'z' || c >= '0' && c <= '9') {
78 return true;
79 }
80
81 // TODO: error handling: IllegalArgumentException("Source name must contain only alphanumeric characters");
82
83 return false;
84 }

86 static std::string decodeName(int id) noexcept {
87 if (id == 0) {
88 // TODO: error handling: throw IllegalArgumentException("Source name must contain from 1 to 4 characters");
89
90 return "";
91 }
92
93 std::string name(4, '\0');
94
95 std::size_t n = 0;
96
97 for (int i = 24; i >= 0; i -= 8) {
98 if (sar(id, i) == 0) { // skip highest contiguous zeros
99 continue;
100 }
101
102 char c = static_cast<char>(andOp(sar(id, i), 0xff));
103
104 if (!checkChar(c)) {
105 return "";
106 }
107
108 name[n++] = c;
109 }
110
111 return name.substr(0, n);
112 }

465 /**
466 * Returns order source for the specified source identifier.
467 *
468 * @param sourceId the source identifier.
469 * @return order source.
470 */
471 static const OrderSource &valueOf(std::int32_t sourceId) noexcept {
472 if (auto found = PREDEFINED_SOURCES.find(sourceId); found != PREDEFINED_SOURCES.end()) {
473 return found->second;
474 }
475
476 std::lock_guard lock(MTX_);
477
478 if (auto found = USER_SOURCES_.find(sourceId); found != USER_SOURCES_.end()) {
479 return found->second;
480 }
481
482 return USER_SOURCES_.try_emplace(sourceId, OrderSource(sourceId, decodeName(sourceId))).first->second;
483 }
484
485 /**
486 * Returns order source for the specified source name.
487 * The name must be either predefined, or contain at most 4 alphanumeric characters.
488 *
489 * @param name the name of the source.
490 * @return order source.
491 */
492 static const OrderSource &valueOf(const std::string &name) noexcept {
493 if (auto found = PREDEFINED_SOURCES.find(name); found != PREDEFINED_SOURCES.end()) {
494 return found->second;
495 }
496
497 std::lock_guard lock(MTX_);
498
499 auto sourceId = composeId(name);
500
501 if (auto found = USER_SOURCES_.find(sourceId); found != USER_SOURCES_.end()) {
502 return found->second;
503 }
504
505 return USER_SOURCES_.try_emplace(sourceId, OrderSource(sourceId, name)).first->second;
506 }
