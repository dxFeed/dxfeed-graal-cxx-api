24 /**
25  * Provides on-demand historical tick data replay controls.
26  * This class is used to seamlessly transition from ordinary real-time or delayed data feed to the replay of the
27  * tick-by-tick history of market data behaviour without any changes to the code that consumes and process these market
28  * events.
29  *
30  * <p>As single OnDemandService instance is conceptually associated with each DXEndpoint endpoint instance with the role
31  * of @ref DXEndpoint::Role::FEED "FEED" or @ref DXEndpoint::Role::ON_DEMAND_FEED "ON_DEMAND_FEED".
32  * You can retrieve OnDemandService instance that is associated with a given
33  * DXEndpoint using @ref OnDemandService::getInstance(std::shared_ptr<DXEndpoint>) "getInstance(endpoint)" method.
34  *
35  * <p>For example, OnDemandService instance for a default DXFeed that is featured in the documentation of DXFeed class
36  * and is retrieved via
37  * @ref DXFeed::getInstance() "DXFeed::getInstance()" method can be acquired with the following code:
38  * <pre><tt>
39  * @ref DXEndpoint "auto" endpoint = DXEndpoint::getInstance();
40  * @ref OnDemandService "auto" onDemand = OnDemandService::@ref OnDemandService::getInstance(std::shared_ptr<DXEndpoint>) "getInstance"(endpoint);
41  * </tt></pre>
42  *
43  * This instance can be used for on-demand historical tick data replay only when endpoint is connected to
44  * the on-demand data provider and the appropriate credentials are provided.
45  * @ref OnDemandService::isReplaySupported() "isReplaySupported" returns `true` when it is so.
46  *
47  * <p>On-demand replay is started with @ref OnDemandService::replay() "replay" method that takes time as a parameter.
48  * DXFeed is then disconnected from other data providers, pre-buffers historical tick data for replay,
49  * and starts replay of the data as if it was coming from the data feeds now. All the usual APIs that are
50  * part of DXFeed like @ref DXFeedSubscription "subscription" and various dxFeed models
51  * can be used normally as with the ordinary real-time or delayed data feed.
52  *
53  * <p>Replay speed can be changed on the fly with @ref OnDemandService::setSpeed() "setSpeed" method or it can be set
54  * initially when starting replay with a two-argument version of @ref OnDemandService::replay() "replay" method.
55  * @ref OnDemandService::pause() "pause" method is the same as @ref OnDemandService::setSpeed() "setSpeed(0)".
56  *
57  * <p>@ref OnDemandService::stopAndResume() "stopAndResume" method stops data replay and resumes ordinary real-time or delayed data feed
58  * that was used before @ref OnDemandService::replay() "replay" method was invoked.
59  * Endpoints with a role of
60  * @ref DXEndpoint::Role::ON_DEMAND_FEED "ON_DEMAND_FEED" do not have an ordinary feed (they are essentially on-demand only)
61  * and thus @ref OnDemandService::stopAndResume() "stopAndResume" method works like @ref OnDemandService::stopAndClear() "stopAndClear" for them.
62  *
63  * <h3>State and change notifications</h3>
64  *
65  * On-demand historical tick data replay state can be queried with @ref OnDemandService::isReplaySupported() "isReplaySupported",
66  * @ref OnDemandService::isReplay() "isReplay", @ref OnDemandService::getSpeed() "getSpeed", and @ref OnDemandService::getTime() "getTime" methods.
67  *
68  * <h3>Threads and locks</h3>
69  *
70  * This class is thread-safe and can be used concurrently from multiple threads without external synchronization.
71  */
72 struct DXFCPP_EXPORT OnDemandService : SharedEntity {
73  /// The alias to a type of shared pointer to the OnDemandService object
74  using Ptr = std::shared_ptr<OnDemandService>;
75 
76  /// The alias to a type of unique pointer to the OnDemandService object
77  using Unique = std::unique_ptr<OnDemandService>;
78 
79  private:
80  JavaObjectHandle<OnDemandService> handle_;
81  std::shared_ptr<DXEndpoint> endpoint_{};
82 
83  protected:
84  OnDemandService() noexcept;
85 
86  public:
87  ~OnDemandService() noexcept override;
88 
89  /**
90  * Returns on-demand service for the default DXEndpoint instance with
91  * @ref DXEndpoint::Role::ON_DEMAND_FEED "ON_DEMAND_FEED" role that is
92  * not connected to any other real-time or delayed data feed.
93  * This method is a shortcut for:
94  * <pre><tt>
95  * OnDemandService::@ref OnDemandService::getInstance(std::shared_ptr<DXEndpoint>) "getInstance"(@ref DXEndpoint "DXEndpoint"::@ref DXEndpoint::getInstance(DXEndpoint::Role) "getInstance"(DXEndpoint::Role::ON_DEMAND_FEED))</tt></pre>
96  *
97  * <p>If you need an instance of OnDemandService to seamlessly switch from other real-time or delayed
98  * data feed to on-demand historical tick data replay, then use
99  * {@ref OnDemandService::getInstance(std::shared_ptr<DXEndpoint>) "getInstance(endpoint)" method for a specific `endpoint` that
100  * you are using.
101  *
102  * @return on-demand endpoint for the default DXEndpoint instance.
103  */
104  static std::shared_ptr<OnDemandService> getInstance() noexcept;
105 
106  /**
107  * Returns on-demand service for the specified DXEndpoint.
108  * Each DXEndpoint is conceptually associated with a single instance of on-demand service to
109  * control its historic data replay and this method returns this instance.
110  * The endpoint must have a role of
111  * @ref DXEndpoint::Role::FEED "FEED" or @ref DXEndpoint::Role::ON_DEMAND_FEED "ON_DEMAND_FEED".
112  *
113  * @param endpoint the endpoint.
114  * @return the on-demand service.
115  */
116  static std::shared_ptr<OnDemandService> getInstance(std::shared_ptr<DXEndpoint> endpoint);
117 
118  /**
119  * Returns DXEndpoint that is associated with this on-demand service.
120  *
121  * @return DXEndpoint that is associated with this on-demand service.
122  */
123  std::shared_ptr<DXEndpoint> getEndpoint() const noexcept;
124 
125  /**
126  * Returns `true` when on-demand historical data replay mode is supported.
127  * DXEndpoint should be @ref DXEndpoint::connect() "connected"
128  * to an address with "(ondemand:<address>)" component that specifies an on-demand historical
129  * data provider address.
130  * When this method returns `false`, @ref OnDemandService::replay() "replay" method does nothing.
131  *
132  * @return `true` when on-demand historical data replay mode is supported.
133  */
134  bool isReplaySupported() const noexcept;
135 
136  /**
137  * Returns `true` when this on-demand historical data replay service is in replay mode.
138  * Replay mode is in effect after invocation of @ref OnDemandService::replay() "replay" method and
139  * before invocation @ref OnDemandService::stopAndResume() "stopAndResume" or @ref OnDemandService::stopAndClear() "stopAndClear" methods.
140  *
141  * @return `true` when this on-demand historical data replay service is in replay mode.
142  */
143  bool isReplay() const noexcept;
144 
145  /**
146  * Returns `true` when this on-demand historical data replay service is in clear mode.
147  * Clear mode is in effect after invocation of @ref OnDemandService::stopAndClear() "stopAndClear" method and
148  * before invocation @ref OnDemandService::stopAndResume() "stopAndResume" or @ref OnDemandService::replay() "replay" methods.
149  *
150  * @return `true` when this on-demand historical data replay service is in clear mode.
151  */
152  bool isClear() const noexcept;
153 
154  /**
155  * Returns current or last on-demand historical data replay time.
156  * In @ref OnDemandService::isReplay() "replay mode" this is the time that is being currently replayed,
157  * otherwise this is the last time that was replayed.
158  * If replay was never started, then result is 0.
159  *
160  * @return current or last on-demand historical data replay time in millis.
161  */
162  std::int64_t getTime() const noexcept;
163 
164  /**
165  * Returns on-demand historical data replay speed.
166  * Speed is measured with respect to the real-time playback speed.
167  * The result of this method is zero when this service is not in @ref OnDemandService::isReplay() "replay mode".
168  * The speed is set when starting replay by @ref OnDemandService::replay() "replay(time, speed)" method
169  * and with @ref OnDemandService::setSpeed() "setSpeed(speed)" method during replay.
170  *
171  * @return on-demand historical data replay speed.
172  */
173  double getSpeed() const noexcept;
174 
175  /**
176  * Turns on-demand historical data replay mode from a specified time with real-time speed.
177  * This is a shortcut for:
178  * <pre><tt>
179  * @ref OnDemandService::replay(std::int64_t,double) "replay"(time, 1);</tt></pre>
180  * This method can be used only when OnDemandService::isReplaySupported() method returns `true`,
181  * that is when DXEndpoint is @ref DXEndpoint::connect() "connected"
182  * to an address with "(ondemand:<address>)" component that specifies an on-demand historical
183  * data provider address.
184  *
185  * @param time time (timestamp in millis) to start replay from.
186  */
187  void replay(std::int64_t time) const noexcept;
188 
189  /**
190  * Turns on-demand historical data replay mode from a specified time and with a specified speed.
191  * This method can be used only when OnDemandService::isReplaySupported() method returns `true`,
192  * that is when DXEndpoint is @ref DXEndpoint::connect() "connected"
193  * to an address with "(ondemand:<address>)" component that specifies an on-demand historical
194  * data provider address.
195  *
196  * <p>After invocation of this method:
197  * <ul>
198  * <li>OnDemandService::isReplay() returns `true`.
199  * <li>OnDemandService::isClear() returns `false`.
200  * <li>OnDemandService::getTime() returns time that is currently being replayed.
201  * <li>OnDemandService::getSpeed() returns `speed`.
202  * </ul>
203  *
204  * @param time time (timestamp in millis) to start replay from.
205  * @param speed speed to start replay with. Use 1 for real-time speed, >1 for faster than real-time speed,
206  * <1 for slower than real-time speed, and 0 for pause.
207  */
208  void replay(std::int64_t time, double speed) const noexcept;
209 
210  /**
211  * Pauses on-demand historical data replay and keeps data snapshot.
212  * This method can only be called in @ref OnDemandService::isReplay() "replay mode".
213  * This is a shortcut for:
214  * <pre><tt>
215  * @ref OnDemandService::setSpeed() "setSpeed"(0);</tt></pre>
216  * This method atomically captures current replay time and pauses at it.
217  * After invocation of this method:
218  * <ul>
219  * <li>OnDemandService::isReplay() returns `true`.
220  * <li>OnDemandService::isClear() returns `false`.
221  * <li>OnDemandService::getTime() returns time where replay was paused at.
222  * <li>OnDemandService::getSpeed() returns `0`.
223  * </ul>
224  */
225  void pause() const noexcept;
226 
227  /**
228  * Stops on-demand historical data replay and resumes ordinary data feed.
229  * This method has no effect when invoked not in @ref OnDemandService::isReplay() "replay mode".
230  * After invocation of this method:
231  * <ul>
232  * <li>OnDemandService::isReplay() returns `false`.
233  * <li>OnDemandService::isClear() returns `false`.
234  * <li>OnDemandService::getTime() returns last replayed time.
235  * <li>OnDemandService::getSpeed() returns `0`.
236  * </ul>
237  *
238  * <p>To stop on-demand historical data replay without resuming ordinary data feed use
239  * either OnDemandService::pause() to keep data snapshot or OnDemandService::stopAndClear() to clear data.
240  *
241  * <p>Note, that endpoints with a role of
242  * @ref DXEndpoint::Role::ON_DEMAND_FEED "ON_DEMAND_FEED" do not have an ordinary feed (they are essentially on-demand only)
243  * and thus `stopAndResume` method works like @ref OnDemandService::stopAndClear() "stopAndClear" for them.
244  */
245  void stopAndResume() const noexcept;
246 
247  /**
248  * Stops incoming data and clears it without resuming data updates.
249  * This method works both in on-demand historical data replay mode and
250  * for ordinary data feed. All incoming data updates are suspended and current data is cleared.
251  * After invocation of this method:
252  * <ul>
253  * <li>OnDemandService::isReplay() returns `false`.
254  * <li>OnDemandService::isClear() returns `true`.
255  * <li>OnDemandService::getTime() returns last replayed time.
256  * <li>OnDemandService::getSpeed() returns `0`.
257  * </ul>
258  *
259  * <p>Ordinary data feed can be resumed with OnDemandService::stopAndResume() method and on-demand historical data
260  * replay can be continued with @ref OnDemandService::replay() "replay(...)" method.
261  */
262  void stopAndClear() const noexcept;
263 
264  /**
265  * Changes on-demand historical data replay speed while continuing replay at current @ref OnDemandService::getTime() "time".
266  * Speed is measured with respect to the real-time playback speed.
267  * This method can only be called with non-zero speed in @ref OnDemandService::isReplay() "replay mode".
268  *
269  * @param speed on-demand historical data replay speed.
270  */
271  void setSpeed(double speed) const noexcept;
272 };
