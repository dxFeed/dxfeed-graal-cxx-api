32 /**
33  * A filter for sessions used by various search methods.
34  * This class provides predefined filters for certain Session attributes,
35  * although users can create their own filters to suit their needs.
36  * <p>
37  * Please note that sessions can be either trading or non-trading, and this distinction can be
38  * either based on rules (e.g. weekends) or dictated by special occasions (e.g. holidays).
39  * Different filters treat this distinction differently - some accept only trading sessions,
40  * some only non-trading, and some ignore type of session altogether.
41  */
42 struct DXFCPP_EXPORT SessionFilter {
43  friend struct Session;
44  friend struct Schedule;
45  friend struct Day;
46 
47  /** Accepts any session - useful for pure schedule navigation. */
48  static const SessionFilter ANY;
49  /** Accepts trading sessions only - those with <code>(Session::isTrading() == true)</code>. */
50  static const SessionFilter TRADING;
51  /** Accepts non-trading sessions only - those with <code>(Session::isTrading() == false)</code>. */
52  static const SessionFilter NON_TRADING;
53 
54  /** Accepts any session with type SessionType::NO_TRADING. */
55  static const SessionFilter NO_TRADING;
56  /** Accepts any session with type SessionType::PRE_MARKET. */
57  static const SessionFilter PRE_MARKET;
58  /** Accepts any session with type SessionType::REGULAR. */
59  static const SessionFilter REGULAR;
60  /** Accepts any session with type SessionType::AFTER_MARKET. */
61  static const SessionFilter AFTER_MARKET;
62 
63  protected:
64  SessionFilterEnum code_{};
65 
66  std::string name_;
67 
68  /// Required type, std::nullopt if not relevant.
69  std::optional<SessionType> type_;
70  /// Required trading flag, std::nullopt if not relevant.
71  std::optional<bool> trading_;
72 
73  JavaObjectHandle<SessionFilter> handle_;
74 
75  public:
76  SessionFilter() noexcept = default;
77  virtual ~SessionFilter() noexcept = default;
78 
79  SessionFilter(const SessionFilter &) = delete;
80  SessionFilter(SessionFilter &&) noexcept = delete;
81  SessionFilter &operator=(const SessionFilter &) = delete;
82  SessionFilter &operator=(const SessionFilter &&) noexcept = delete;
83 
84  /**
85  * Creates filter with specified type and trading flag conditions.
86  * <p>
87  * Both parameters specify what value corresponding attributes should have.
88  * If some parameter is std::nullopt then corresponding attribute is ignored (any value is accepted).
89  *
90  * @param code The enum code
91  * @param name The enum name
92  * @param type required type, std::nullopt if not relevant
93  * @param trading required trading flag, std::nullopt if not relevant
94  */
95  SessionFilter(SessionFilterEnum code, std::string name, std::optional<SessionType> type, std::optional<bool> trading) noexcept;
96 
97  /**
98  * Tests whether or not the specified session is an acceptable result.
99  *
100  * @param session the session to be tested
101  * @return `true` if specified session is accepted
102  */
103  template <typename Session> bool accept(Session session) const noexcept {
104  return (!type_ || type_.value() == session.getType()) && (!trading_ || trading_.value() == session.isTrading());
105  }
106 
107  bool operator==(const SessionFilter &sessionFilter) const noexcept {
108  return type_ == sessionFilter.type_ && trading_ == sessionFilter.trading_;
109  }
110 
111  std::string toString() const noexcept {
112  return std::string("SessionFilter(") + ((!type_) ? "null" : type_.value().toString()) + ", " +
113  ((!trading_) ? "null" : dxfcpp::toString(trading_.value())) + ")";
114  }
115  const std::optional<SessionType> &getType() const & noexcept {
116  return type_;
117  }
118 
119  const std::optional<bool> &getTrading() const & noexcept {
120  return trading_;
121  }
122 };

139  }
