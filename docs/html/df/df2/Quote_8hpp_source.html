24 /**
25  * Quote event is a snapshot of the best bid and ask prices, and other fields that change with each quote.
26  * It represents the most recent information that is available about the best quote on the market
27  * at any given moment of time.
28  *
29  * Bid corresponds to the best (maximal price) order to buy,
30  * ask corresponds to the best (minimal price) order to sell.
31  */
32 class DXFCPP_EXPORT Quote final : public MarketEvent, public LastingEvent {

52  void recomputeTimeMillisPart() noexcept {
53  data_.timeMillisSequence =
54  orOp(sal(time_util::getMillisFromTime(std::max(data_.askTime, data_.bidTime)), MILLISECONDS_SHIFT),
55  getSequence());

139  /**
140  * Returns sequence number of this quote to distinguish quotes that have the same @ref Quote::getTime() "time". This
141  * sequence number does not have to be unique and does not need to be sequential. Sequence can range from 0 to
142  * Quote::MAX_SEQUENCE.
143  *
144  * @return sequence of this quote.
145  */
146  std::int32_t getSequence() const noexcept {
147  return andOp(data_.timeMillisSequence, MAX_SEQUENCE);
148  }
149 
150  /**
151  * Changes @ref Quote::getSequence() "sequence number" of this quote.
152  *
153  * @param sequence The sequence.
154  *
155  * @see Quote::getSequence()
156  */
157  void setSequence(std::int32_t sequence) {
158  assert(sequence >= 0 && static_cast<std::uint32_t>(sequence) <= MAX_SEQUENCE);
159 
160  if (sequence < 0 || static_cast<std::uint32_t>(sequence) > MAX_SEQUENCE) {
161  throw std::invalid_argument("Invalid sequence value = " + std::to_string(sequence));
162  }
163 
164  data_.timeMillisSequence = orOp(andOp(data_.timeMillisSequence, ~MAX_SEQUENCE), sequence);
165  }

180  /**
