35/**
36 * Symbol that should be used with DXFeedSubscription class to subscribe for Candle events. `DXFeedSubscription` also
37 * accepts a string representation of the candle symbol for subscription.
38 *
39 * <h3>String representation</h3>
40 *
41 * The string representation of the candle symbol consist of a @ref CandleSymbol::getBaseSymbol() "baseSymbol" followed
42 * by an optional '&' with an @ref CandleSymbol::getExchange() "exchange" code letter and followed by an optional list
43 * of comma-separated key=value pairs in curly braces:
44 *
45 * <p>`<baseSymbol> [ '&' <exchange> ] [ '{' <key1>=<value1> [ ',' <key2>=<value2> ] ... '}' ]`
46 *
47 * <p>Properties of the candle symbol correspond to the keys in the string representation in the following way:
48 *
49 * <ul>
50 * <li>Empty key corresponds to @ref CandleSymbol::getPeriod() "period" &mdash; aggregation period of this symbol.
51 * The period value is composed of an optional @ref CandlePeriod::getValue() "value" which defaults to 1 when not
52 * specified, followed by a @ref CandlePeriod::getType() "type" string which is defined by one of the CandleType
53 * values and can be abbreviated to first letters. For example, a daily candle of "IBM" base symbol can be
54 * specified as "IBM{=d}" and 15 minute candle on it as "IBM{=15m}". The shortest possible abbreviation for
55 * CandleType::MONTH is "mo", so the monthly candle can be specified as "IBM{=mo}". When period is not specified,
56 * then the CandlePeriod::TICK aggregation period is assumed as default. Note, that tick aggregation may not be
57 * available on the demo system which is limited to a subset of symbols and aggregation periods.
58 * <li>"price" key corresponds to @ref CandleSymbol::getPrice() "price" &mdash; price type attribute of this symbol.
59 * The CandlePrice defines possible values with CandlePrice::LAST being default.
60 * For legacy backwards-compatibility purposes, most of the price values cannot be abbreviated, so a one-minute
61 * candle of "EUR/USD" bid price shall be specified with "EUR/USD{=m,price=bid}" candle symbol string. However, the
62 * CandlePrice::SETTLEMENT can be abbreviated to "s", so a daily candle on "/ES" futures settlement prices can be
63 * specified with "/ES{=d,price=s}" string.
64 * <li>"tho" key with a value of "true" corresponds to @ref CandleSymbol::getSession() "session" set to
65 * CandleSession::REGULAR which limits the candle to trading hours only, so a 133 tick candles on "GOOG" base symbol
66 * collected over trading hours only can be specified with "GOOG{=133t,tho=true}" string. Note, that the default daily
67 * candles for US equities are special for historical reasons and correspond to the way US equity exchange report their
68 * daily summary data. The volume the US equity default daily candle corresponds to the total daily traded volume,
69 * while open, high, low, and close correspond to the regular trading hours only.
70 * <li>"a" key corresponds to @ref CandleSymbol::getAlignment() "alignment" &mdash; alignment attribute of this symbol.
71 * The CandleAlignment defines possible values with CandleAlignment::MIDNIGHT being default. The alignment values
72 * can be abbreviated to the first letter. So, a 1 hour candle on a symbol "AAPL" that starts at the regular
73 * trading session at 9:30 am ET can be specified with "AAPL{=h,a=s,tho=true}". Contrast that to
74 * the "AAPL{=h,tho=true}" candle that is aligned at midnight and thus starts at 9:00 am.
75 * <li>"pl" key corresponds to @ref CandleSymbol::getPriceLevel() "price level" &mdash; price level attribute of this
76 * symbol. The CandlePriceLevel defines additional axis to split candles within particular price corridor in addition to
77 * CandlePeriod attribute with the default value `NaN`. So a one-minute candles of "AAPL" with price level 0.1
78 * shall be specified with "AAPL{=m,pl=0.1}".
79 * </ul>
80 *
81 * Keys in the candle symbol are case-sensitive, while values are not. The CandleSymbol::valueOf() method parses any
82 * valid string representation into a candle symbol object. The result of the candle symbol CandleSymbol::toString()
83 * method is always normalized: keys are ordered lexicographically, values are in lower-case and are abbreviated to
84 * their shortest possible form.
85 */

98 static std::string changeAttribute(std::string symbol, const CandleSymbolAttributeVariant &attribute) noexcept {
99 return std::visit(
100 [symbol = std::move(symbol)](auto &&a) {
101 return a.changeAttributeForSymbol(symbol);
102 },
103 attribute);
104 }
105
106 static std::string changeAttributes(std::string symbol,
107 std::initializer_list<CandleSymbolAttributeVariant> attributes) noexcept {
108 for (const auto &a : attributes) {
109 symbol = changeAttribute(symbol, a);
110 }
111
112 return symbol;
113 }
114
115 static std::string normalize(std::string symbol) noexcept {
116 symbol = CandlePrice::normalizeAttributeForSymbol(symbol);
117 symbol = CandleSession::normalizeAttributeForSymbol(symbol);
118 symbol = CandlePeriod::normalizeAttributeForSymbol(symbol);
119 symbol = CandleAlignment::normalizeAttributeForSymbol(symbol);
120 symbol = CandlePriceLevel::normalizeAttributeForSymbol(symbol);
121
122 return symbol;
123 }
124
125 void initTransientFields(bool force = false) noexcept {
126 baseSymbol_ = MarketEventSymbols::getBaseSymbol(symbol_);
127
128 if (!exchange_ || force) {
129 exchange_.emplace(CandleExchange::getAttributeForSymbol(symbol_));
130 }
131
132 if (!price_ || force) {
133 price_ = CandlePrice::getAttributeForSymbol(symbol_);
134 }
135
136 if (!session_ || force) {
137 session_ = CandleSession::getAttributeForSymbol(symbol_);
138 }
139
140 if (!period_ || force) {
141 period_ = CandlePeriod::getAttributeForSymbol(symbol_);
142 }
143
144 if (!alignment_ || force) {
145 alignment_ = CandleAlignment::getAttributeForSymbol(symbol_);
146 }
147
148 if (!priceLevel_ || force) {
149 priceLevel_ = CandlePriceLevel::getAttributeForSymbol(symbol_);
150 }
151 }
152
153 explicit CandleSymbol(std::string symbol) noexcept : symbol_{normalize(std::move(symbol))} {
154 initTransientFields();
155 }
156
157 CandleSymbol(std::string symbol, const CandleSymbolAttributeVariant &attribute) noexcept
