22 struct StopWatch final {
23  private:
24  mutable std::mutex mutex_{};
25  std::chrono::milliseconds elapsed_{};
26  std::chrono::steady_clock::time_point startTimeStamp_{};
27  std::atomic<bool> isRunning_{};
28 
29  public:
30  StopWatch() noexcept {
31  reset();
32  }
33 
34  void start() noexcept {
35  if (!isRunning_) {
36  std::lock_guard lock{mutex_};
37  startTimeStamp_ = std::chrono::steady_clock::now();
38  isRunning_ = true;
39  }
40  }
41 
42  void stop() noexcept {
43  if (isRunning_) {
44  auto endTimestamp = std::chrono::steady_clock::now();
45 
46  std::lock_guard lock{mutex_};
47  auto elapsedThisPeriod = endTimestamp - startTimeStamp_;
48  elapsed_ += std::chrono::duration_cast<std::chrono::milliseconds>(elapsedThisPeriod);
49  isRunning_ = false;
50  }
51  }
52 
53  void reset() noexcept {
54  std::lock_guard lock{mutex_};
55 
56  elapsed_ = std::chrono::milliseconds::zero();
57  isRunning_ = false;
58  startTimeStamp_ = std::chrono::steady_clock::time_point{};
59  }
60 
61  void restart() noexcept {
62  std::lock_guard lock{mutex_};
63 
64  elapsed_ = std::chrono::milliseconds::zero();
65  isRunning_ = true;
66  startTimeStamp_ = std::chrono::steady_clock::now();
67  }
68 
69  bool isRunning() const noexcept {
70  return isRunning_;
71  }
72 
73  std::chrono::milliseconds elapsed() const noexcept {
74  std::lock_guard lock{mutex_};
75 
76  auto elapsed = elapsed_;
77 
78  if (isRunning_) {
79  auto currentTimestamp = std::chrono::steady_clock::now();
80  auto elapsedUntilNow = currentTimestamp - startTimeStamp_;
81 
82  elapsed += std::chrono::duration_cast<std::chrono::milliseconds>(elapsedUntilNow);
83  }
84 
85  return elapsed;
86  }
87 };
