24/**
25 * A concept describing a symbol that can be wrapped.
26 *
27 * @tparam T Probable symbol type
28 */
29template <typename T>
30concept ConvertibleToSymbolWrapper =
31 ConvertibleToStringSymbol<std::decay_t<T>> || std::is_same_v<std::decay_t<T>, WildcardSymbol> ||
32 std::is_same_v<std::decay_t<T>, IndexedEventSubscriptionSymbol> ||
33 std::is_same_v<std::decay_t<T>, TimeSeriesSubscriptionSymbol> || std::is_same_v<std::decay_t<T>, CandleSymbol>;
34
35/**
36 * A concept that defines a collection of wrapped or wrapping symbols.
37 *
38 * @tparam Collection The collection type
39 */
40template <typename Collection>
41concept ConvertibleToSymbolWrapperCollection =
42 requires(Collection c) {
43 std::begin(c);
44 std::end(c);
45 } &&
46 (
47 requires(Collection c) {
48 { *std::begin(c) } -> dxfcpp::ConvertibleTo<SymbolWrapper>;
49 } ||
50 requires(Collection c) {
51 { *std::begin(c) } -> ConvertibleToSymbolWrapper;
52 });

62 class DXFCPP_EXPORT SymbolListUtils final {
63 static std::ptrdiff_t calculateGraalListSize(std::ptrdiff_t initSize) noexcept;
64 static void *newGraalList(std::ptrdiff_t size) noexcept;
65 static bool setGraalListElement(void *graalList, std::ptrdiff_t elementIdx, void *element) noexcept;
66 static bool freeGraalListElements(void *graalList, std::ptrdiff_t count) noexcept;
67
68 public:
69 template <typename SymbolIt> static void *toGraalList(SymbolIt begin, SymbolIt end) noexcept {
70 if constexpr (Debugger::isDebug) {
71 Debugger::debug("SymbolWrapper::toGraalList(symbols = " + elementsToString(begin, end) + ")");
72 }
73
74 auto size = calculateGraalListSize(std::distance(begin, end));
75
76 // Zero size is needed, for example, to clear the list of symbols.
77 auto *list = newGraalList(size);
78
79 if (!list || size == 0) {
80 return list;
81 }
82
83 std::ptrdiff_t elementIdx = 0;
84 bool needToFree = false;
85
86 for (auto it = begin; it != end && elementIdx < size; it++, elementIdx++) {
87 if constexpr (requires { it->toGraal(); }) {
88 needToFree = setGraalListElement(list, elementIdx, it->toGraal()) == false;
89 } else if constexpr (std::is_convertible_v<decltype(*it), SymbolWrapper> ||
90 dxfcpp::ConvertibleToSymbolWrapper<decltype(*it)>) {
91 needToFree = setGraalListElement(list, elementIdx, SymbolWrapper(*it).toGraal()) == false;
92 }
93
94 if (needToFree) {
95 break;
96 }
97 }
98
99 if (needToFree) {
100 freeGraalListElements(list, elementIdx);
101
102 return nullptr;
103 }
104
105 return list;
106 }
107
108 template <ConvertibleToSymbolWrapperCollection SymbolsCollection>
109 static void *toGraalList(const SymbolsCollection &collection) noexcept {
110 return SymbolListUtils::toGraalList(std::begin(collection), std::end(collection));
111 }
112
113 static void *toGraalList(std::initializer_list<SymbolWrapper> collection) noexcept {
114 return SymbolListUtils::toGraalList(collection.begin(), collection.end());
115 }
116
117 static void freeGraalList(void *graalList) noexcept;
118
119 static std::vector<SymbolWrapper> fromGraalList(void *graalList) noexcept;
120 };

216 /**
217 * Releases the memory occupied by the dxFeed Graal SDK structure (recursively if necessary).

219 * @param graalNative The pointer to the dxFeed Graal SDK structure.

225 /**
226 * Allocates memory for the dxFeed Graal SDK structure (recursively if necessary).
227 * Fills the dxFeed Graal SDK structure's fields by the data of the current entity (recursively if necessary).
228 * Returns the pointer to the filled structure.
229 *
230 * @return The pointer to the filled dxFeed Graal SDK structure
231 */
232 void *toGraal() const noexcept {
233 if constexpr (Debugger::isDebug) {
234 Debugger::debug("SymbolWrapper::toGraal()");
235 }
236
237 return std::visit(
238 [](const auto &symbol) {
239 return symbol.toGraal();
240 },
241 data_);
242 }
243
244 std::unique_ptr<void, decltype(&SymbolWrapper::freeGraal)> toGraalUnique() const noexcept {
245 return {toGraal(), SymbolWrapper::freeGraal};
246 }
