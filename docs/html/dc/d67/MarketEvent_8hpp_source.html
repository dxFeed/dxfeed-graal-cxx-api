19 /**
20  * Base class for all market events. All market events are POCO that
21  * extend this class. Market event classes are simple beans with setter and getter methods for their
22  * properties and minimal business logic. All market events have `eventSymbol` property that is
23  * defined by this class.
24  */
25 struct DXFCPP_EXPORT MarketEvent : public EventTypeWithSymbol<std::string> {
26  /// The alias to a type of shared pointer to the MarketEvent object
27  using Ptr = std::shared_ptr<MarketEvent>;
28 
29  private:
30  std::optional<std::string> eventSymbol_{};
31  std::int64_t eventTime_{};
32 
33  protected:
34  MarketEvent() noexcept = default;
35 
36  /**
37  * Protected constructor for concrete implementation classes that initializes `eventSymbol` property.
38  *
39  * @param eventSymbol The event symbol.
40  */
41  explicit MarketEvent(std::string eventSymbol) noexcept : eventSymbol_{std::move(eventSymbol)} {
42  }
43 
44  virtual void fillData(void *graalNative) noexcept;
45  virtual void fillGraalData(void *graalNative) const noexcept;
46  static void freeGraalData(void *graalNative) noexcept;
47 
48  public:
49  /**
50  * Returns symbol of this event.
51  *
52  * @return symbol of this event or dxfcpp::String::NUL (`std::string{"<null>"}`)
53  */
54  const std::string &getEventSymbol() const & noexcept override {
55  if (!eventSymbol_) {
56  return dxfcpp::String::NUL;
57  }
58 
59  return eventSymbol_.value();
60  }
61 
62  /**
63  * Returns symbol of this event.
64  *
65  * @return symbol of this event or `std::nullopt`.
66  */
67  const std::optional<std::string> &getEventSymbolOpt() const & noexcept override {
68  return eventSymbol_;
69  }
70 
71  /**
72  * Changes symbol of this event.
73  *
74  * @param eventSymbol The symbol of this event.
75  */
76  void setEventSymbol(const std::string &eventSymbol) noexcept override {
77  // TODO: check invalid utf-8 [EN-8233]
78  eventSymbol_ = eventSymbol;
79  }
80 
81  ///
82  std::int64_t getEventTime() const noexcept override {
83  return eventTime_;
84  }
85 
86  ///
87  void setEventTime(std::int64_t eventTime) noexcept override {
88  eventTime_ = eventTime;
89  }
90 };
