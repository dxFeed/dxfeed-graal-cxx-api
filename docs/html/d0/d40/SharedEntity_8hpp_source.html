16 /// Base abstract "shared entity" class. Has some helpers for dynamic polymorphism
17 struct DXFCPP_EXPORT SharedEntity : public Entity, std::enable_shared_from_this<SharedEntity> {
18  /// The alias to a type of shared pointer to the SharedEntity object.
19  using Ptr = std::shared_ptr<SharedEntity>;
20 
21  /**
22  * Checks that pointer to the current type could be converted to type T*
23  * In other words: whether type T belongs to the type hierarchy in which the current type resides.
24  * @tparam T The type to check
25  * @return true if type belongs to the type hierarchy in which the current type resides.
26  */
27  template <typename T> bool is() const noexcept {
28  try {
29  auto p = dynamic_cast<const T *>(this);
30  (void)(p);
31 
32  return true;
33  } catch (const std::bad_cast &) {
34  return false;
35  }
36  }
37 
38  /**
39  * Returns a pointer to the current object wrapped in a smart pointer to type T
40  *
41  * @warning Please do not use this method unless the object was created with `std::shared_ptr<T>(new T(...))` or
42  * `std::make_shared<T>(...)`
43  *
44  * @tparam T The type to convert to a pointer to
45  * @return a smart pointer to type T
46  */
47  template <typename T> std::shared_ptr<T> sharedAs() noexcept {
48  return std::dynamic_pointer_cast<T>(shared_from_this());
49  }
50 
51  /**
52  * Returns a pointer to the current object wrapped in a smart pointer to type T
53  *
54  * @warning Please do not use this method unless the object was created with `std::shared_ptr<T>(new T(...))` or
55  * `std::make_shared<T>(...)`
56  *
57  * @tparam T The type to convert to a pointer to
58  * @return a smart pointer to type T
59  */
60  template <typename T> std::shared_ptr<T> sharedAs() const noexcept {
61  return std::dynamic_pointer_cast<T>(shared_from_this());
62  }
