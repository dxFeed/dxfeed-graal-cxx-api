26/**
27 * Time and Sale represents a trade or other market event with price, like market open/close price, etc.
28 * Time and Sales are intended to provide information about trades <b>in a continuous time slice</b>
29 * (unlike Trade events which are supposed to provide snapshot about the <b>current last</b> trade).
30 *
31 * <p> Time and Sale events have unique @ref TimeAndSale::getIndex() "index" which can be used for later
32 * correction/cancellation processing.
33 *
34 * Some time and sale sources provide a consistent view of the set of known time and sales
35 * for a given time range when used with DXFeedTimeSeriesSubscription}.
36 * The corresponding information is carried in @ref TimeAndSale::getEventFlags() "eventFlags" property.
37 * The logic behind this property is detailed in IndexedEvent class documentation.
38 * Multiple event sources for the same symbol are not supported for time and sales, thus
39 * @ref TimeAndSale::getSource() "source" property is always @ref IndexedEventSource::DEFAULT "DEFAULT".
40 *
41 * <p> Regular subscription via DXFeedSubscription produces a stream of time and
42 * sale events as they happen and their @ref TimeAndSale::getEventFlags() "eventFlags" are always zero.
43 *
44 * Publishing of time and sales events follows the general rules explained in TimeSeriesEvent class
45 * documentation.
46 *
47 * This event is implemented on top of QDS record `TimeAndSale`.
48 */
49class DXFCPP_EXPORT TimeAndSale final : public MarketEvent, public TimeSeriesEvent {

51 friend class OptionSale;

53 static constexpr std::uint64_t SECONDS_SHIFT = 32ULL;
54 static constexpr std::uint64_t MILLISECONDS_SHIFT = 22ULL;
55 static constexpr std::uint64_t MILLISECONDS_MASK = 0x3ffULL;

57 /*
58 * Flags property has several significant bits that are packed into an integer in the following way:
59 * 31..16 15...8 7 6 5 4 3 2 1 0
60 * +--------+--------+----+----+----+----+----+----+----+----+
61 * | | TTE | | Side | SL | ETH| VT | Type |
62 * +--------+--------+----+----+----+----+----+----+----+----+
63 */
64
65 // TTE (TradeThroughExempt) values are ASCII chars in [0, 255].
66 static constexpr std::uint32_t TTE_MASK = 0xFFU;
67 static constexpr std::uint32_t TTE_SHIFT = 8U;
68
69 // SIDE values are taken from Side enum.
70 static constexpr std::uint32_t SIDE_MASK = 3U;
71 static constexpr std::uint32_t SIDE_SHIFT = 5U;
72
73 static constexpr std::uint32_t SPREAD_LEG = 1U << 4U;
74 static constexpr std::uint32_t ETH = 1U << 3U;
75 static constexpr std::uint32_t VALID_TICK = 1U << 2U;
76
77 // TYPE values are taken from TimeAndSaleType enum.
78 static constexpr std::uint32_t TYPE_MASK = 3U;
79 static constexpr std::uint32_t TYPE_SHIFT = 0;
80
81 /**
82 * EventFlags property has several significant bits that are packed into an integer in the following way:
83 * <br>
84 * <pre><tt>
85 * 31..7 6 5 4 3 2 1 0
86 * +--------+----+----+----+----+----+----+----+
87 * | | SM | | SS | SE | SB | RE | TX |
88 * +--------+----+----+----+----+----+----+----+
89 * </tt></pre>
90 */
