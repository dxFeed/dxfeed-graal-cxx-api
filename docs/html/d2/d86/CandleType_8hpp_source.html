22 /**
23  * Type of the candle aggregation period constitutes CandlePeriod type together its actual
24  * @ref CandlePeriod::getValue() "value".
25  */
26 struct DXFCPP_EXPORT CandleType {
27  /**
28  * Certain number of ticks.
29  */
30  static const CandleType TICK;
31 
32  /**
33  * Certain number of seconds.
34  */
35  static const CandleType SECOND;
36 
37  /**
38  * Certain number of minutes.
39  */
40  static const CandleType MINUTE;
41 
42  /**
43  * Certain number of hours.
44  */
45  static const CandleType HOUR;
46 
47  /**
48  * Certain number of days.
49  */
50  static const CandleType DAY;
51 
52  /**
53  * Certain number of weeks.
54  */
55  static const CandleType WEEK;
56 
57  /**
58  * Certain number of months.
59  */
60  static const CandleType MONTH;
61 
62  /**
63  * Certain number of option expirations.
64  */
65  static const CandleType OPTEXP;
66 
67  /**
68  * Certain number of years.
69  */
70  static const CandleType YEAR;
71 
72  /**
73  * Certain volume of trades.
74  */
75  static const CandleType VOLUME;
76 
77  /**
78  * Certain price change, calculated according to the following rules:
79  * <ol>
80  * <li>high(n) - low(n) = price range</li>
81  * <li>close(n) = high(n) or close(n) = low(n)</li>
82  * <li>open(n+1) = close(n)</li>
83  * </ol>
84  * where n is the number of the bar.
85  */
86  static const CandleType PRICE;
87 
88  /**
89  * Certain price change, calculated according to the following rules:
90  * <ol>
91  * <li>high(n) - low(n) = price range</li>
92  * <li>close(n) = high(n) or close(n) = low(n)</li>
93  * <li>open(n+1) = close(n) + tick size, if close(n) = high(n)</li>
94  * <li>open(n+1) = close(n) - tick size, if close(n) = low(n)</li>
95  * </ol>
96  * where n is the number of the bar.
97  */
98  static const CandleType PRICE_MOMENTUM;
99 
100  /**
101  * Certain price change, calculated according to the following rules:
102  * <ol>
103  * <li>high(n+1) - high(n) = price range or low(n) - low(n+1) = price range</li>
104  * <li>close(n) = high(n) or close(n) = low(n)</li>
105  * <li>open(n+1) = high(n), if high(n+1) - high(n) = price range</li>
106  * <li>open(n+1) = low(n), if low(n) - low(n+1) = price range</li>
107  * </ol>
108  * where n is the number of the bar.
109  */
110  static const CandleType PRICE_RENKO;
111 
112  static const std::unordered_map<std::string, std::reference_wrapper<const CandleType>, dxfcpp::StringHash,
113  std::equal_to<>>
114  BY_STRING;
115  static const std::vector<std::reference_wrapper<const CandleType>> VALUES;
116 
117  private:
118  std::string name_{};
119  std::string string_{};
120  std::int64_t periodIntervalMillis_{};
121 
122  CandleType(const dxfcpp::StringLikeWrapper &name, const dxfcpp::StringLikeWrapper &string,
123  std::int64_t periodIntervalMillis) noexcept
124  : name_{name}, string_{string}, periodIntervalMillis_{periodIntervalMillis} {
125  }
126 
127  public:
128  CandleType() noexcept = default;
129  virtual ~CandleType() noexcept = default;
130 
131  /**
132  * Returns candle type period in milliseconds as closely as possible.
133  * Certain types like CandleType::SECOND and CandleType::DAY span a specific number of milliseconds.
134  * CandleType::MONTH, CandleType::OPTEXP and CandleType::YEAR are approximate. Candle type period of
135  * CandleType::TICK, CandleType::VOLUME, CandleType::PRICE, CandleType::PRICE_MOMENTUM and CandleType::PRICE_RENKO
136  * is not defined and this method returns `0`.
137  *
138  * @return aggregation period in milliseconds.
139  */
140  std::int64_t getPeriodIntervalMillis() const noexcept {
141  return periodIntervalMillis_;
142  }
143 
144  /**
145  * Returns a name of this candle type
146  *
147  * @return A name of this candle type
148  */
149  const std::string &getName() const & noexcept {
150  return name_;
151  }
152 
153  /**
154  * Returns string representation of this candle type.
155  * The string representation of candle type is the shortest unique prefix of the lower case string that corresponds
156  * to its @ref CandleType::getName() "name".
157  * For example, CandleType::TICK is represented as `"t"`, while CandleType::MONTH is represented as `"mo"` to
158  * distinguish it from CandleType::MINUTE that is represented as `"m"`.
159  *
160  * @return string representation of this candle price type.
161  * @throws std::invalid_argument if the string representation is invalid.
162  */
163  const std::string &toString() const & noexcept {
164  return string_;
165  }
166 
167  /**
168  * Parses string representation of candle type into object.
169  * Any string that that is a prefix of candle type CandleType::getName() can be parsed
170  * (including the one that was returned by CandleType::toString()) and case is ignored for parsing.
171  *
172  * @param s The string representation of candle type.
173  * @return A candle type.
174  */
175  static std::reference_wrapper<const CandleType> parse(const dxfcpp::StringLikeWrapper &s) {
176  auto sw = s.operator std::string_view();
177  auto n = sw.length();
178 
179  if (n == 0) {
180  throw std::invalid_argument("Missing candle type");
181  }
182 
183  auto result = BY_STRING.find(sw);
184 
185  if (result != BY_STRING.end()) {
186  return result->second;
187  }
188 
189  for (const auto &typeRef : VALUES) {
190  const auto &name = typeRef.get().getName();
191 
192  // Tick|TICK|tick, Minute|MINUTE|minute, Second|SECOND|second, etc
193  if (name.length() >= n && iEquals(name.substr(0, n), sw)) {
194  return typeRef;
195  }
196 
197  // Ticks, Minutes, Seconds, etc
198  if (sw.ends_with("s") && iEquals(name, sw.substr(0, n - 1))) {
199  return typeRef;
200  }
201  }
202 
203  throw std::invalid_argument("Unknown candle type: " + s);
204  }
205 
206  bool operator==(const CandleType &candleType) const noexcept {
207  return string_ == candleType.string_;
208  }
209 };
