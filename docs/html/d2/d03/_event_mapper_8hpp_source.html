15struct DXFCPP_EXPORT EventMapper {
16 static std::vector<std::shared_ptr<EventType>> fromGraalList(void *graalNativeList) noexcept;
17
18 // TODO: check that EventIt is iterator of EventType* or shared_ptr<EventType> or T* or shared_ptr<T>, where T is
19 // child of EventType
20 template <typename EventIt> static void *toGraalList(EventIt begin, EventIt end) noexcept {
21 if constexpr (Debugger::isDebug) {
22 Debugger::debug("EventMapper::toGraalList(symbols = " + elementsToString(begin, end) + ")");
23 }
24
25 auto size = calculateGraalListSize(std::distance(begin, end));
26
27 // Zero size is needed, for example, to clear the list of symbols.
28 auto *list = newGraalList(size);
29
30 if (!list || size == 0) {
31 return list;
32 }
33
34 std::ptrdiff_t elementIdx = 0;
35 bool needToFree = false;
36
37 for (auto it = begin; it != end && elementIdx < size; it++, elementIdx++) {
38 if constexpr (requires { it->toGraal(); }) { // It<EventType>
39 needToFree = setGraalListElement(list, elementIdx, it->toGraal()) == false;
40 } else if constexpr (requires { (*it)->toGraal(); }) { // It<Ptr<EventType>>
41 needToFree = setGraalListElement(list, elementIdx, (*it)->toGraal()) == false;
42 }
43
44 if (needToFree) {
45 break;
46 }
47 }
48
49 if (needToFree) {
50 freeGraalListElements(list, elementIdx);
51
52 return nullptr;
53 }
54
55 return list;
56 }
57
58 static void freeGraalList(void *graalList) noexcept;
59
60 private:
61 static std::ptrdiff_t calculateGraalListSize(std::ptrdiff_t initSize) noexcept;
62 static void *newGraalList(std::ptrdiff_t size) noexcept;
63 static bool setGraalListElement(void *graalList, std::ptrdiff_t elementIdx, void *element) noexcept;
64 static bool freeGraalListElements(void *graalList, std::ptrdiff_t count) noexcept;
65};
