268 /**
269 * Adds listener for events.
270 * Event lister can be added only when subscription is not producing any events.
271 * The subscription must be either empty
272 * (its set of @ref DXFeedSubscription::getSymbols() "symbols" is empty or not @ref DXFeedSubscription::attach()
273 * "attached" to any feed (its set of change listeners is empty).
274 *
275 * This method does nothing if this subscription is closed.
276 *
277 * Example:
278 * ```cpp
279 * auto sub = endpoint->getFeed()->createSubscription({dxfcpp::Quote::TYPE, dxfcpp::TimeAndSale::TYPE});
280 *
281 * sub->addEventListener([](auto &&events) {
282 * for (const auto &e : events) {
283 * if (auto quote = e->template sharedAs<dxfcpp::Quote>(); quote) {
284 * std::cout << "Q : " + quote->toString() << std::endl;
285 * } else if (auto tns = e->template sharedAs<dxfcpp::TimeAndSale>(); tns) {
286 * std::cout << "TnS : " + tns->toString() << std::endl;
287 * }
288 * }
289 * });
290 *
291 * sub->addSymbols({"$TOP10L/Q", "$SP500#45", "$TICK", "SPX"});
292 * ```
293 *
294 * @tparam EventListener The listener type. Listener can be callable with signature: `void(const
295 * std::vector<std::shared_ptr<EventType>&)`
296 * @param listener The event listener
297 * @return The listener id
298 */
299 template <typename EventListener>
300 std::size_t addEventListener(EventListener &&listener) noexcept
301#if __cpp_concepts
302 requires requires {
303 { listener(std::vector<std::shared_ptr<EventType>>{}) } -> std::same_as<void>;
304 }
305#endif
306 {
307 if (!tryToSetEventListenerHandle()) {
308 return SimpleHandler<void(const std::vector<std::shared_ptr<EventType>> &)>::FAKE_ID;
309 }
310
311 return onEvent_ += listener;
312 }

314 /**
315 * Adds typed listener for events.
316 * Event lister can be added only when subscription is not producing any events.
317 * The subscription must be either empty
318 * (its set of @ref DXFeedSubscription::getSymbols() "symbols" is empty or not @ref DXFeedSubscription::attach()
319 * "attached" to any feed (its set of change listeners is empty).
320 *
321 * This method does nothing if this subscription is closed.
322 *
323 * Example:
324 * ```cpp
325 * auto sub = endpoint->getFeed()->createSubscription({dxfcpp::Quote::TYPE, dxfcpp::TimeAndSale::TYPE});
326 *
327 * sub->addEventListener(std::function([](const std::vector<std::shared_ptr<dxfcpp::Quotes>> &quotes) -> void {
328 * for (const auto &q : quotes) {
329 * std::cout << "Q : " + q->toString() << std::endl;
330 * }
331 * }));
332 *
333 * sub->addEventListener<dxfcpp::TimeAndSale>([](const auto &timeAndSales) -> void {
334 * for (const auto &tns : timeAndSales) {
335 * std::cout << "TnS : " + tns->toString() << std::endl;
336 * }
337 * });
338 *
339 * sub->addEventListener<dxfcpp::MarketEvent>([](const auto &marketEvents) -> void {
340 * for (const auto &me : marketEvents) {
341 * std::cout << "Market Event's symbol: " + me->getEventSymbol() << std::endl;
342 * }
343 * });
344 *
345 * sub->addSymbols({"$TOP10L/Q", "AAPL", "$TICK", "SPX"});
346 * ```
347 *
348 * @tparam EventT The event type (EventType's child with field TYPE, convertible to EventTypeEnum or MarketEvent or
349 * LastingEvent or TimeSeriesEvent or IndexedEvent)
350 * @param listener The listener. Listener can be callable with signature: `void(const
351 * std::vector<std::shared_ptr<EventT>&)`
352 * @return The listener id
353 */
354 template <typename EventT>
355 std::size_t addEventListener(std::function<void(const std::vector<std::shared_ptr<EventT>> &)> &&listener) noexcept
356#if __cpp_concepts
357 requires std::is_base_of_v<EventType, EventT> &&
358 (requires {
359 { EventT::TYPE } -> dxfcpp::ConvertibleTo<EventTypeEnum>;
360 } || std::is_same_v<EventT, MarketEvent> || std::is_same_v<EventT, LastingEvent> ||
361 std::is_same_v<EventT, TimeSeriesEvent> || std::is_same_v<EventT, IndexedEvent>)
362#endif
363 {
364 if (!tryToSetEventListenerHandle()) {
365 return SimpleHandler<void(const std::vector<std::shared_ptr<EventType>> &)>::FAKE_ID;
366 }
367
368 return onEvent_ += [l = listener](auto &&events) {
369 std::vector<std::shared_ptr<EventT>> filteredEvents{};
370
371 filteredEvents.reserve(events.size());
372
373 for (const auto &e : events) {
374 if (auto expected = e->template sharedAs<EventT>(); expected) {
375 filteredEvents.emplace_back(expected);
376 }
377 }
378
379 l(filteredEvents);
380 };
381 }
