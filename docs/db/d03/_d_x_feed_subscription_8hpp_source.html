556 /**
557 * Adds listener for events.
558 * Event lister can be added only when subscription is not producing any events.
559 * The subscription must be either empty
560 * (its set of @ref DXFeedSubscription::getSymbols() "symbols" is empty or not @ref DXFeedSubscription::attach()
561 * "attached" to any feed (its set of change listeners is empty).
562 *
563 * This method does nothing if this subscription is closed.
564 *
565 * Example:
566 * ```cpp
567 * auto sub = endpoint->getFeed()->createSubscription({dxfcpp::Quote::TYPE, dxfcpp::TimeAndSale::TYPE});
568 *
569 * sub->addEventListener([](auto &&events) {
570 * for (const auto &e : events) {
571 * if (auto quote = e->template sharedAs<dxfcpp::Quote>(); quote) {
572 * std::cout << "Q : " + quote->toString() << std::endl;
573 * } else if (auto tns = e->template sharedAs<dxfcpp::TimeAndSale>(); tns) {
574 * std::cout << "TnS : " + tns->toString() << std::endl;
575 * }
576 * }
577 * });
578 *
579 * sub->addSymbols({"$TOP10L/Q", "$SP500#45", "$TICK", "SPX"});
580 * ```
581 *
582 * @tparam EventListener The listener type. Listener can be callable with signature: `void(const
583 * std::vector<std::shared_ptr<EventType>&)`
584 * @param listener The event listener
585 * @return The listener id
586 */
587 template <typename EventListener>
588 std::size_t addEventListener(EventListener &&listener)
589#if __cpp_concepts
590 requires requires {
591 { listener(std::vector<std::shared_ptr<EventType>>{}) } -> std::same_as<void>;
592 }
593#endif
594 {
595 if (!tryToSetEventListenerHandle()) {
596 return OnEventHandler::FAKE_ID;
597 }
598
599 return onEvent_ += listener;
600 }

602 /**
603 * Adds typed listener for events.
604 * Event lister can be added only when subscription is not producing any events.
605 * The subscription must be either empty
606 * (its set of @ref DXFeedSubscription::getSymbols() "symbols" is empty or not @ref DXFeedSubscription::attach()
607 * "attached" to any feed (its set of change listeners is empty).
608 *
609 * This method does nothing if this subscription is closed.
610 *
611 * Example:
612 * ```cpp
613 * auto sub = endpoint->getFeed()->createSubscription({dxfcpp::Quote::TYPE, dxfcpp::TimeAndSale::TYPE});
614 *
615 * sub->addEventListener(std::function([](const std::vector<std::shared_ptr<dxfcpp::Quotes>> &quotes) -> void {
616 * for (const auto &q : quotes) {
617 * std::cout << "Q : " + q->toString() << std::endl;
618 * }
619 * }));
620 *
621 * sub->addEventListener<dxfcpp::TimeAndSale>([](const auto &timeAndSales) -> void {
622 * for (const auto &tns : timeAndSales) {
623 * std::cout << "TnS : " + tns->toString() << std::endl;
624 * }
625 * });
626 *
627 * sub->addEventListener<dxfcpp::MarketEvent>([](const auto &marketEvents) -> void {
628 * for (const auto &me : marketEvents) {
629 * std::cout << "Market Event's symbol: " + me->getEventSymbol() << std::endl;
630 * }
631 * });
632 *
633 * sub->addSymbols({"$TOP10L/Q", "AAPL", "$TICK", "SPX"});
634 * ```
635 *
636 * @tparam EventT The event type (EventType's child with field TYPE, convertible to EventTypeEnum or MarketEvent
637 * or LastingEvent or TimeSeriesEvent or IndexedEvent)
638 * @param listener The listener. Listener can be callable with signature: `void(const
639 * std::vector<std::shared_ptr<EventT>&)`
640 * @return The listener id
641 */
642 template <typename EventT>
643 std::size_t addEventListener(std::function<void(const std::vector<std::shared_ptr<EventT>> &)> &&listener)
644#if __cpp_concepts
645 requires std::is_base_of_v<EventType, EventT> &&
646 (requires {
647 { EventT::TYPE } -> dxfcpp::ConvertibleTo<EventTypeEnum>;
648 } || std::is_same_v<EventT, MarketEvent> || std::is_same_v<EventT, LastingEvent> ||
649 std::is_same_v<EventT, TimeSeriesEvent> || std::is_same_v<EventT, IndexedEvent>)
650#endif
651 {
652 if (!tryToSetEventListenerHandle()) {
653 return SimpleHandler<void(const std::vector<std::shared_ptr<EventType>> &)>::FAKE_ID;
654 }
655
656 return onEvent_ += [l = listener](auto &&events) {
657 std::vector<std::shared_ptr<EventT>> filteredEvents{};
658
659 filteredEvents.reserve(events.size());
660
661 for (const auto &e : events) {
662 if (auto expected = e->template sharedAs<EventT>(); expected) {
663 filteredEvents.emplace_back(expected);
664 }
665 }
666
667 l(filteredEvents);
668 };
669 }
