15 /**
16  * Action enum for the Full Order Book (FOB) Orders. Action describes business meaning of the Order event:
17  * whether order was added or replaced, partially or fully executed, etc.
18  */
19 struct DXFCPP_EXPORT OrderAction : Enum<OrderAction, std::uint32_t> {
20  using Enum::Enum;
21 
22  /**
23  * Default enum value for orders that do not support "Full Order Book" and for backward compatibility -
24  * action must be derived from other Order fields.
25  *
26  * <p>All Full Order Book related fields for this action will be empty.
27  */
28  static const OrderAction UNDEFINED;
29 
30  /**
31  * New Order is added to Order Book.
32  *
33  * <p>Full Order Book fields:
34  * <ul>
35  * <li>@ref Order::getOrderId() "orderId" - always present</li>
36  * <li>@ref Order::getAuxOrderId() "auxOrderId" - ID of the order replaced by this new order - if available.</li>
37  * <li>Trade fields will be empty</li>
38  * </ul>
39  */
40  static const OrderAction NEW;
41 
42  /**
43  * Order is modified and price-time-priority is not maintained (i.e. order has re-entered Order Book).
44  * Order @ref Order::getEventSymbol() "symbol" and @ref Order::getOrderSide() "side" will remain the same.
45  *
46  * <p>Full Order Book fields:
47  * <ul>
48  * <li>@ref Order::getOrderId() "orderId" - always present</li>
49  * <li>Trade fields will be empty</li>
50  * </ul>
51  */
52  static const OrderAction REPLACE;
53 
54  /**
55  * Order is modified without changing its price-time-priority (usually due to partial cancel by user).
56  * Order's @ref Order::getSize() "size" will contain new updated size.
57  *
58  * <p>Full Order Book fields:
59  * <ul>
60  * <li>@ref Order::getOrderId() "orderId" - always present</li>
61  * <li>Trade fields will be empty</li>
62  * </ul>
63  */
64  static const OrderAction MODIFY;
65 
66  /**
67  * Order is fully canceled and removed from Order Book.
68  * Order's @ref Order::getSize() "size" will be equal to 0.
69  *
70  * <p>Full Order Book fields:
71  * <ul>
72  * <li>@ref Order::getOrderId() "orderId" - always present</li>
73  * <li>@ref Order::getAuxOrderId() "auxOrderId" - ID of the new order replacing this order - if available.</li>
74  * <li>Trade fields will be empty</li>
75  * </ul>
76  */
77  static const OrderAction DELETE;
78 
79  /**
80  * Size is changed (usually reduced) due to partial order execution.
81  * Order's @ref Order::getSize() "size" will be updated to show current outstanding size.
82  *
83  * <p>Full Order Book fields:
84  * <ul>
85  * <li>@ref Order::getOrderId() "orderId" - always present</li>
86  * <li>@ref Order::getAuxOrderId() "auxOrderId" - aggressor order ID, if available</li>
87  * <li>@ref Order::getTradeId() "tradeId" - if available</li>
88  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" -
89  * contain size and price of this execution</li>
90  * </ul>
91  */
92  static const OrderAction PARTIAL;
93 
94  /**
95  * Order is fully executed and removed from Order Book.
96  * Order's {@ref Order#getSize() size} will be equals to 0.
97  *
98  * <p>Full Order Book fields:
99  * <ul>
100  * <li>@ref Order::getOrderId() "orderId" - always present</li>
101  * <li>@ref Order::getAuxOrderId() "auxOrderId" - aggressor order ID, if available</li>
102  * <li>@ref Order::getTradeId() "tradeId" - if available</li>
103  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" -
104  * contain size and price of this execution - always present</li>
105  * </ul>
106  */
107  static const OrderAction EXECUTE;
108 
109  /**
110  * Non-Book Trade - this Trade not refers to any entry in Order Book.
111  * Order's @ref Order::getSize() "size" and @ref Order::getPrice() "price" will be equals to 0.
112  *
113  * <p>Full Order Book fields:
114  * <ul>
115  * <li>@ref Order::getOrderId() "orderId" - always empty</li>
116  * <li>@ref Order::getTradeId() "tradeId" - if available</li>
117  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" -
118  * contain size and price of this trade - always present</li>
119  * </ul>
120  */
121  static const OrderAction TRADE;
122 
123  /**
124  * Prior Trade/Order Execution bust.
125  * Order's @ref Order::getSize() "size" and @ref Order::getPrice() "price" will be equals to 0.
126  *
127  * <p>Full Order Book fields:
128  * <ul>
129  * <li>@ref Order::getOrderId() "orderId" - always empty</li>
130  * <li>@ref Order::getTradeId() "tradeId" - always present</li>
131  * <li>@ref Order::getTradeSize() "tradeSize" and @ref Order::getTradePrice() "tradePrice" - always empty</li>
132  * </ul>
133  */
134  static const OrderAction BUST;
135 };
