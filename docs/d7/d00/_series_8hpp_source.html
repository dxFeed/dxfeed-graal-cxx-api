23/**
24 * Series event is a snapshot of computed values that are available for all option series for
25 * a given underlying symbol based on the option prices on the market.
26 * It represents the most recent information that is available about the corresponding values on
27 * the market at any given moment of time.
28 *
29 * <p>Series is an IndexedEvent with multiple instances of event available for
30 * each underlying symbol. Each series event instance corresponds to an OptionSeries
31 * of the corresponding underlying. The correspondence between a series event instance and
32 * an OptionSeries is established via @ref Series::getExpiration() "expiration" property.
33 * If case where there are multiple series at the same expiration day id, then series events are
34 * are ordered by their @ref Series::getIndex() "index" in the same order as the corresponding
35 * OptionSeries are @ref OptionSeries::compareTo(OptionSeries) "ordered" by their attributes.
36 *
37 * <h3><a name="eventFlagsSection">Event flags, transactions and snapshots</a></h3>
38 *
39 * Series data source provides a consistent view of the set of known series.
40 * The corresponding information is carried in @ref ::getEventFlags() "eventFlags" property.
41 * The logic behind this property is detailed in IndexedEvent class documentation.
42 * Multiple event sources for the same symbol are not supported for series, thus
43 * @ref ::getSource() "source" property is always @ref IndexedEventSource::DEFAULT "DEFAULT".
44 *
45 * <p>IndexedEventModel class handles all the snapshot and transaction logic and conveniently represents
46 * a list current of events.
47 *
48 * <h3>Implementation details</h3>
49 *
50 * This event is implemented on top of QDS records `Series`.
51 */

348 * @return time of this series event.
349 */
350 std::int64_t getTime() const noexcept {
351 return sar(data_.timeSequence, SECONDS_SHIFT) * 1000 +
352 andOp(sar(data_.timeSequence, MILLISECONDS_SHIFT), MILLISECONDS_MASK);
353 }

362 data_.timeSequence =
363 orOp(orOp(sal(static_cast<std::int64_t>(time_util::getSecondsFromTime(time)), SECONDS_SHIFT),
364 sal(static_cast<std::int64_t>(time_util::getMillisFromTime(time)), MILLISECONDS_SHIFT)),
365 getSequence());
366 }
