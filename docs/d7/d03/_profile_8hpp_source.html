24/**
25 * Profile information snapshot that contains security instrument description.
26 * It represents the most recent information that is available about the traded security
27 * on the market at any given moment of time.
28 */
29class DXFCPP_EXPORT Profile final : public MarketEvent, public LastingEvent {

32 /*
33 * Flags property has several significant bits that are packed into an integer in the following way:
34 * 31..4 3 2 1 0
35 * +--------+----+----+----+----+
36 * | | SSR | Status |
37 * +--------+----+----+----+----+
38 */
39
40 // SSR values are taken from ShortSaleRestriction enum.
41 static const std::uint32_t SSR_MASK = 3;
42 static const std::uint32_t SSR_SHIFT = 2;
43
44 // STATUS values are taken from TradingStatus enum.
45 static const std::uint32_t STATUS_MASK = 3;
46 static const std::uint32_t STATUS_SHIFT = 0;
47
48 struct Data {

65 };
