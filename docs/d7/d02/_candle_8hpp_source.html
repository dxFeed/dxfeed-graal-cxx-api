26/**
27 * Candle event with open, high, low, close prices and other information for a specific period.
28 * Candles are build with a specified CandlePeriod using a specified CandlePrice type with a data taken from the
29 * specified CandleExchange from the specified CandleSession with further details of aggregation provided by
30 * CandleAlignment.
31 *
32 * <p> Event symbol of the candle is represented with CandleSymbol class.
33 * Since the `Candle is a time-series event, it is typically subscribed to using DXFeedTimeSeriesSubscription class
34 * that handles the necessarily wrapping of the symbol into TimeSeriesSubscriptionSymbol to specify a subscription
35 * time range.
36 *
37 * <h3><a name="eventFlagsSection">Event flags, transactions and snapshots</a></h3>
38 *
39 * Some candle sources provide a consistent view of the set of known candles.
40 * The corresponding information is carried in @ref Candle::getEventFlags() "eventFlags" property.
41 * The logic behind this property is detailed in IndexedEvent class documentation.
42 * Multiple event sources for the same symbol are not supported for candles, thus @ref Candle::getSource() "source"
43 * property is always @ref IndexedEventSource::DEFAULT "DEFAULT".
44 *
45 * <p>TimeSeriesEventModel class handles all the snapshot and transaction logic and conveniently represents a list
46 * current of time-series events order by their @ref Candle::getTime() "time".
47 * It relies on the code of AbstractIndexedEventModel to handle this logic.
48 * Use the source code of AbstractIndexedEventModel for clarification on transactions and snapshot logic.
49 *
50 * <h3>Publishing Candles</h3>
51 *
52 * Publishing of candle events follows the general rules explained in TimeSeriesEvent class documentation.
53 *
54 * <h3>Implementation details</h3>
55 *
56 * This event is implemented on top of QDS record `TradeHistory` for tick candles with CandlePeriod::TICK,
57 * records `Trade.<period>` for a certain set of popular periods, and QDS record `Candle` for arbitrary custom
58 * periods, with a set of `Candle{<attributes>}` records for a popular combinations of custom candle symbol attributes
59 * like CandlePrice for an efficient support of bid-ask charting.
60 */
61class DXFCPP_EXPORT Candle final : public EventTypeWithSymbol<CandleSymbol>,
62 public TimeSeriesEvent,
63 public LastingEvent {
64 friend struct EventMapper;

365 /**
366 * Returns timestamp of the event in milliseconds.
367 *
368 * @return timestamp of the event in milliseconds
369 */
370 std::int64_t getTime() const noexcept override {
371 return sar(data_.index, SECONDS_SHIFT) * 1000 + andOp(sar(data_.index, MILLISECONDS_SHIFT), MILLISECONDS_MASK);
372 }

374 /**
375 * Changes timestamp of the event in milliseconds.
376 *
377 * @param time timestamp of the event in milliseconds.
378 * @see Candle::getTime()

381 data_.index = orOp(orOp(sal(time_util::getSecondsFromTime(time), SECONDS_SHIFT),
382 sal(time_util::getMillisFromTime(time), MILLISECONDS_SHIFT)),
383 getSequence());
384 }

415 /**
416 * Returns the sequence number of this event to distinguish events that have the same @ref Candle::getTime() "time".
417 * This sequence number does not have to be unique and does not need to be sequential.
418 * Sequence can range from 0 to Candle::MAX_SEQUENCE.
419 *
420 * @return The sequence number of this event
421 */
422 std::int32_t getSequence() const noexcept {
423 return static_cast<std::int32_t>(andOp(data_.index, MAX_SEQUENCE));
424 }
425
426 /**
427 * Changes @ref Candle::getSequence() "sequence number" of this event.
428 *
429 * @param sequence the sequence.
430 * @see Candle::getSequence()
431 */
432 void setSequence(int sequence) noexcept {
433 // TODO: Improve error handling [EN-8232]
434 assert(sequence >= 0 && sequence <= MAX_SEQUENCE);
435
436 data_.index = orOp(andOp(data_.index, ~MAX_SEQUENCE), sequence);
437 }
