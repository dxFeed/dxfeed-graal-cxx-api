30/**
31 * Main entry class for dxFeed API (<b>read it first</b>).
32 *
33 * <h3>Sample usage</h3>
34 *
35 * This section gives sample usage scenarios.
36 *
37 * <h4>Default singleton instance</h4>
38 *
39 * There is a singleton instance of the feed that is returned by DXFeed::getInstance() method.
40 * It is created on the first use with default configuration properties that are explained in detail in
41 * documentation for DXEndpoint class in the "Default properties" section.
42 *
43 * <p>In particular,
44 * you can provide a default address to connect and credentials using
45 * "@ref DXEndpoint::DXFEED_ADDRESS_PROPERTY "dxfeed.address"",
46 * "@ref DXEndpoint::DXFEED_USER_PROPERTY "dxfeed.user"", and
47 * "@ref DXEndpoint::DXFEED_PASSWORD_PROPERTY "dxfeed.password""
48 * system properties or by putting them into
49 * "@ref DXEndpoint::DXFEED_PROPERTIES_PROPERTY "dxfeed.properties""
50 * file in the same directory. dxFeed API samples come with a ready-to-use "<b>dxfeed.properties</b>"
51 * file that contains an address of dxFeed demo feed at "<b>demo.dxfeed.com:7300</b>" and
52 * demo access credentials.
53 *
54 * <h4>Subscribe for single event type</h4>
55 *
56 * The following code creates listener that prints mid price for each quote
57 * and subscribes for quotes on SPDR S&P 500 ETF symbol:
58 * <pre><tt>
59 * auto sub = @ref DXFeed "DXFeed"::@ref DXFeed::getInstance() "getInstance"()->@ref DXFeed::createSubscription()
60 * "createSubscription"(Quote::TYPE);
61 *
62 * sub->@ref DXFeedSubscription::addEventListener() "addEventListener"<Quote>([](const auto& quotes) {
63 * for (const auto& quote : quotes) {
64 * std::cout << "Mid = " + (quote->@ref Quote::getBidPrice() "getBidPrice"() + quote->@ref Quote::getAskPrice()
65 * "getAskPrice"()) / 2) << std::endl;
66 * }
67 * });
68 *
69 * sub->@ref DXFeedSubscription::addSymbols() "addSymbols"("SPY");</tt></pre>
70 *
71 * Note, that order of calls is important here. By attaching listeners first and then setting
72 * subscription we ensure that the current quote gets received by the listener. See DXFeedSubscription::addSymbols() for
73 * details. If a set of symbols is changed first, then @ref DXFeedSubscription::addEventListener()
74 * "sub->addEventListener" raises an IllegalStateException in JVM to protected from hard-to-catch bugs with potentially
75 * missed events.
76 *
77 * <h4>Subscribe for multiple event types</h4>
78 *
79 * The following code creates listener that prints each received event and
80 * subscribes for quotes and trades on SPDR S&P 500 ETF symbol:
81 * <pre><tt>
82 * auto sub = @ref DXFeed "DXFeed"::@ref DXFeed::getInstance() "getInstance"()->@ref DXFeed::createSubscription()
83 * "createSubscription"({Quote::TYPE, Trade::TYPE});
84 *
85 * sub->@ref DXFeedSubscription::addEventListener() "addEventListener"([](auto&& events) {
86 * for (const auto& event : events) {
87 * std::cout << event << std::endl;
88 * }
89 * });
90 *
91 * sub->@ref DXFeedSubscription::addSymbols() "addSymbols"("SPY");</tt></pre>
92 *
93 * <h4>Subscribe for event and query periodically its last value</h4>
94 *
95 * The following code subscribes for trades on SPDR S&P 500 ETF symbol and
96 * prints last trade every second.
97 *
98 * <pre><tt>
99 * using namespace std::chrono_literals;
100 *
101 * auto sub = @ref DXFeed "DXFeed"::@ref DXFeed::getInstance() "getInstance"()->@ref DXFeed::createSubscription()
102 * "createSubscription"({Trade::TYPE});
103 *
104 * sub->@ref DXFeedSubscription::addSymbols() "addSymbols"("SPY");
105 *
106 * auto feed = @ref DXFeed "DXFeed"::@ref DXFeed::getInstance() "getInstance"();
107 *
108 * while (true) {
109 * std::cout << System.out.println(feed->@ref DXFeed::getLastEvent() "getLastEvent"(Trade::create("SPY")));
110 * std::this_thread::sleep_for(1000ms);
111 * }</tt></pre>
112 *
113 * <h3>Threads and locks</h3>
114 *
115 * This class is thread-safe and can be used concurrently from multiple threads without external synchronization.
116 */

211 /**
212 * Returns the last event for the specified event instance.
213 * This method works only for event types that implement LastingEvent marker interface.
214 * This method <b>does not</b> make any remote calls to the uplink data provider.
215 * It just retrieves last received event from the local cache of this feed.
216 * The events are stored in the cache only if there is some attached DXFeedSubscription that is subscribed to the
217 * corresponding symbol and event type.
218 * WildcardSymbol::ALL subscription does not count for that purpose.
219 *
220 * <p>Use @ref ::getLastEventPromise() "getLastEventPromise" method if an event needs to be requested in the absence
221 * of subscription.
222 *
223 * <p> This method fills in the values for the last event into the `event argument.
224 * If the last event is not available for any reason (no subscription, no connection to uplink, etc.)
225 * then the event object is not changed.
226 * This method always returns the same `event` instance that is passed to it as an argument.
227 *
228 * <p>This method provides no way to distinguish a case when there is no subscription from the case when
229 * there is a subscription, but the event data have not arrived yet. It is recommended to use
230 * @ref ::getLastEventIfSubscribed() "getLastEventIfSubscribed" method instead of this `getLastEvent` method to
231 * fail-fast in case when the subscription was supposed to be set by the logic of the code, since
232 * @ref ::getLastEventIfSubscribed() "getLastEventIfSubscribed" method returns null when there is no subscription.
233 *
234 * <p>Note, that this method does not work when DXEndpoint was created with
235 * @ref DXEndpoint::Role::STREAM_FEED "STREAM_FEED" role (never fills in the event).
236 *
237 * @tparam E The type of event.
238 * @param event The event.
239 * @return The same event.
240 */
241 template <Derived<LastingEvent> E> std::shared_ptr<E> getLastEvent(std::shared_ptr<E> event) {
242 if (auto last = getLastEventIfSubscribed<E>(event->getEventSymbol())) {
243 event->assign(last);
244 }
245
246 return event;
247 }

249 /**
250 * Returns the last events for the specified list of event instances.
251 * This is a bulk version of @ref ::getLastEvent() "getLastEvent" method.
252 *
253 * <p>Note, that this method does not work when DXEndpoint was created with
254 * @ref DXEndpoint::Role::STREAM_FEED "STREAM_FEED" role.
255 *
256 * @tparam Collection The collection type.
257 * @param events The collection of shared ptrs of events.
258 * @return The same collection of shared ptrs of events.
259 */
260 template <typename Collection, typename Element = std::decay_t<decltype(std::begin(Collection()))>,
261 typename Event = std::decay_t<decltype(*Element())>>
262 const Collection &getLastEvents(const Collection &events) {
263 static_assert(
264 std::is_same_v<Element, std::shared_ptr<Event>> && std::is_base_of_v<LastingEvent, Event>,
265 "The collection element must be of type `std::shared_ptr<Event>`, where `Event` is a descendant of "
266 "`LastingEvent`");
267
268 for (auto e : events) {
269 getLastEvent(e);
270 }
271
272 return events;
273 }

409 /**
410 * Requests the last event for the specified event type and symbol.
411 * This method works only for event types that implement LastingEvent marker "interface".
412 * This method requests the data from the uplink data provider, creates new event of the specified event type,
413 * and completes the resulting promise with this event.
414 *
415 * <p>The promise is cancelled when the underlying DXEndpoint is @ref DXEndpoint::close() "closed".
416 * If the event is not available for any transient reason (no subscription, no connection to uplink, etc),
417 * then the resulting promise completes when the issue is resolved, which may involve an arbitrarily long wait.
418 * Use Promise::await() method to specify timeout while waiting for promise to complete.
419 * If the event is permanently not available (not supported), then the promise completes exceptionally with
420 * JavaException "IllegalArgumentException".
421 *
422 * <p>There is a bulk version of this method that works much faster for a single event type and multiple symbols.
423 * See getLastEventsPromises() .
424 *
425 * <p>Note, that this method does not work when DXEndpoint was created with @ref DXEndpoint::Role::STREAM_FEED
426 * "STREAM_FEED" role (promise completes exceptionally).
427 *
428 * @tparam E The type of event.
429 * @param symbol The symbol.
430 * @return The promise for the result of the request.
431 */
432 template <Derived<LastingEvent> E>
433 std::shared_ptr<Promise<std::shared_ptr<E>>> getLastEventPromise(const SymbolWrapper &symbol) const {
434 return std::make_shared<Promise<std::shared_ptr<E>>>(getLastEventPromiseImpl(E::TYPE, symbol));
435 }

591 *

593 * have multiple sources (like Series). For events with multiple sources (like Order,

595 * This method does not support <em>synthetic</em> sources of orders (orders that are automatically
596 * generated from Quote events).
597 *
598 * <p>This method does not accept an instance of IndexedEventSubscriptionSymbol as a `symbol`.
599 * The later class is designed for use with DXFeedSubscription and to observe source-specific subscription
600 * in DXPublisher.
601 *
602 * <h3>Event flags and consistent snapshot</h3>
603 *
604 * This method completes promise only when a consistent snapshot of indexed events has been received from
605 * the data feed. The @ref IndexedEvent::getEventFlags() "eventFlags" property of the events in the resulting list
606 * is always zero.
607 *

609 * the latter expects events in a different order and with an appropriate flags set. See documentation on a specific
610 * event class for details on how they should be published.
611 *

613 * @param symbol The symbol.
614 * @param source The source.
615 * @return The promise for the result of the request.
