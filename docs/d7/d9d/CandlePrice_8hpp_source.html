19 /**
20  * Price type attribute of CandleSymbol defines price that is used to build the candles.
21  *
22  * <h3>Implementation details</h3>
23  *
24  * This attribute is encoded in a symbol string with
25  * @ref MarketEventSymbols::getAttributeStringByKey() "MarketEventSymbols.getAttributeStringByKey",
26  * @ref MarketEventSymbols::changeAttributeStringByKey() "changeAttributeStringByKey", and
27  * @ref MarketEventSymbols::removeAttributeStringByKey() "removeAttributeStringByKey" methods.
28  * The key to use with these methods is available via CandlePrice::ATTRIBUTE_KEY constant.
29  * The value that this key shall be set to is equal to the corresponding CandlePrice::toString()
30  */
31 struct DXFCPP_EXPORT CandlePrice : public CandleSymbolAttribute {
32  /**
33  * Last trading price.
34  */
35  static const CandlePrice LAST;
36 
37  /**
38  * Quote bid price.
39  */
40  static const CandlePrice BID;
41 
42  /**
43  * Quote ask price.
44  */
45  static const CandlePrice ASK;
46 
47  /**
48  * Market price defined as average between quote bid and ask prices.
49  */
50  static const CandlePrice MARK;
51 
52  /**
53  * Official settlement price that is defined by exchange or last trading price otherwise.
54  * It updates based on all PriceType values:
55  * PriceType::INDICATIVE, PriceType::PRELIMINARY, and PriceType::FINAL}.
56  */
57  static const CandlePrice SETTLEMENT;
58 
59  /**
60  * Default price type is CandlePrice::LAST.
61  */
62  static const CandlePrice DEFAULT;
63 
64  /**
65  * The attribute key that is used to store the value of `CandlePrice` in a symbol string using methods of
66  * MarketEventSymbols class.
67  * The value of this constant is "price".
68  * The value that this key shall be set to is equal to the corresponding CandlePrice::toString()
69  */
70  static const std::string ATTRIBUTE_KEY;
71 
72  static const std::unordered_map<std::string, std::reference_wrapper<const CandlePrice>> BY_STRING;
73  static const std::vector<std::reference_wrapper<const CandlePrice>> VALUES;
74 
75  private:
76 
77  std::string string_;
78 
79  explicit CandlePrice(std::string string) noexcept : string_{std::move(string)} {
80  }
81 
82  public:
83  CandlePrice() noexcept = default;
84  virtual ~CandlePrice() noexcept = default;
85 
86  /**
87  * Returns candle event symbol string with this candle price type set.
88  * @param symbol original candle event symbol.
89  * @return candle event symbol string with this candle price type set.
90  */
91  std::string changeAttributeForSymbol(const std::string &symbol) const noexcept override {
92  return *this == DEFAULT ? MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY)
93  : MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY, toString());
94  }
95 
96  /**
97  * Returns string representation of this candle price type.
98  * The string representation of candle price type is a lower case string that corresponds to
99  * its @ref CandlePrice::name() "name". For example, CandlePrice::LAST is represented as "last".
100  *
101  * @return string representation of this candle price type.
102  */
103  const std::string &toString() const & noexcept {
104  return string_;
105  }
106 
107  bool operator==(const CandlePrice &candlePrice) const noexcept {
108  return string_ == candlePrice.string_;
109  }
110 
111  /**
112  * Parses string representation of candle price type into object.
113  * Any string that was returned by CandlePrice::toString() can be parsed and case is ignored for parsing.
114  *
115  * @param s The string representation of candle price type.
116  * @return The candle price type (reference) or std::nullopt if there is no supported attribute's value.
117  */
118  static std::optional<std::reference_wrapper<const CandlePrice>> parse(const std::string &s) noexcept {
119  auto n = s.length();
120 
121  if (n == 0) {
122  // TODO: error handling throw IllegalArgumentException("Missing candle price");
123 
124  return std::nullopt;
125  }
126 
127  auto found = BY_STRING.find(s);
128 
129  if (found != BY_STRING.end()) {
130  return found->second;
131  }
132 
133  for (const auto &priceRef : VALUES) {
134  const auto &priceStr = priceRef.get().toString();
135 
136  if (priceStr.length() >= n && iEquals(priceStr.substr(0, n), s)) {
137  return priceRef;
138  }
139  }
140 
141  // TODO: error handling throw IllegalArgumentException("Unknown candle price: " + s);
142 
143  return std::nullopt;
144  }
145 
146  /**
147  * Returns candle price type of the given candle symbol string.
148  * The result is CandlePrice::DEFAULT if the symbol does not have candle price type attribute or std::nullopt if there is no
149  * supported attribute's value.
150  *
151  * @param symbol The candle symbol string.
152  * @return candle price type of the given candle symbol string or std::nullopt if there is no supported attribute's
153  * value.
154  */
155  static std::optional<std::reference_wrapper<const CandlePrice>>
156  getAttributeForSymbol(const std::string &symbol) noexcept {
157  auto stringOpt = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
158 
159  return !stringOpt ? DEFAULT : parse(stringOpt.value());
160  }
161 
162  /**
163  * Returns candle symbol string with the normalized representation of the candle price type attribute.
164  *
165  * @param symbol candle symbol string.
166  * @return candle symbol string with the normalized representation of the the candle price type attribute.
167  */
168  static DXFCPP_CXX20_CONSTEXPR_STRING std::string normalizeAttributeForSymbol(const std::string &symbol) noexcept {
169  auto a = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
170 
171  if (!a) {
172  return symbol;
173  }
174 
175  auto other = parse(a.value());
176 
177  if (other.has_value()) {
178  if (other.value() == DEFAULT) {
179  return MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY);
180  }
181 
182  if (a.value() != other.value().get().toString()) {
183  return MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY,
184  other.value().get().toString());
185  }
186  }
187 
188  return symbol;
189  }
190 };

198 };
