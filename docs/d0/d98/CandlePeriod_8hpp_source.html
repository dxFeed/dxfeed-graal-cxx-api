19 /**
20  * Period attribute of CandleSymbol defines aggregation period of the candles.
21  * Aggregation period is defined as pair of a @ref CandlePeriod::getValue() "value" and @ref CandlePeriod::getType()
22  * "type".
23  *
24  * <h3>Implementation details</h3>
25  *
26  * This attribute is encoded in a symbol string with
27  * @ref MarketEventSymbols::getAttributeStringByKey() "MarketEventSymbols.getAttributeStringByKey",
28  * @ref MarketEventSymbols::changeAttributeStringByKey() "changeAttributeStringByKey", and
29  * @ref MarketEventSymbols::removeAttributeStringByKey() "removeAttributeStringByKey" methods.
30  * The key to use with these methods is available via CandlePeriod::ATTRIBUTE_KEY constant.
31  * The value that this key shall be set to is equal to the corresponding CandlePeriod::toString()
32  * "CandlePeriod.toString()"
33  */
34 struct DXFCPP_EXPORT CandlePeriod : public CandleSymbolAttribute {
35  /**
36  * Tick aggregation where each candle represents an individual tick.
37  */
38  static const CandlePeriod TICK;
39 
40  /**
41  * Day aggregation where each candle represents a day.
42  */
43  static const CandlePeriod DAY;
44 
45  /**
46  * Default period is CandlePeriod::TICK.
47  */
48  static const CandlePeriod DEFAULT;
49 
50  /**
51  * The attribute key that is used to store the value of `CandlePeriod` in a symbol string using methods of
52  * MarketEventSymbols class.
53  * The value of this constant is an empty string, because this is the main attribute that every CandleSymbol must
54  * have.
55  * The value that this key shall be set to is equal to the corresponding CandlePeriod::toString()
56  */
57  static const std::string ATTRIBUTE_KEY; // empty string as attribute key is allowed!
58 
59  private:
60  static const std::int64_t DEFAULT_PERIOD_VALUE = 1LL;
61 
62  double value_{};
63  CandleType type_{};
64  mutable std::string string_{};
65 
66  CandlePeriod(double value, const CandleType &type) noexcept : value_{value}, type_{type}, string_{} {
67  }
68 
69  public:
70  CandlePeriod() noexcept = default;
71  virtual ~CandlePeriod() noexcept = default;
72 
73  /**
74  * Returns aggregation period in milliseconds as closely as possible.
75  * Certain aggregation types like @ref CandleType::SECOND "SECOND" and
76  * @ref CandleType::DAY "DAY" span a specific number of milliseconds.
77  * CandleType::MONTH, CandleType::OPTEXP and CandleType::YEAR are approximate. Candle period of
78  * CandleType::TICK, CandleType::VOLUME, CandleType::PRICE, CandleType::PRICE_MOMENTUM and CandleType::PRICE_RENKO
79  * is not defined and this method returns `0`.
80  * The result of this method is equal to
81  * `static_cast<std::int64_t>(this->getType().getPeriodIntervalMillis() * this->getValue())`
82  * @see CandleType::getPeriodIntervalMillis()
83  * @return aggregation period in milliseconds.
84  */
85  std::int64_t getPeriodIntervalMillis() const noexcept {
86  return static_cast<std::int64_t>(static_cast<double>(type_.getPeriodIntervalMillis()) * value_);
87  }
88 
89  /**
90  * Returns candle event symbol string with this aggregation period set.
91  * @param symbol original candle event symbol.
92  * @return candle event symbol string with this aggregation period set.
93  */
94  std::string changeAttributeForSymbol(const std::string &symbol) const noexcept override {
95  return *this == DEFAULT ? MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY)
96  : MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY, toString());
97  }
98 
99  /**
100  * Returns aggregation period value. For example, the value of `5` with
101  * the candle type of @ref CandleType::MINUTE "MINUTE" represents `5` minute
102  * aggregation period.
103  *
104  * @return aggregation period value.
105  */
106  double getValue() const noexcept {
107  return value_;
108  }
109 
110  /**
111  * Returns aggregation period type.
112  * @return aggregation period type.
113  */
114  const CandleType &getType() const & noexcept {
115  return type_;
116  }
117 
118  /**
119  * Returns string representation of this aggregation period.
120  * The string representation is composed of value and type string.
121  * For example, 5 minute aggregation is represented as `"5m"`.
122  * The value of `1` is omitted in the string representation, so
123  * CandlePeriod::DAY (one day) is represented as `"d"`.
124  * This string representation can be converted back into object with CandlePeriod::parse() method.
125  *
126  * @return string representation of this aggregation period.
127  */
128  const std::string &toString() const & noexcept {
129  if (string_.empty()) {
130  string_ = math::equals(value_, DEFAULT_PERIOD_VALUE) ? type_.toString()
131  : math::equals(value_, static_cast<std::int64_t>(value_))
132  ? std::to_string(static_cast<std::int64_t>(value_)) + "" + type_.toString()
133  : std::to_string(value_) + "" + type_.toString();
134  }
135 
136  return string_;
137  }
138 
139  bool operator==(const CandlePeriod &candlePeriod) const noexcept {
140  return math::equals(value_, candlePeriod.getValue()) && type_ == candlePeriod.getType();
141  }
142 
143  /**
144  * Parses string representation of aggregation period into object.
145  * Any string that was returned by CandlePeriod::toString() can be parsed.
146  * This method is flexible in the way candle types can be specified.
147  * See CandleType::parse() for details.
148  *
149  * @param s The string representation of aggregation period.
150  * @return The aggregation period object or std::nullopt if parsing fails (if string representation is invalid)
151  */
152  static std::optional<CandlePeriod> parse(const std::string &s) noexcept {
153  if (s == CandleType::DAY.toString()) {
154  return DAY;
155  }
156 
157  if (s == CandleType::TICK.toString()) {
158  return TICK;
159  }
160 
161  std::size_t i = 0;
162  for (; i < s.length(); i++) {
163  auto c = s[i];
164 
165  if ((c < '0' || c > '9') && c != '.' && c != '-' && c != '+' && c != 'e' && c != 'E')
166  break;
167  }
168 
169  auto valueStr = s.substr(0, i);
170  auto typeStr = s.substr(i);
171 
172  try {
173  auto value = valueStr.empty() ? 1.0 : std::stod(valueStr);
174  auto typeOpt = CandleType::parse(typeStr);
175 
176  if (!typeOpt) {
177  return std::nullopt;
178  }
179 
180  return valueOf(value, typeOpt.value());
181  } catch (...) {
182  return std::nullopt;
183  }
184  }
185 
186  /**
187  * Returns candle period with the given value and type.
188  *
189  * @param value candle period value.
190  * @param type candle period type.
191  * @return candle period with the given value and type.
192  */
193  static CandlePeriod valueOf(double value, const CandleType &type) noexcept {
194  if (value == 1 && type == CandleType::DAY) {
195  return DAY;
196  }
197 
198  if (value == 1 && type == CandleType::TICK) {
199  return TICK;
200  }
201 
202  return {value, type};
203  }
204 
205  /**
206  * Returns candle period of the given candle symbol string.
207  * The result is CandlePeriod::DEFAULT if the symbol does not have candle period attribute.
208  *
209  * @param symbol candle symbol string.
210  * @return candle period of the given candle symbol string or std::nullopt if parsing fails (if string
211  * representation is invalid)
212  */
213  static std::optional<CandlePeriod> getAttributeForSymbol(const std::string &symbol) noexcept {
214  auto string = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
215 
216  return !string ? DEFAULT : parse(string.value());
217  }
218 
219  /**
220  * Returns candle symbol string with the normalized representation of the candle period attribute.
221  *
222  * @param symbol candle symbol string.
223  * @return candle symbol string with the normalized representation of the the candle period attribute.
224  */
225  static DXFCPP_CXX20_CONSTEXPR_STRING std::string normalizeAttributeForSymbol(const std::string &symbol) noexcept {
226  auto a = MarketEventSymbols::getAttributeStringByKey(symbol, ATTRIBUTE_KEY);
227 
228  if (!a) {
229  return symbol;
230  }
231 
232  auto other = parse(a.value());
233 
234  if (other.has_value()) {
235  if (other.value() == DEFAULT) {
236  return MarketEventSymbols::removeAttributeStringByKey(symbol, ATTRIBUTE_KEY);
237  }
238 
239  if (a.value() != other.value().toString()) {
240  return MarketEventSymbols::changeAttributeStringByKey(symbol, ATTRIBUTE_KEY, other.value().toString());
241  }
242  }
243 
244  return symbol;
245  }
246 };

255  dxfcpp::hashCombine(seed, candlePeriod.getType());
256 
257  return seed;
258  }
259 };
