14 /// Base abstract "shared entity" class. Has some helpers for dynamic polymorphism
15 struct DXFCPP_EXPORT SharedEntity : public Entity, std::enable_shared_from_this<SharedEntity> {
16  /// The simple type synonym for the SharedEntity type
17  using Ptr = std::shared_ptr<SharedEntity>;
18 
19  /**
20  * Checks that pointer to the current type could be converted to type T*
21  * In other words: whether type T belongs to the type hierarchy in which the current type resides.
22  * @tparam T The type to check
23  * @return true if type belongs to the type hierarchy in which the current type resides.
24  */
25  template <typename T> bool is() const noexcept {
26  try {
27  auto p = dynamic_cast<const T *>(this);
28  (void)(p);
29 
30  return true;
31  } catch (const std::bad_cast &) {
32  return false;
33  }
34  }
35 
36  /**
37  * Returns a pointer to the current object wrapped in a smart pointer to type T
38  *
39  * @tparam T The type to convert to a pointer to
40  * @return a smart pointer to type T
41  */
42  template <typename T> std::shared_ptr<T> sharedAs() noexcept {
43  return std::dynamic_pointer_cast<T>(shared_from_this());
44  }
45 
46  /**
47  * Returns a pointer to the current object wrapped in a smart pointer to type T
48  *
49  * @tparam T The type to convert to a pointer to
50  * @return a smart pointer to type T
51  */
52  template <typename T> std::shared_ptr<T> sharedAs() const noexcept {
53  return std::dynamic_pointer_cast<T>(shared_from_this());
54  }
55 
56  /**
57  * Returns a string representation of the current object.
58  *
59  * @return a string representation
60  */
61  virtual std::string toString() const noexcept {
62  return "SharedEntity{}";
63  }
64 };
