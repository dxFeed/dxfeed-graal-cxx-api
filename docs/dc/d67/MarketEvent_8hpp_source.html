17 /**
18  * Base class for all market events. All market events are POCO that
19  * extend this class. Market event classes are simple beans with setter and getter methods for their
20  * properties and minimal business logic. All market events have `eventSymbol` property that is
21  * defined by this class.
22  */
23 struct DXFCPP_EXPORT MarketEvent : public EventTypeWithSymbol<std::string> {
24  /// The alias to a type of shared pointer to the MarketEvent object
25  using Ptr = std::shared_ptr<MarketEvent>;
26 
27  private:
28  std::optional<std::string> eventSymbol_{};
29  std::int64_t eventTime_{};
30 
31  protected:
32  MarketEvent() noexcept = default;
33 
34  /**
35  * Protected constructor for concrete implementation classes that initializes `eventSymbol` property.
36  *
37  * @param eventSymbol The event symbol.
38  */
39  explicit MarketEvent(std::string eventSymbol) noexcept : eventSymbol_{std::move(eventSymbol)} {
40  }
41 
42  virtual void fillData(void *graalNative) noexcept;
43  virtual void fillGraalData(void *graalNative) const noexcept;
44  static void freeGraalData(void *graalNative) noexcept;
45 
46  public:
47  /**
48  * Returns symbol of this event.
49  *
50  * @return symbol of this event or dxfcpp::String::NUL (`std::string{"<null>"}`)
51  */
52  const std::string &getEventSymbol() const & noexcept override {
53  if (!eventSymbol_) {
54  return dxfcpp::String::NUL;
55  }
56 
57  return eventSymbol_.value();
58  }
59 
60  /**
61  * Returns symbol of this event.
62  *
63  * @return symbol of this event or `std::nullopt`.
64  */
65  const std::optional<std::string> &getEventSymbolOpt() const & noexcept override {
66  return eventSymbol_;
67  }
68 
69  /**
70  * Changes symbol of this event.
71  *
72  * @param eventSymbol The symbol of this event.
73  */
74  void setEventSymbol(const std::string &eventSymbol) noexcept override {
75  // TODO: check invalid utf-8 [EN-8233]
76  eventSymbol_ = eventSymbol;
77  }
78 
79  ///
80  std::int64_t getEventTime() const noexcept override {
81  return eventTime_;
82  }
83 
84  ///
85  void setEventTime(std::int64_t eventTime) noexcept override {
86  eventTime_ = eventTime;
87  }
88 };
