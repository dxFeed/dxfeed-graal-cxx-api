17 template <typename Child, typename Code> struct DXFCPP_EXPORT Enum {
18  using ParentType = Enum<Child, Code>;
19  using ChildType = Child;
20  using CodeType = Code;
21 
22  private:
23  const CodeType code_;
24  const std::string name_;
25 
26  protected:
27  template <Integral OtherCodeType> static constexpr CodeType convertToInnerCodeType(OtherCodeType code) noexcept {
28  return static_cast<Code>(static_cast<std::make_unsigned_t<Code>>(code));
29  }
30 
31  template <Integral OtherCodeType>
32  Enum(OtherCodeType code, std::string name) noexcept : code_{convertToInnerCodeType(code)}, name_{std::move(name)} {
33  }
34 
35  static const std::unordered_map<CodeType, std::reference_wrapper<const Child>> ALL;
36 
37  public:
38  /**
39  * Returns code that is used in flag bits.
40  *
41  * @return code.
42  */
43  [[nodiscard]] Code getCode() const {
44  return code_;
45  }
46 
47  /**
48  * Returns the enum element name
49  *
50  * @return name
51  */
52  [[nodiscard]] const std::string &getName() const & noexcept {
53  return name_;
54  }
55 
56  /**
57  * Returns a string representation of an enum element.
58  *
59  * @return a string representation
60  */
61  [[nodiscard]] const std::string &toString() const & noexcept {
62  return name_;
63  }
64 
65  /**
66  * Compares two elements of an enum.
67  *
68  * @tparam E Type convertible to enum `Enum<Child, Code>`
69  * @param e1 The first element of the enum to be compared
70  * @param e2 The second element of the enum to be compared
71  *
72  * @return `true` if the elements are the same
73  */
74  template <std::convertible_to<Enum<Child, Code>> E> friend bool operator==(const E &e1, const E &e2) {
75  return e1.getCode() == e2.getCode();
76  }
77 
78  /**
79  * Puts the string representation of an enum element to the output stream.
80  *
81  * @tparam OStream A type of the output stream
82  * @tparam E Type convertible to enum `Enum<Child, Code>`
83  * @param os The output stream
84  * @param e The element of the enum
85  *
86  * @return The output stream
87  */
88  template <typename OStream, std::convertible_to<Enum<Child, Code>> E>
89  friend OStream &operator<<(OStream &os, const E &e) {
90  return os << e.toString();
91  }
92 
93  /**
94  * Returns enum element by integer code bit pattern.
95  *
96  * @param code integer code.
97  * @return enum element.
98  */
99  template <Integral OtherCodeType> inline static const Child &valueOf(OtherCodeType code) {
100  if (auto found = ALL.find(convertToInnerCodeType(code)); found != ALL.end()) {
101  return found->second;
102  }
103 
104  // TODO: try to implement C++11-like code for this
105  if constexpr (requires { Child::getDefault(); }) {
106  return Child::getDefault();
107  } else {
108  return ALL.at(0);
109  }
110  }
111 };
