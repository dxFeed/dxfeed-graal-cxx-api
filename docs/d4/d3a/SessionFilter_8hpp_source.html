12 /**
13  * A filter for sessions used by various search methods.
14  * This class provides predefined filters for certain Session attributes,
15  * although users can create their own filters to suit their needs.
16  * <p>
17  * Please note that sessions can be either trading or non-trading, and this distinction can be
18  * either based on rules (e.g. weekends) or dictated by special occasions (e.g. holidays).
19  * Different filters treat this distinction differently - some accept only trading sessions,
20  * some only non-trading, and some ignore type of session altogether.
21  */
22 struct DXFCPP_EXPORT SessionFilter {
23  /** Accepts any session - useful for pure schedule navigation. */
24  static const SessionFilter ANY;
25  /** Accepts trading sessions only - those with <code>(Session::isTrading() == true)</code>. */
26  static const SessionFilter TRADING;
27  /** Accepts non-trading sessions only - those with <code>(Session::isTrading() == false)</code>. */
28  static const SessionFilter NON_TRADING;
29 
30  /** Accepts any session with type SessionType::NO_TRADING. */
31  static const SessionFilter NO_TRADING;
32  /** Accepts any session with type SessionType::PRE_MARKET. */
33  static const SessionFilter PRE_MARKET;
34  /** Accepts any session with type SessionType::REGULAR. */
35  static const SessionFilter REGULAR;
36  /** Accepts any session with type SessionType::AFTER_MARKET. */
37  static const SessionFilter AFTER_MARKET;
38 
39  protected:
40  /// Required type, std::nullopt if not relevant.
41  std::optional<SessionType> type_;
42  /// Required trading flag, std::nullopt if not relevant.
43  std::optional<bool> trading_;
44 
45  public:
46  SessionFilter() noexcept = default;
47  virtual ~SessionFilter() noexcept = default;
48 
49  /**
50  * Creates filter with specified type and trading flag conditions.
51  * <p>
52  * Both parameters specify what value corresponding attributes should have.
53  * If some parameter is std::nullopt then corresponding attribute is ignored (any value is accepted).
54  *
55  * @param type required type, std::nullopt if not relevant
56  * @param trading required trading flag, std::nullopt if not relevant
57  */
58  SessionFilter(std::optional<SessionType> type, std::optional<bool> trading) noexcept
59  : type_{type}, trading_{trading} {
60  }
61 
62  /**
63  * Tests whether or not the specified session is an acceptable result.
64  *
65  * @param session the session to be tested
66  * @return `true` if specified session is accepted
67  */
68  template <typename Session> bool accept(Session session) const noexcept {
69  return (!type_ || type_.value() == session.getType()) && (!trading_ || trading_.value() == session.isTrading());
70  }
71 
72  bool operator==(const SessionFilter &sessionFilter) const noexcept {
73  return type_ == sessionFilter.type_ && trading_ == sessionFilter.trading_;
74  }
75 
76  std::string toString() const noexcept {
77  return std::string("SessionFilter(") + ((!type_) ? "null" : type_.value().toString()) + ", " +
78  ((!trading_) ? "null" : dxfcpp::toString(trading_.value())) + ")";
79  }
80  const std::optional<SessionType> &getType() const & noexcept {
81  return type_;
82  }
83 
84  const std::optional<bool> &getTrading() const & noexcept {
85  return trading_;
86  }
87 };
88 
89 } // namespace dxfcpp
90 
91 template <> struct std::hash<dxfcpp::SessionFilter> {
92  std::size_t operator()(const dxfcpp::SessionFilter &sessionFilter) const noexcept {
93  std::size_t hash = 0;
94 
95  if (sessionFilter.getType()) {
96  dxfcpp::hashCombine(hash, sessionFilter.getType().value());
97  }
98 
99  if (sessionFilter.getTrading()) {
100  dxfcpp::hashCombine(hash, sessionFilter.getTrading().value());
101  }
102 
103  return hash;
104  }
105 };
